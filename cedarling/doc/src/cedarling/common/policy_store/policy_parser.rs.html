6//! Cedar policy and template parsing and validation.
7//!
8//! This module handles parsing Cedar policy files (.cedar) and extracting
9//! policy IDs from @id() annotations. It provides validation and error
10//! reporting with file names and line numbers.
11
12use cedar_policy::{Policy, PolicyId, PolicySet, Template};
13#[cfg(test)]
14use std::collections::HashMap;
15
16use super::errors::{CedarParseErrorDetail, PolicyStoreError, ValidationError};

18/// Represents a parsed Cedar policy with metadata.
19#[derive(Debug, Clone)]
20pub struct ParsedPolicy {
21    /// The policy ID (from Cedar engine or @id annotation)
22    pub id: PolicyId,
23    /// The original filename
24    pub filename: String,
25    /// The parsed Cedar policy
26    pub policy: Policy,
27}
28
29/// Represents a parsed Cedar template with metadata.
30#[derive(Debug, Clone)]
31pub struct ParsedTemplate {
32    /// The original filename
33    pub filename: String,
34    /// The parsed Cedar template
35    pub template: Template,
36}
37
38/// Cedar policy and template parser.
39///
40/// Provides methods for parsing Cedar policies and templates from text,
41/// extracting @id() annotations, and validating syntax.
42pub struct PolicyParser;
43
44impl PolicyParser {
45    /// Parse a single policy from Cedar policy text.
46    ///
47    /// The policy ID is determined by:
48    /// 1. Extracting from @id() annotation in the policy text, OR
49    /// 2. Deriving from the filename (without .cedar extension)
50    ///
51    /// Pass the ID to `Policy::parse()` using the annotation or the filename (without
52    /// the .cedar extension).
53    pub fn parse_policy(content: &str, filename: &str) -> Result<ParsedPolicy, PolicyStoreError> {
54        // Extract policy ID from @id() annotation or derive from filename
55        let policy_id_str = Self::extract_id_annotation(content)
56            .or_else(|| Self::derive_id_from_filename(filename));
57
58        let policy_id = match policy_id_str {
59            Some(id_str) => {
60                // Validate the ID format
61                Self::validate_policy_id(&id_str, filename)
62                    .map_err(PolicyStoreError::Validation)?;
63                PolicyId::new(&id_str)
64            },
65            None => {
66                return Err(PolicyStoreError::CedarParsing {
67                    file: filename.to_string(),
68                    detail: CedarParseErrorDetail::MissingIdAnnotation,
69                });
70            },
71        };
72
73        // Parse the policy using Cedar engine with the policy ID
74        let policy = Policy::parse(Some(policy_id.clone()), content).map_err(|e| {
75            PolicyStoreError::CedarParsing {
76                file: filename.to_string(),
77                detail: CedarParseErrorDetail::ParseError(e.to_string()),
78            }
79        })?;
80
81        Ok(ParsedPolicy {
82            id: policy_id,
83            filename: filename.to_string(),
84            policy,
85        })
86    }
87
88    /// Parse a single template from Cedar policy text.
89    ///
90    /// Templates support slots (e.g., ?principal) and are parsed similarly to policies.
91    /// The template ID is extracted from @id() annotation or derived from filename.
92    ///
93    /// the ID to `Template::parse()` based on annotation or filename.
94    pub fn parse_template(
95        content: &str,
96        filename: &str,
97    ) -> Result<ParsedTemplate, PolicyStoreError> {
98        // Extract template ID from @id() annotation or derive from filename
99        let template_id_str = Self::extract_id_annotation(content)
100            .or_else(|| Self::derive_id_from_filename(filename));
101
102        let template_id = match template_id_str {
103            Some(id_str) => {
104                // Validate the ID format
105                Self::validate_policy_id(&id_str, filename)
106                    .map_err(PolicyStoreError::Validation)?;
107                PolicyId::new(&id_str)
108            },
109            None => {
110                return Err(PolicyStoreError::CedarParsing {
111                    file: filename.to_string(),
112                    detail: CedarParseErrorDetail::MissingIdAnnotation,
113                });
114            },
115        };
116
117        // Parse the template using Cedar engine with the template ID
118        let template = Template::parse(Some(template_id.clone()), content).map_err(|e| {
119            PolicyStoreError::CedarParsing {
120                file: filename.to_string(),
121                detail: CedarParseErrorDetail::ParseError(e.to_string()),
122            }
123        })?;
124
125        Ok(ParsedTemplate {
126            filename: filename.to_string(),
127            template,
128        })
129    }
130
131    /// Parse multiple policies and return a map of policy ID to filename.
132    ///
133    /// Useful for batch processing of policy files in tests and tooling.
134    #[cfg(test)]
135    pub fn parse_policies<'a, I>(
136        policy_files: I,
137    ) -> Result<HashMap<PolicyId, String>, PolicyStoreError>
138    where
139        I: IntoIterator<Item = (&'a str, &'a str)>,
140    {
141        let policy_files_vec: Vec<_> = policy_files.into_iter().collect();
142        let mut policy_map = HashMap::with_capacity(policy_files_vec.len());
143
144        for (filename, content) in policy_files_vec {
145            let parsed = Self::parse_policy(content, filename)?;
146            policy_map.insert(parsed.id, parsed.filename);
147        }
148
149        Ok(policy_map)
150    }

152    /// Create a PolicySet from parsed policies and templates.
153    ///
154    /// Validates that all policies and templates can be successfully added
155    /// to the policy set, ensuring no ID conflicts or other issues.
156    pub fn create_policy_set(
157        policies: Vec<ParsedPolicy>,
158        templates: Vec<ParsedTemplate>,
159    ) -> Result<PolicySet, PolicyStoreError> {
160        let mut policy_set = PolicySet::new();
161
162        // Add all policies
163        for parsed in policies {
164            policy_set
165                .add(parsed.policy)
166                .map_err(|e| PolicyStoreError::CedarParsing {
167                    file: parsed.filename,
168                    detail: CedarParseErrorDetail::AddPolicyFailed(e.to_string()),
169                })?;
170        }
171
172        // Add all templates
173        for parsed in templates {
174            policy_set.add_template(parsed.template).map_err(|e| {
175                PolicyStoreError::CedarParsing {
176                    file: parsed.filename,
177                    detail: CedarParseErrorDetail::AddTemplateFailed(e.to_string()),
178                }
179            })?;
180        }
181
182        Ok(policy_set)
183    }

185    /// Derive a policy ID from a filename.
186    ///
187    /// Removes the .cedar extension, sanitizes characters, and returns the ID.
188    /// Returns None if the filename is empty or invalid.
189    pub fn derive_id_from_filename(filename: &str) -> Option<String> {
190        // Extract just the filename without path
191        let base_name = filename.rsplit('/').next().unwrap_or(filename);
192
193        // Remove .cedar extension
194        let without_ext = base_name.strip_suffix(".cedar").unwrap_or(base_name);
195
196        // If empty after stripping, return None
197        if without_ext.is_empty() {
198            return None;
199        }
200
201        // Replace invalid characters with underscores
202        let sanitized: String = without_ext
203            .chars()
204            .map(|c| {
205                if c.is_alphanumeric() || c == '_' || c == '-' || c == ':' {
206                    c
207                } else {
208                    '_'
209                }
210            })
211            .collect();
212
213        Some(sanitized)
214    }

216    /// Extract @id() annotation from Cedar policy text.
217    ///
218    /// Looks for @id("...") or @id('...') pattern in comments.
219    pub fn extract_id_annotation(content: &str) -> Option<String> {
220        // Look for @id("...") or @id('...') pattern
221        for line in content.lines() {
222            let trimmed = line.trim();
223            if let Some(start_idx) = trimmed.find("@id(") {
224                let after_id = &trimmed[start_idx + 4..];
225                // Find the string content between quotes
226                if let Some(open_quote) = after_id.find('"').or_else(|| after_id.find('\'')) {
227                    let quote_char = after_id.chars().nth(open_quote).unwrap();
228                    let after_open = &after_id[open_quote + 1..];
229                    if let Some(close_quote) = after_open.find(quote_char) {
230                        return Some(after_open[..close_quote].to_string());
231                    }
232                }
233            }
234        }
235        None
236    }

238    /// Validate policy ID format (alphanumeric, underscore, hyphen, colon only).
239    pub fn validate_policy_id(id: &str, filename: &str) -> Result<(), ValidationError> {
240        if id.is_empty() {
241            return Err(ValidationError::EmptyPolicyId {
242                file: filename.to_string(),
243            });
244        }
245
246        // Check for valid characters (alphanumeric, underscore, hyphen, colon)
247        if !id
248            .chars()
249            .all(|c| c.is_alphanumeric() || c == '_' || c == '-' || c == ':')
250        {
251            return Err(ValidationError::InvalidPolicyIdCharacters {
252                file: filename.to_string(),
253                id: id.to_string(),
254            });
255        }
256
257        Ok(())
258    }
