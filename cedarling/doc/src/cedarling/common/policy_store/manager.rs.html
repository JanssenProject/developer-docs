6//! Policy Store Manager - Converts new format to legacy format.
7//!
8//! This module provides the conversion layer between `LoadedPolicyStore` (new directory/archive format)
9//! and `PolicyStore` (legacy format used by the rest of Cedarling).
10//!
11//! # Architecture
12//!
13//! ```text
14//! LoadedPolicyStore (new)          PolicyStore (legacy)
15//! ├── metadata                  →  name, version, description, cedar_version
16//! ├── schema (raw string)       →  schema: CedarSchema
17//! ├── policies: Vec<PolicyFile> →  policies: PoliciesContainer
18//! ├── trusted_issuers           →  trusted_issuers: HashMap<String, TrustedIssuer>
19//! └── entities                  →  default_entities: HashMap<String, Value>
20//! ```

39/// Errors that can occur during policy store conversion.
40#[derive(Debug, thiserror::Error)]
41pub enum ConversionError {
42    /// Schema conversion failed
43    #[error("Failed to convert schema: {0}")]
44    SchemaConversion(String),
45
46    /// Policy conversion failed
47    #[error("Failed to convert policies: {0}")]
48    PolicyConversion(String),
49
50    /// Trusted issuer conversion failed
51    #[error("Failed to convert trusted issuers: {0}")]
52    IssuerConversion(String),
53
54    /// Entity conversion failed
55    #[error("Failed to convert entities: {0}")]
56    EntityConversion(String),
57
58    /// Version parsing failed
59    #[error("Failed to parse cedar version '{version}': {details}")]
60    VersionParsing { version: String, details: String },
61
62    /// Policy set creation failed
63    #[error("Failed to create policy set: {0}")]
64    PolicySetCreation(String),
65}

70impl PolicyStoreManager {
71    /// Convert a `LoadedPolicyStore` (new format) to `PolicyStore` (legacy format).
72    ///
73    /// This is the main entry point for converting policy stores loaded from
74    /// directory or archive format into the legacy format used by the rest of Cedarling.
75    ///
76    /// # Arguments
77    ///
78    /// * `loaded` - The loaded policy store from the new loader
79    ///
80    /// # Returns
81    ///
82    /// Returns a `PolicyStore` that can be used with existing Cedarling services.
83    ///
84    /// # Errors
85    ///
86    /// Returns `ConversionError` if any component fails to convert.
87    pub fn convert_to_legacy(loaded: LoadedPolicyStore) -> Result<PolicyStore, ConversionError> {
88        Self::convert_to_legacy_with_logger(loaded, None)
89    }

91    /// Convert a `LoadedPolicyStore` to `PolicyStore` with optional logging.
92    ///
93    /// This version accepts an optional logger for structured logging during conversion.
94    /// Use this when a logger is available to get detailed conversion logs.
95    ///
96    /// # Arguments
97    ///
98    /// * `loaded` - The loaded policy store from the new loader
99    /// * `logger` - Optional logger for structured logging
100    ///
101    /// # Returns
102    ///
103    /// Returns a `PolicyStore` that can be used with existing Cedarling services.
104    ///
105    /// # Errors
106    ///
107    /// Returns `ConversionError` if any component fails to convert.
108    pub fn convert_to_legacy_with_logger(
109        loaded: LoadedPolicyStore,
110        logger: Option<Logger>,
111    ) -> Result<PolicyStore, ConversionError> {
112        // Log manifest info if available
113        if let Some(manifest) = &loaded.manifest {
114            logger.log_any(PolicyStoreLogEntry::info(format!(
115                "Converting policy store '{}' (generated: {})",
116                manifest.policy_store_id, manifest.generated_date
117            )));
118        }
119
120        // 1. Convert schema
121        let cedar_schema = Self::convert_schema(&loaded.schema)?;
122
123        // 2. Convert policies and templates into a single PoliciesContainer
124        let policies_container =
125            Self::convert_policies_and_templates(&loaded.policies, &loaded.templates)?;
126
127        // 3. Convert trusted issuers
128        let trusted_issuers = Self::convert_trusted_issuers(&loaded.trusted_issuers)?;
129
130        // 4. Convert entities (logs hierarchy warnings if logger provided)
131        let raw_entities = Self::convert_entities(&loaded.entities, &logger)?;
132
133        // Convert raw entities to DefaultEntitiesWithWarns
134        let default_entities = parse_default_entities_with_warns(raw_entities).map_err(|e| {
135            ConversionError::EntityConversion(format!("Failed to parse default entities: {}", e))
136        })?;
137
138        // 5. Parse cedar version
139        let cedar_version = Self::parse_cedar_version(&loaded.metadata.cedar_version)?;
140
141        logger.log_any(PolicyStoreLogEntry::info(format!(
142            "Policy store conversion complete: {} policies, {} issuers, {} entities",
143            policies_container.get_set().policies().count(),
144            trusted_issuers.as_ref().map(|i| i.len()).unwrap_or(0),
145            default_entities.entities().len()
146        )));
147
148        Ok(PolicyStore {
149            name: loaded.metadata.policy_store.name,
150            version: Some(loaded.metadata.policy_store.version),
151            description: loaded.metadata.policy_store.description,
152            cedar_version: Some(cedar_version),
153            schema: cedar_schema,
154            policies: policies_container,
155            trusted_issuers,
156            default_entities,
157        })
158    }

160    /// Convert raw schema string to `CedarSchema`.
161    ///
162    /// Uses `ParsedSchema::parse` to parse and validate the schema, then converts
163    /// to the `CedarSchema` format required by the legacy system.
164    ///
165    /// The `CedarSchema` requires:
166    /// - `schema: cedar_policy::Schema`
167    /// - `json: CedarSchemaJson`
168    /// - `validator_schema: ValidatorSchema`
169    fn convert_schema(schema_content: &str) -> Result<CedarSchema, ConversionError> {
170        use super::schema_parser::ParsedSchema;
171        use cedar_policy::SchemaFragment;
172        use std::str::FromStr;
173
174        // Parse and validate schema
175        let parsed_schema =
176            ParsedSchema::parse(schema_content, "schema.cedarschema").map_err(|e| {
177                ConversionError::SchemaConversion(format!("Failed to parse schema: {}", e))
178            })?;
179
180        // Validate the schema
181        parsed_schema.validate().map_err(|e| {
182            ConversionError::SchemaConversion(format!("Schema validation failed: {}", e))
183        })?;
184
185        // Get the Cedar schema from the parsed result
186        let schema = parsed_schema.get_schema().clone();
187
188        // Convert to JSON for CedarSchemaJson and ValidatorSchema
189        // NOTE: This parses the schema content again (SchemaFragment::from_str).
190        // For large schemas, this double-parsing could be optimized by having
191        // ParsedSchema return both the validated schema and the fragment, but
192        // this is a performance consideration rather than a correctness issue.
193        let fragment = SchemaFragment::from_str(schema_content).map_err(|e| {
194            ConversionError::SchemaConversion(format!("Failed to parse schema fragment: {}", e))
195        })?;
196
197        let json_string = fragment.to_json_string().map_err(|e| {
198            ConversionError::SchemaConversion(format!("Failed to serialize schema to JSON: {}", e))
199        })?;
200
201        // Parse CedarSchemaJson
202        let json: CedarSchemaJson = serde_json::from_str(&json_string).map_err(|e| {
203            ConversionError::SchemaConversion(format!("Failed to parse CedarSchemaJson: {}", e))
204        })?;
205
206        // Create ValidatorSchema
207        let validator_schema = ValidatorSchema::from_json_str(
208            &json_string,
209            Extensions::all_available(),
210        )
211        .map_err(|e| {
212            ConversionError::SchemaConversion(format!("Failed to create ValidatorSchema: {}", e))
213        })?;
214
215        Ok(CedarSchema {
216            schema,
217            json,
218            validator_schema,
219        })
220    }

222    /// Convert policy and template files to `PoliciesContainer`.
223    ///
224    /// The `PoliciesContainer` requires:
225    /// - `policy_set: cedar_policy::PolicySet` (includes both policies and templates)
226    /// - `raw_policy_info: HashMap<String, RawPolicy>` (for descriptions)
227    fn convert_policies_and_templates(
228        policy_files: &[super::loader::PolicyFile],
229        template_files: &[super::loader::PolicyFile],
230    ) -> Result<PoliciesContainer, ConversionError> {
231        if policy_files.is_empty() && template_files.is_empty() {
232            // Return empty policy set
233            let policy_set = PolicySet::new();
234            return Ok(PoliciesContainer::new_empty(policy_set));
235        }
236
237        // Parse each policy file
238        let mut parsed_policies = Vec::with_capacity(policy_files.len());
239        for file in policy_files {
240            let parsed = PolicyParser::parse_policy(&file.content, &file.name).map_err(|e| {
241                ConversionError::PolicyConversion(format!("Failed to parse '{}': {}", file.name, e))
242            })?;
243            parsed_policies.push(parsed);
244        }
245
246        // Parse each template file
247        let mut parsed_templates = Vec::with_capacity(template_files.len());
248        for file in template_files {
249            let parsed = PolicyParser::parse_template(&file.content, &file.name).map_err(|e| {
250                ConversionError::PolicyConversion(format!(
251                    "Failed to parse template '{}': {}",
252                    file.name, e
253                ))
254            })?;
255            parsed_templates.push(parsed);
256        }
257
258        // Create policy set using PolicyParser (includes both policies and templates)
259        let policy_set =
260            PolicyParser::create_policy_set(parsed_policies.clone(), parsed_templates.clone())
261                .map_err(|e| ConversionError::PolicySetCreation(e.to_string()))?;
262
263        // Build raw_policy_info for descriptions (policies only, templates don't have descriptions in legacy format)
264        let raw_policy_info = parsed_policies
265            .into_iter()
266            .map(|p| (p.id.to_string(), format!("Policy from {}", p.filename)))
267            .collect();
268
269        Ok(PoliciesContainer::new(policy_set, raw_policy_info))
270    }

272    /// Convert issuer files to `HashMap<String, TrustedIssuer>`.
273    fn convert_trusted_issuers(
274        issuer_files: &[super::loader::IssuerFile],
275    ) -> Result<Option<HashMap<String, TrustedIssuer>>, ConversionError> {
276        if issuer_files.is_empty() {
277            return Ok(None);
278        }
279
280        let mut all_issuers = Vec::new();
281        for file in issuer_files {
282            let parsed = IssuerParser::parse_issuer(&file.content, &file.name).map_err(|e| {
283                ConversionError::IssuerConversion(format!("Failed to parse '{}': {}", file.name, e))
284            })?;
285            all_issuers.extend(parsed);
286        }
287
288        // Validate for duplicates - include content in error for debugging
289        if let Err(errors) = IssuerParser::validate_issuers(&all_issuers) {
290            // Return validation errors directly, joined into a single string
291            let error_details = errors
292                .iter()
293                .map(|e| e.to_string())
294                .collect::<Vec<_>>()
295                .join("; ");
296            return Err(ConversionError::IssuerConversion(error_details));
297        }
298
299        // Create issuer map
300        let issuer_map = IssuerParser::create_issuer_map(all_issuers)
301            .map_err(|e| ConversionError::IssuerConversion(e.to_string()))?;
302
303        Ok(Some(issuer_map))
304    }

306    /// Convert entity files to `HashMap<String, serde_json::Value>`.
307    ///
308    /// This function:
309    /// 1. Parses all entity files
310    /// 2. Detects duplicate entity UIDs (returns error if found)
311    /// 3. Optionally validates entity hierarchy (parent references - logs warnings if logger provided)
312    /// 4. Converts to the required HashMap format
313    ///
314    /// # Arguments
315    ///
316    /// * `entity_files` - The entity files to convert
317    /// * `logger` - Optional logger for hierarchy warnings
318    fn convert_entities(
319        entity_files: &[super::loader::EntityFile],
320        logger: &Option<Logger>,
321    ) -> Result<Option<HashMap<String, serde_json::Value>>, ConversionError> {
322        if entity_files.is_empty() {
323            return Ok(None);
324        }
325
326        // Step 1: Parse all entity files
327        let mut all_parsed_entities = Vec::new();
328        for file in entity_files {
329            let parsed =
330                EntityParser::parse_entities(&file.content, &file.name, None).map_err(|e| {
331                    ConversionError::EntityConversion(format!(
332                        "Failed to parse '{}': {}",
333                        file.name, e
334                    ))
335                })?;
336            all_parsed_entities.extend(parsed);
337        }
338
339        // Step 2: Detect duplicate entity UIDs (warns but doesn't fail on duplicates)
340        // Note: We clone all_parsed_entities here because EntityParser::detect_duplicates
341        // takes ownership of the Vec. This preserves the original for later hierarchy validation.
342        // Duplicates are handled gracefully - the latest entity wins and a warning is logged.
343        let unique_entities = EntityParser::detect_duplicates(all_parsed_entities.clone(), logger);
344
345        // Step 3: Validate entity hierarchy (optional - parent entities may be provided at runtime)
346        // This ensures all parent references point to entities that exist in this store
347        // Note: Hierarchy validation errors are non-fatal since parent entities
348        // might be provided at runtime via authorization requests
349        if let Err(warnings) = EntityParser::validate_hierarchy(&all_parsed_entities) {
350            logger.log_any(PolicyStoreLogEntry::warn(format!(
351                "Entity hierarchy validation warnings (non-fatal): {:?}",
352                warnings
353            )));
354        }
355
356        // Step 4: Validate entities can form a valid Cedar entity store
357        // This validates entity constraints like types and attribute compatibility
358        EntityParser::create_entities_store(all_parsed_entities).map_err(|e| {
359            ConversionError::EntityConversion(format!("Failed to create entity store: {}", e))
360        })?;
361
362        // Step 5: Convert to HashMap<String, Value>
363        let mut result = HashMap::with_capacity(unique_entities.len());
364        for (uid, parsed_entity) in unique_entities {
365            let json_value = parsed_entity.entity.to_json_value().map_err(|e| {
366                // Include the original content in the error for debugging
367                ConversionError::EntityConversion(format!(
368                    "Failed to serialize entity '{}' from '{}': {}. Original content: {}",
369                    uid,
370                    parsed_entity.filename,
371                    e,
372                    if parsed_entity.content.len() > 200 {
373                        format!("{}...(truncated)", &parsed_entity.content[..200])
374                    } else {
375                        parsed_entity.content.clone()
376                    }
377                ))
378            })?;
379            result.insert(uid.to_string(), json_value);
380        }
381
382        Ok(Some(result))
383    }

385    /// Parse cedar version string to `semver::Version`.
386    fn parse_cedar_version(version_str: &str) -> Result<Version, ConversionError> {
387        // Handle optional "v" prefix
388        let version_str = version_str.strip_prefix('v').unwrap_or(version_str);
389
390        Version::parse(version_str).map_err(|e| ConversionError::VersionParsing {
391            version: version_str.to_string(),
392            details: e.to_string(),
393        })
394    }
