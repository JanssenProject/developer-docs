6use std::collections::HashMap;
7
8use crate::log::Decision;
9use datalogic_rs::JsonLogic;
10use smol_str::SmolStr;
11
12use super::JsonRule;
13
14pub(crate) struct RuleApplier<'a> {
15    rule: &'a JsonRule,
16    data: serde_json::Value,
17}
18
19impl<'a> RuleApplier<'a> {
20    pub fn new(rule: &'a JsonRule, data: HashMap<SmolStr, Decision>) -> Self {
21        Self {
22            rule,
23            data: serde_json::json!(data),
24        }
25    }
26
27    pub fn apply(&self) -> Result<bool, ApplyRuleError> {
28        let result = JsonLogic::apply(self.rule.rule(), &self.data)?;
29
30        if let serde_json::Value::Bool(b) = result {
31            Ok(b)
32        } else {
33            Err(ApplyRuleError::from(datalogic_rs::Error::Custom(
34                "Result is not a boolean.".to_string(),
35            )))
36        }
37    }
38}
39
40#[derive(Debug, derive_more::Display, derive_more::Error)]
41#[display("Could not apply json logic rule due to error: {source}")]
42pub struct ApplyRuleError {
43    source: datalogic_rs::Error,
44}
45
46impl From<datalogic_rs::Error> for ApplyRuleError {
47    fn from(source: datalogic_rs::Error) -> Self {
48        Self { source }
49    }
50}
