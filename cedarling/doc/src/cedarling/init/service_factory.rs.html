8//! Module to lazily initialize internal cedarling services
9
10use super::service_config::ServiceConfig;
11use crate::LogLevel;
12use crate::authz::{Authz, AuthzConfig, AuthzServiceInitError};
13use crate::bootstrap_config::BootstrapConfig;
14use crate::common::policy_store::{
15    PolicyStoreMetadata, PolicyStoreWithID, TrustedIssuersValidationError,
16};
17use crate::entity_builder::*;
18use crate::jwt::{JwtService, JwtServiceInitError};
19use crate::log::interface::LogWriter;
20use crate::log::{self, BaseLogEntry, LogEntry};
21use std::sync::Arc;
22
23#[derive(Clone)]
24pub(crate) struct ServiceFactory<'a> {
25    bootstrap_config: &'a BootstrapConfig,
26    service_config: ServiceConfig,
27    log_service: log::Logger,
28    container: SingletonContainer,
29}
30
31/// Structure to store singleton of entities.
32#[derive(Clone, Default)]
33struct SingletonContainer {
34    entity_builder_service: Option<Arc<EntityBuilder>>,
35    jwt_service: Option<Arc<JwtService>>,
36    authz_service: Option<Arc<Authz>>,
37}
38
39impl<'a> ServiceFactory<'a> {
40    /// Create new instance of ServiceFactory.
41    pub fn new(
42        bootstrap_config: &'a BootstrapConfig,
43        service_config: ServiceConfig,
44        log_service: log::Logger,
45    ) -> Self {
46        Self {
47            bootstrap_config,
48            service_config,
49            log_service,
50            container: Default::default(),
51        }
52    }
53
54    // get policy store
55    pub fn policy_store(&self) -> Result<&PolicyStoreWithID, ServiceInitError> {
56        // it potentyally can be called many times, but it is only during initialization so it shouldn't be a problem
57        self.service_config
58            .policy_store
59            .validate_trusted_issuers()?;
60
61        Ok(&self.service_config.policy_store)
62    }
63
64    /// Get the policy store metadata if available.
65    ///
66    /// Metadata is only available when the policy store is loaded from the new
67    /// directory/archive format. Legacy JSON/YAML formats do not include metadata.
68    pub fn policy_store_metadata(&self) -> Option<&PolicyStoreMetadata> {
69        self.service_config.policy_store.metadata.as_ref()
70    }
71
72    // get log service
73    pub fn log_service(&mut self) -> log::Logger {
74        self.log_service.clone()
75    }
76
77    // get jwt service
78    pub async fn jwt_service(&mut self) -> Result<Arc<JwtService>, ServiceInitError> {
79        if let Some(jwt_service) = &self.container.jwt_service {
80            Ok(jwt_service.clone())
81        } else {
82            let config = &self.bootstrap_config.jwt_config;
83            let trusted_issuers = self.policy_store()?.trusted_issuers.clone();
84            let logger = self.log_service();
85            let service = Arc::new(JwtService::new(config, trusted_issuers, Some(logger)).await?);
86            self.container.jwt_service = Some(service.clone());
87            Ok(service)
88        }
89    }
90
91    // get jwt service
92    pub fn entity_builder(&mut self) -> Result<Arc<EntityBuilder>, ServiceInitError> {
93        if let Some(entity_builder) = &self.container.entity_builder_service {
94            return Ok(entity_builder.clone());
95        }
96
97        let logger = self.log_service();
98        let policy_store = &self.policy_store()?.store;
99
100        let default_entities_with_warn = &policy_store.default_entities;
101        // Log warns that some default entities loaded not correctly
102        // it will be logged only once.
103        for warn in default_entities_with_warn.warns() {
104            let log_entry = LogEntry::new(BaseLogEntry::new_system_opt_request_id(
105                LogLevel::WARN,
106                None,
107            ))
108            .set_message(warn.to_string());
109
110            logger.log_any(log_entry);
111        }
112
113        let config = &self.bootstrap_config.entity_builder_config;
114        let policy_store = self.policy_store()?;
115
116        let trusted_issuers = policy_store.trusted_issuers.clone().unwrap_or_default();
117        let issuers_index = TrustedIssuerIndex::new(&trusted_issuers, Some(logger));
118        let schema = &policy_store.schema.validator_schema;
119        let entity_builder = EntityBuilder::new(
120            config.clone(),
121            issuers_index,
122            Some(schema),
123            default_entities_with_warn.entities().to_owned(),
124        )?;
125        let service = Arc::new(entity_builder);
126        self.container.entity_builder_service = Some(service.clone());
127        Ok(service)
128    }
129
130    // get authz service
131    pub async fn authz_service(&mut self) -> Result<Arc<Authz>, ServiceInitError> {
132        if let Some(authz) = &self.container.authz_service {
133            Ok(authz.clone())
134        } else {
135            let config = AuthzConfig {
136                log_service: self.log_service(),
137                policy_store: self.policy_store()?.clone(),
138                jwt_service: self.jwt_service().await?,
139                entity_builder: self.entity_builder()?,
140                authorization: self.bootstrap_config.authorization_config.clone(),
141            };
142            let service = Arc::new(Authz::new(config)?);
143            self.container.authz_service = Some(service.clone());
144            Ok(service)
145        }
146    }
147}
148
149/// Error type for failing to initialize a service
150#[derive(Debug, thiserror::Error)]
151pub enum ServiceInitError {
152    #[error(transparent)]
153    AuthzService(#[from] AuthzServiceInitError),
154    #[error(transparent)]
155    JwtService(#[from] JwtServiceInitError),
156    #[error(transparent)]
157    EntityBuilder(#[from] InitEntityBuilderError),
158    #[error(transparent)]
159    PolicyStore(#[from] TrustedIssuersValidationError),
160}
