1use crate::log::LogLevel;

6/// Lock service config
7#[derive(Debug, Clone, PartialEq)]
8pub struct LockServiceConfig {
9    /// The logging level
10    pub log_level: LogLevel,
11    /// URI where Cedarling can get metadata about the lock server.
12    /// i.e. `.well-known/lock-server-configuration`
13    pub config_uri: Url,
14    /// Toggles whether Cedarling should listen for SSE config updates.
15    pub dynamic_config: bool,
16    /// Software Statement Assertion Json Web Token that Cedarling will use for Dynamic
17    /// Client Registration.
18    pub ssa_jwt: Option<String>,
19    /// Intervals to send log messages to the lock server.
20    /// Set this to [`None`] to disable transmission.
21    pub log_interval: Option<Duration>,
22    /// Intervals to send health messages to the lock server.
23    /// Set this to [`None`] to disable transmission.
24    pub health_interval: Option<Duration>,
25    /// Intervals to send telemetry messages to the lock server.
26    /// Set this to [`None`] to disable transmission.
27    pub telemetry_interval: Option<Duration>,
28    /// Controls whether Cedarling should listen for updates from the Lock Server.
29    pub listen_sse: bool,
30    /// Allow interaction with a Lock server with invalid certificates. Used for testing.
31    pub accept_invalid_certs: bool,
32}

94impl TryFrom<&BootstrapConfigRaw> for LockServiceConfig {
95    type Error = BootstrapConfigLoadingError;
96
97    fn try_from(raw: &BootstrapConfigRaw) -> Result<Self, Self::Error> {
98        let config_uri = raw
99            .lock_server_configuration_uri
100            .clone()
101            .ok_or(BootstrapConfigLoadingError::MissingLockServerConfigUri)?
102            .parse()?;
103
104        let ssa_jwt = raw.lock_ssa_jwt.clone();
105
106        let log_interval =
107            (raw.audit_log_interval > 0).then(|| Duration::from_secs(raw.audit_log_interval));
108        let health_interval =
109            (raw.audit_health_interval > 0).then(|| Duration::from_secs(raw.audit_health_interval));
110        let telemetry_interval = (raw.audit_telemetry_interval > 0)
111            .then(|| Duration::from_secs(raw.audit_telemetry_interval));
112
113        let listen_sse = raw.listen_sse.into();
114
115        Ok(LockServiceConfig {
116            config_uri,
117            dynamic_config: raw.dynamic_configuration.into(),
118            ssa_jwt,
119            log_interval,
120            health_interval,
121            telemetry_interval,
122            listen_sse,
123            log_level: raw.log_level,
124            accept_invalid_certs: raw.accept_invalid_certs.into(),
125        })
126    }
