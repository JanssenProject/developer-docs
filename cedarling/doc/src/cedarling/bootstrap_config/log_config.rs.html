12/// A set of properties used to configure logging in the `Cedarling` application.
13#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
14pub struct LogConfig {
15    /// `CEDARLING_LOG_TYPE` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
16    pub log_type: LogTypeConfig,
17
18    /// Log level filter for logging. TRACE is lowest. FATAL is highest.
19    /// `CEDARLING_LOG_LEVEL` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
20    pub log_level: LogLevel,
21}
22
23///  Log type configuration.
24///  `CEDARLING_LOG_TYPE` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
25///   Current type represent this value.
26#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
27pub enum LogTypeConfig {
28    /// Logger do nothing. It means that all logs will be ignored.
29    Off,
30    /// Logger holds all logs in database (in memory) with eviction policy.
31    Memory(MemoryLogConfig),
32    /// Logger writes log information to std output stream.
33    StdOut,
34}
35
36/// Configuration for memory log.
37#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
38pub struct MemoryLogConfig {
39    /// `CEDARLING_LOG_TTL` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
40    /// The maximum time to live (in seconds) of the log entries.
41    pub log_ttl: u64,
42
43    /// `CEDARLING_LOG_MAX_ITEMS` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
44    /// The maximum number of log entries to keep in memory.
45    pub max_items: Option<usize>,
46
47    /// `CEDARLING_LOG_MAX_ITEM_SIZE` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
48    /// The maximum size of a log entry in bytes.
49    pub max_item_size: Option<usize>,
50}

52/// Config for the lock logger that are set using the bootstrap configs.
53#[derive(Debug, Clone, PartialEq)]
54pub struct LockServiceConfig {
55    /// The logging level
56    pub log_level: LogLevel,
57    /// URI where Cedarling can get metadata about the lock server.
58    /// i.e. `.well-known/lock-server-configuration`
59    ///  
60    /// This is set using the `CEDARLING_LOCK_SERVER_CONFIGURATION_URI` bootstrap property.
61    pub config_uri: Url,
62    /// Toggles whether Cedarling should listen for SSE config updates.
63    ///
64    /// This is set using the `CEDARLING_LOCK_DYNAMIC_CONFIGURATION ` bootstrap property.
65    //  TODO: This feature is not yet implemented
66    pub dynamic_config: bool,
67    /// Software Statement Assertion Json Web Token that Cedarling will use for Dynamic
68    /// Client Registration.
69    ///
70    /// This is set using the `CEDARLING_LOCK_SSA_JWT` bootstrap property.
71    pub ssa_jwt: Option<String>,
72    /// Intervals to send log messges to the lock server.
73    /// Set this to [`None`] to disable transmission.
74    ///
75    /// This is set using the `CEDARLING_LOCK_LOG_INTERVAL` bootstrap property.
76    pub log_interval: Option<Duration>,
77    /// Intervals to send health messges to the lock server.
78    /// Set this to [`None`] to disable transmission.
79    ///
80    /// This is set using the `CEDARLING_LOCK_HEALTH_INTERVAL` bootstrap property.
81    //  TODO: This feature is not yet implemented
82    pub health_interval: Option<Duration>,
83    /// Intervals to send telemetry messges to the lock server.
84    /// Set this to [`None`] to disable transmission.
85    ///
86    /// This is set using the `CEDARLING_LOCK_TELEMETRY_INTERVAL` bootstrap property.
87    //  TODO: This feature is not yet implemented
88    pub telemetry_interval: Option<Duration>,
89    /// Controls whether Cedarling should listen for updates from the Lock Server.
90    ///
91    /// This is set using the `CEDARLING_LOCK_LISTEN_SSE` bootstrap property.
92    //  TODO: This feature is not yet implemented
93    pub listen_sse: bool,
94    /// Allow interaction with a Lock server with invalid certificates. Used for testing.
95    pub accept_invalid_certs: bool,
96}

98impl TryFrom<&BootstrapConfigRaw> for LockServiceConfig {
99    type Error = BootstrapConfigLoadingError;
100
101    fn try_from(raw: &BootstrapConfigRaw) -> Result<Self, Self::Error> {
102        let config_uri = raw
103            .lock_server_configuration_uri
104            .clone()
105            .ok_or(BootstrapConfigLoadingError::MissingLockServerConfigUri)?
106            .parse()?;
107
108        let ssa_jwt = raw.lock_ssa_jwt.clone();
109
110        let log_interval =
111            (raw.audit_log_interval > 0).then(|| Duration::from_secs(raw.audit_log_interval));
112        let health_interval =
113            (raw.audit_health_interval > 0).then(|| Duration::from_secs(raw.audit_health_interval));
114        let telemetry_interval = (raw.audit_telemetry_interval > 0)
115            .then(|| Duration::from_secs(raw.audit_telemetry_interval));
116
117        let listen_sse = raw.listen_sse.into();
118
119        Ok(LockServiceConfig {
120            config_uri,
121            dynamic_config: raw.dynamic_configuration.into(),
122            ssa_jwt,
123            log_interval,
124            health_interval,
125            telemetry_interval,
126            listen_sse,
127            log_level: raw.log_level,
128            accept_invalid_certs: raw.accept_invalid_certs.into(),
129        })
130    }
131}

133/// Raw log config
134pub struct LogConfigRaw {
135    /// Log type
136    pub log_type: String,
137    /// Log level
138    pub log_level: String,
139}
140
141/// Raw memory log config
142pub struct MemoryLogConfigRaw {
143    /// Log TTL
144    pub log_ttl: u64,
145    /// Max item size
146    pub max_item_size: Option<usize>,
147    /// Max items
148    pub max_items: Option<usize>,
149}
150
151impl From<LogConfigRaw> for LogConfig {
152    fn from(raw: LogConfigRaw) -> Self {
153        Self {
154            log_type: match raw.log_type.as_str() {
155                "off" => LogTypeConfig::Off,
156                "memory" => LogTypeConfig::Memory(MemoryLogConfig {
157                    log_ttl: 60,
158                    max_item_size: None,
159                    max_items: None,
160                }),
161                "stdout" => LogTypeConfig::StdOut,
162                _ => LogTypeConfig::StdOut,
163            },
164            log_level: match raw.log_level.as_str() {
165                "TRACE" => LogLevel::TRACE,
166                "DEBUG" => LogLevel::DEBUG,
167                "INFO" => LogLevel::INFO,
168                "WARN" => LogLevel::WARN,
169                "ERROR" => LogLevel::ERROR,
170                _ => LogLevel::INFO,
171            },
172        }
173    }
174}
