27impl BootstrapConfig {
28    /// Construct `BootstrapConfig` from environment variables and `BootstrapConfigRaw` config.
29    /// Environment variables have bigger priority.
30    //
31    // Simple implementation that map input structure to JSON map
32    // and map environment variables with prefix `CEDARLING_` to JSON map. And merge it.
33    #[cfg(not(target_arch = "wasm32"))]
34    pub fn from_raw_config_and_env(
35        raw: Option<BootstrapConfigRaw>,
36    ) -> Result<Self, BootstrapConfigLoadingError> {
37        let config_raw = BootstrapConfigRaw::from_raw_config_and_env(raw)?;
38        Self::from_raw_config(&config_raw)
39    }
40
41    /// Construct an instance from BootstrapConfigRaw
42    pub fn from_raw_config(raw: &BootstrapConfigRaw) -> Result<Self, BootstrapConfigLoadingError> {
43        if !raw.workload_authz.is_enabled() && !raw.user_authz.is_enabled() {
44            return Err(BootstrapConfigLoadingError::BothPrincipalsDisabled);
45        }
46
47        let lock_config = raw.lock.is_enabled().then(|| raw.try_into()).transpose()?;
48
49        // Decode LogCofig
50        let log_type = match raw.log_type {
51            LoggerType::Off => LogTypeConfig::Off,
52            LoggerType::Memory => LogTypeConfig::Memory(MemoryLogConfig {
53                log_ttl: raw
54                    .log_ttl
55                    .ok_or(BootstrapConfigLoadingError::MissingLogTTL)?,
56                max_item_size: raw.log_max_item_size,
57                max_items: raw.log_max_items,
58            }),
59            LoggerType::StdOut => LogTypeConfig::StdOut,
60        };
61        let log_config = LogConfig {
62            log_type,
63            log_level: raw.log_level,
64        };
65
66        // Decode policy store
67        let policy_store_config = match (
68            raw.local_policy_store.clone(),
69            raw.policy_store_uri.clone(),
70            raw.policy_store_local_fn.clone(),
71        ) {
72            // Case: no policy store provided
73            (None, None, None) => Err(BootstrapConfigLoadingError::MissingPolicyStore)?,
74            // Case: get the policy store from a JSON string
75            (Some(policy_store), None, None) => PolicyStoreConfig {
76                source: PolicyStoreSource::Json(policy_store),
77            },
78            // Case: get the policy store from the lock server
79            (None, Some(policy_store_uri), None) => PolicyStoreConfig {
80                source: PolicyStoreSource::LockServer(policy_store_uri),
81            },
82            // Case: get the policy store from a local JSON file
83            (None, None, Some(raw_path)) => {
84                let path = Path::new(&raw_path);
85                let file_ext = Path::new(&path)
86                    .extension()
87                    .and_then(|ext| ext.to_str())
88                    .map(|x| x.to_lowercase());
89
90                let source = match file_ext.as_deref() {
91                    Some("json") => PolicyStoreSource::FileJson(path.into()),
92                    Some("yaml") | Some("yml") => PolicyStoreSource::FileYaml(path.into()),
93                    _ => Err(
94                        BootstrapConfigLoadingError::UnsupportedPolicyStoreFileFormat(raw_path),
95                    )?,
96                };
97                PolicyStoreConfig { source }
98            },
99            // Case: multiple polict stores were set
100            _ => Err(BootstrapConfigLoadingError::ConflictingPolicyStores)?,
101        };
102
103        // Load the jwks from a local file
104        let jwks = raw
105            .local_jwks
106            .as_ref()
107            .map(|path| {
108                fs::read_to_string(path).map_err(|e| {
109                    BootstrapConfigLoadingError::LoadLocalJwks(path.to_string(), e.to_string())
110                })
111            })
112            .transpose()?;
113
114        // JWT Config
115        let jwt_config = JwtConfig {
116            jwks,
117            jwt_sig_validation: raw.jwt_sig_validation.into(),
118            jwt_status_validation: raw.jwt_status_validation.into(),
119            signature_algorithms_supported: raw.jwt_signature_algorithms_supported.clone(),
120        };
121
122        let authorization_config = AuthorizationConfig {
123            use_user_principal: raw.user_authz.is_enabled(),
124            use_workload_principal: raw.workload_authz.is_enabled(),
125            principal_bool_operator: raw.principal_bool_operation.clone(),
126            decision_log_user_claims: raw.decision_log_user_claims.clone(),
127            decision_log_workload_claims: raw.decision_log_workload_claims.clone(),
128            decision_log_default_jwt_id: raw.decision_log_default_jwt_id.clone(),
129            id_token_trust_mode: raw.id_token_trust_mode.clone(),
130        };
131
132        Ok(Self {
133            application_name: raw.application_name.clone(),
134            log_config,
135            policy_store_config,
136            jwt_config,
137            authorization_config,
138            entity_builder_config: raw.into(),
139            lock_config,
140            max_default_entities: raw.max_default_entities,
141            max_base64_size: raw.max_base64_size,
142            token_cache_max_ttl_secs: raw.token_cache_max_ttl,
143        })
144    }
