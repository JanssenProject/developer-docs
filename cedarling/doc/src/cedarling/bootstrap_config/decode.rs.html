27impl BootstrapConfig {
28    /// Construct `BootstrapConfig` from environment variables and `BootstrapConfigRaw` config.
29    /// Environment variables have bigger priority.
30    //
31    // Simple implementation that map input structure to JSON map
32    // and map environment variables with prefix `CEDARLING_` to JSON map. And merge it.
33    #[cfg(not(target_arch = "wasm32"))]
34    pub fn from_raw_config_and_env(
35        raw: Option<BootstrapConfigRaw>,
36    ) -> Result<Self, BootstrapConfigLoadingError> {
37        let config_raw = BootstrapConfigRaw::from_raw_config_and_env(raw)?;
38        Self::from_raw_config(&config_raw)
39    }
40
41    /// Construct an instance from BootstrapConfigRaw
42    pub fn from_raw_config(raw: &BootstrapConfigRaw) -> Result<Self, BootstrapConfigLoadingError> {
43        if !raw.workload_authz.is_enabled() && !raw.user_authz.is_enabled() {
44            return Err(BootstrapConfigLoadingError::BothPrincipalsDisabled);
45        }
46
47        let lock_config = raw.lock.is_enabled().then(|| raw.try_into()).transpose()?;
48
49        // Decode LogCofig
50        let log_type = match raw.log_type {
51            LoggerType::Off => LogTypeConfig::Off,
52            LoggerType::Memory => LogTypeConfig::Memory(MemoryLogConfig {
53                log_ttl: raw
54                    .log_ttl
55                    .ok_or(BootstrapConfigLoadingError::MissingLogTTL)?,
56                max_item_size: raw.log_max_item_size,
57                max_items: raw.log_max_items,
58            }),
59            LoggerType::StdOut => {
60                let std_out_logger_conf = match raw.stdout_mode {
61                    // WASM does not support async
62                    #[cfg(not(target_arch = "wasm32"))]
63                    super::log_config::StdOutMode::Async => StdOutLoggerMode::Async {
64                        timeout_millis: raw.stdout_timeout_millis,
65                        buffer_limit: raw.stdout_buffer_limit,
66                    },
67                    super::log_config::StdOutMode::Immediate => StdOutLoggerMode::Immediate,
68                };
69                LogTypeConfig::StdOut(std_out_logger_conf)
70            },
71        };
72        let log_config = LogConfig {
73            log_type,
74            log_level: raw.log_level,
75        };
76
77        // Decode policy store
78        let policy_store_config = match (
79            raw.local_policy_store.clone(),
80            raw.policy_store_uri.clone(),
81            raw.policy_store_local_fn.clone(),
82        ) {
83            // Case: no policy store provided
84            (None, None, None) => Err(BootstrapConfigLoadingError::MissingPolicyStore)?,
85            // Case: get the policy store from a JSON string
86            (Some(policy_store), None, None) => PolicyStoreConfig {
87                source: PolicyStoreSource::Json(policy_store),
88            },
89            // Case: get the policy store from the lock server
90            (None, Some(policy_store_uri), None) => PolicyStoreConfig {
91                source: PolicyStoreSource::LockServer(policy_store_uri),
92            },
93            // Case: get the policy store from a local JSON file
94            (None, None, Some(raw_path)) => {
95                let path = Path::new(&raw_path);
96                let file_ext = Path::new(&path)
97                    .extension()
98                    .and_then(|ext| ext.to_str())
99                    .map(|x| x.to_lowercase());
100
101                let source = match file_ext.as_deref() {
102                    Some("json") => PolicyStoreSource::FileJson(path.into()),
103                    Some("yaml") | Some("yml") => PolicyStoreSource::FileYaml(path.into()),
104                    _ => Err(
105                        BootstrapConfigLoadingError::UnsupportedPolicyStoreFileFormat(raw_path),
106                    )?,
107                };
108                PolicyStoreConfig { source }
109            },
110            // Case: multiple polict stores were set
111            _ => Err(BootstrapConfigLoadingError::ConflictingPolicyStores)?,
112        };
113
114        // Load the jwks from a local file
115        let jwks = raw
116            .local_jwks
117            .as_ref()
118            .map(|path| {
119                fs::read_to_string(path).map_err(|e| {
120                    BootstrapConfigLoadingError::LoadLocalJwks(path.to_string(), e.to_string())
121                })
122            })
123            .transpose()?;
124
125        // JWT Config
126        let jwt_config = JwtConfig {
127            jwks,
128            jwt_sig_validation: raw.jwt_sig_validation.into(),
129            jwt_status_validation: raw.jwt_status_validation.into(),
130            signature_algorithms_supported: raw.jwt_signature_algorithms_supported.clone(),
131            token_cache_max_ttl_secs: raw.token_cache_max_ttl,
132            token_cache_capacity: raw.token_cache_capacity,
133            token_cache_earliest_expiration_eviction: raw.token_cache_earliest_expiration_eviction,
134        };
135
136        let authorization_config = AuthorizationConfig {
137            use_user_principal: raw.user_authz.is_enabled(),
138            use_workload_principal: raw.workload_authz.is_enabled(),
139            principal_bool_operator: raw.principal_bool_operation.clone(),
140            decision_log_user_claims: raw.decision_log_user_claims.clone(),
141            decision_log_workload_claims: raw.decision_log_workload_claims.clone(),
142            decision_log_default_jwt_id: raw.decision_log_default_jwt_id.clone(),
143            id_token_trust_mode: raw.id_token_trust_mode.clone(),
144        };
145
146        Ok(Self {
147            application_name: raw.application_name.clone(),
148            log_config,
149            policy_store_config,
150            jwt_config,
151            authorization_config,
152            entity_builder_config: raw.into(),
153            lock_config,
154            max_default_entities: raw.max_default_entities,
155            max_base64_size: raw.max_base64_size,
156        })
157    }
