30impl BootstrapConfig {
31    /// Construct `BootstrapConfig` from environment variables and `BootstrapConfigRaw` config.
32    /// Environment variables have bigger priority.
33    //
34    // Simple implementation that map input structure to JSON map
35    // and map environment variables with prefix `CEDARLING_` to JSON map. And merge it.
36    #[cfg(not(target_arch = "wasm32"))]
37    pub fn from_raw_config_and_env(
38        raw: Option<BootstrapConfigRaw>,
39    ) -> Result<Self, BootstrapConfigLoadingError> {
40        let config_raw = BootstrapConfigRaw::from_raw_config_and_env(raw)?;
41        Self::from_raw_config(&config_raw)
42    }
43
44    /// Construct an instance from [`BootstrapConfigRaw`]
45    pub fn from_raw_config(raw: &BootstrapConfigRaw) -> Result<Self, BootstrapConfigLoadingError> {
46        if !raw.workload_authz.is_enabled() && !raw.user_authz.is_enabled() {
47            return Err(BootstrapConfigLoadingError::BothPrincipalsDisabled);
48        }
49
50        let lock_config = raw.lock.is_enabled().then(|| raw.try_into()).transpose()?;
51
52        // Decode LogCofig
53        let log_config = LogConfig {
54            log_type: resolve_log_type(raw)?,
55            log_level: raw.log_level,
56        };
57
58        // Decode policy store
59        let policy_store_config = match (
60            raw.local_policy_store.clone(),
61            raw.policy_store_uri.clone(),
62            raw.policy_store_local_fn.clone(),
63        ) {
64            // Case: no policy store provided
65            (None, None, None) => Err(BootstrapConfigLoadingError::MissingPolicyStore)?,
66            // Case: get the policy store from a JSON string
67            (Some(policy_store), None, None) => PolicyStoreConfig {
68                source: PolicyStoreSource::Json(policy_store),
69            },
70            // Case: get the policy store from a URI (auto-detect .cjar archives)
71            (None, Some(policy_store_uri), None) => {
72                let source = if policy_store_uri.to_lowercase().ends_with(".cjar") {
73                    PolicyStoreSource::CjarUrl(policy_store_uri)
74                } else {
75                    PolicyStoreSource::LockServer(policy_store_uri)
76                };
77                PolicyStoreConfig { source }
78            },
79            // Case: get the policy store from a local file or directory
80            (None, None, Some(raw_path)) => {
81                let path = Path::new(&raw_path);
82
83                // Check if it's a directory first
84                let source = if path.is_dir() {
85                    PolicyStoreSource::Directory(path.into())
86                } else {
87                    let file_ext = path
88                        .extension()
89                        .and_then(|ext| ext.to_str())
90                        .map(str::to_lowercase);
91
92                    match file_ext.as_deref() {
93                        Some("json") => PolicyStoreSource::FileJson(path.into()),
94                        Some("yaml" | "yml") => PolicyStoreSource::FileYaml(path.into()),
95                        Some("cjar") => PolicyStoreSource::CjarFile(path.into()),
96                        _ => Err(
97                            BootstrapConfigLoadingError::UnsupportedPolicyStoreFileFormat(raw_path),
98                        )?,
99                    }
100                };
101                PolicyStoreConfig { source }
102            },
103            // Case: multiple polict stores were set
104            _ => Err(BootstrapConfigLoadingError::ConflictingPolicyStores)?,
105        };
106
107        // Load the jwks from a local file
108        let jwks = raw
109            .local_jwks
110            .as_ref()
111            .map(|path| {
112                fs::read_to_string(path).map_err(|e| {
113                    BootstrapConfigLoadingError::LoadLocalJwks(path.clone(), e.to_string())
114                })
115            })
116            .transpose()?;
117
118        // JWT Config
119        let jwt_config = JwtConfig {
120            jwks,
121            jwt_sig_validation: raw.jwt_sig_validation.into(),
122            jwt_status_validation: raw.jwt_status_validation.into(),
123            signature_algorithms_supported: raw.jwt_signature_algorithms_supported.clone(),
124            token_cache_max_ttl_secs: raw.token_cache_max_ttl,
125            token_cache_capacity: raw.token_cache_capacity,
126            token_cache_earliest_expiration_eviction: raw.token_cache_earliest_expiration_eviction,
127            trusted_issuer_loader: raw
128                .trusted_issuer_loader_type
129                .to_config(raw.trusted_issuer_loader_workers),
130        };
131
132        let authorization_config = AuthorizationConfig {
133            use_user_principal: raw.user_authz.is_enabled(),
134            use_workload_principal: raw.workload_authz.is_enabled(),
135            principal_bool_operator: raw.principal_bool_operation.clone(),
136            decision_log_user_claims: raw.decision_log_user_claims.clone(),
137            decision_log_workload_claims: raw.decision_log_workload_claims.clone(),
138            decision_log_default_jwt_id: raw.decision_log_default_jwt_id.clone(),
139            id_token_trust_mode: raw.id_token_trust_mode.clone(),
140        };
141
142        // Build `DataStoreConfig` from raw config, using defaults if not specified
143        let data_store_config = build_data_store_config(raw);
144
145        Ok(Self {
146            application_name: raw.application_name.clone(),
147            log_config,
148            policy_store_config,
149            jwt_config,
150            authorization_config,
151            entity_builder_config: raw.into(),
152            lock_config,
153            max_default_entities: raw.max_default_entities,
154            max_base64_size: raw.max_base64_size,
155            data_store_config,
156        })
157    }
158}

160/// Build `DataStoreConfig` from raw config fields.
161/// Uses default values for any fields that are not specified.
162fn build_data_store_config(raw: &BootstrapConfigRaw) -> DataStoreConfig {
163    let defaults = DataStoreConfig::default();
164
165    DataStoreConfig {
166        max_entries: raw.data_store_max_entries.unwrap_or(defaults.max_entries),
167        max_entry_size: raw
168            .data_store_max_entry_size
169            .unwrap_or(defaults.max_entry_size),
170        default_ttl: raw
171            .data_store_default_ttl
172            .map(std::time::Duration::from_secs)
173            .or(defaults.default_ttl),
174        max_ttl: raw
175            .data_store_max_ttl
176            .map(std::time::Duration::from_secs)
177            .or(defaults.max_ttl),
178        enable_metrics: raw
179            .data_store_enable_metrics
180            .unwrap_or(defaults.enable_metrics),
181        memory_alert_threshold: raw
182            .data_store_memory_alert_threshold
183            .unwrap_or(defaults.memory_alert_threshold),
184    }
185}

187/// Helper function to resolve log type from raw config
188fn resolve_log_type(
189    raw_config: &BootstrapConfigRaw,
190) -> Result<LogTypeConfig, BootstrapConfigLoadingError> {
191    let log_type_config = match raw_config.log_type {
192        LoggerType::Off => LogTypeConfig::Off,
193        LoggerType::Memory => LogTypeConfig::Memory(MemoryLogConfig {
194            log_ttl: raw_config
195                .log_ttl
196                .ok_or(BootstrapConfigLoadingError::MissingLogTTL)?,
197            max_item_size: raw_config.log_max_item_size,
198            max_items: raw_config.log_max_items,
199        }),
200        LoggerType::StdOut => {
201            let std_out_logger_conf = match raw_config.stdout_mode {
202                // WASM does not support async
203                #[cfg(not(target_arch = "wasm32"))]
204                super::log_config::StdOutMode::Async => StdOutLoggerMode::Async {
205                    timeout_millis: raw_config.stdout_timeout_millis,
206                    buffer_limit: raw_config.stdout_buffer_limit,
207                },
208                super::log_config::StdOutMode::Immediate => StdOutLoggerMode::Immediate,
209            };
210            LogTypeConfig::StdOut(std_out_logger_conf)
211        },
212    };
213    Ok(log_type_config)
