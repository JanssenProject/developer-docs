6use derive_more::Deref;
7use serde::{Deserialize, Serialize};
8use crate::BootstrapConfigRaw;
9
10const DEFAULT_USER_ENTITY_NAME: &str = "Jans::User";
11const DEFAULT_WORKLOAD_ENTITY_NAME: &str = "Jans::Workload";
12const DEFAULT_ROLE_ENTITY_NAME: &str = "Jans::Role";
13const DEFAULT_ISS_ENTITY_NAME: &str = "Jans::TrustedIssuer";
14const DEFAULT_UNSIGNED_ROLE_ID_SRC: &str = "role";
15
16pub(crate) const DEFAULT_ACCESS_TKN_ENTITY_NAME: &str = "Jans::Access_token";
17pub(crate) const DEFAULT_ID_TKN_ENTITY_NAME: &str = "Jans::Id_token";
18pub(crate) const DEFAULT_USERINFO_TKN_ENTITY_NAME: &str = "Jans::Userinfo_token";
19
20/// Bootstrap Configurations for the JWT to Cedar entity mappings
21#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
22pub struct EntityBuilderConfig {
23    /// The names of the buildable Cedar entity type names
24    pub entity_names: EntityNames,
25    /// Toggles building the `Workload` entity
26    pub build_workload: bool,
27    /// Toggles building the `User` entity
28    pub build_user: bool,
29    /// The attribute to use when creating Role entities in the unsigned interface
30    pub unsigned_role_id_src: UnsignedRoleIdSrc,
31}
32
33/// Raw entity builder config
34#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
35pub struct EntityBuilderConfigRaw {
36    /// Toggles building the `Workload` entity
37    pub workload_authz: bool,
38    /// Toggles building the `User` entity
39    pub user_authz: bool,
40    /// The attribute to use when creating Role entities in the unsigned interface
41    pub unsigned_role_id_src: UnsignedRoleIdSrc,
42    /// Mapping name of cedar schema User entity
43    pub mapping_user: Option<String>,
44    /// Mapping name of cedar schema Workload entity
45    pub mapping_workload: Option<String>,
46    /// Mapping name of cedar schema Role entity
47    pub mapping_role: Option<String>,
48    /// Mapping name of cedar schema Issuer entity
49    pub mapping_iss: Option<String>,
50}
51
52/// Entity names
53#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
54pub struct EntityNames {
55    /// The entity type name of the `User` entity
56    pub user: String,
57    /// The entity type name of the `Workload` entity
58    pub workload: String,
59    /// The entity type name of the `Role` entity
60    pub role: String,
61    /// The entity type name of the `TrustedIssuer` entity
62    pub iss: String,
63}
64
65/// Unsigned role ID source
66#[derive(Debug, Clone, PartialEq, Deref, Serialize, Deserialize)]
67pub struct UnsignedRoleIdSrc(pub String);
68
69impl Default for UnsignedRoleIdSrc {
70    fn default() -> Self {
71        Self(DEFAULT_UNSIGNED_ROLE_ID_SRC.to_string())
72    }
73}
74
75impl Default for EntityBuilderConfig {
76    fn default() -> Self {
77        Self {
78            entity_names: EntityNames {
79                user: DEFAULT_USER_ENTITY_NAME.to_string(),
80                workload: DEFAULT_WORKLOAD_ENTITY_NAME.to_string(),
81                role: DEFAULT_ROLE_ENTITY_NAME.to_string(),
82                iss: DEFAULT_ISS_ENTITY_NAME.to_string(),
83            },
84            build_workload: true,
85            build_user: true,
86            unsigned_role_id_src: UnsignedRoleIdSrc(DEFAULT_UNSIGNED_ROLE_ID_SRC.to_string()),
87        }
88    }
89}
90
91impl EntityBuilderConfig {
92    /// Enables building the `Workload` entity
93    pub fn with_workload(mut self) -> Self {
94        self.build_workload = true;
95        self
96    }
97
98    /// Enables building the `User` entity
99    pub fn with_user(mut self) -> Self {
100        self.build_user = true;
101        self
102    }
103}
104
105impl From<EntityBuilderConfigRaw> for EntityBuilderConfig {
106    fn from(raw: EntityBuilderConfigRaw) -> Self {
107        let entity_names = EntityNames {
108            user: raw.mapping_user.unwrap_or_else(|| DEFAULT_USER_ENTITY_NAME.to_string()),
109            workload: raw.mapping_workload.unwrap_or_else(|| DEFAULT_WORKLOAD_ENTITY_NAME.to_string()),
110            role: raw.mapping_role.unwrap_or_else(|| DEFAULT_ROLE_ENTITY_NAME.to_string()),
111            iss: raw.mapping_iss.unwrap_or_else(|| DEFAULT_ISS_ENTITY_NAME.to_string()),
112        };
113
114        Self {
115            entity_names,
116            build_workload: raw.workload_authz,
117            build_user: raw.user_authz,
118            unsigned_role_id_src: raw.unsigned_role_id_src,
119        }
120    }
121}
122
123impl From<&BootstrapConfigRaw> for EntityBuilderConfig {
124    fn from(raw: &BootstrapConfigRaw) -> Self {
125        let raw_entity = EntityBuilderConfigRaw {
126            workload_authz: raw.workload_authz.is_enabled(),
127            user_authz: raw.user_authz.is_enabled(),
128            unsigned_role_id_src: raw.unsigned_role_id_src.clone(),
129            mapping_user: raw.mapping_user.clone(),
130            mapping_workload: raw.mapping_workload.clone(),
131            mapping_role: raw.mapping_role.clone(),
132            mapping_iss: raw.mapping_iss.clone(),
133        };
134        EntityBuilderConfig::from(raw_entity)
135    }
136}
