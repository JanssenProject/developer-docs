6use crate::bootstrap_config::jwt_config::TokenValidationConfig;
7use super::super::authorization_config::TokenEntityNames;
8use super::FeatureToggle;
9use serde::{Deserialize, Serialize};
10use std::collections::HashMap;
11
12/// Configuration for token-based entities, mapping token names to their
13/// respective settings.
14///
15/// Each entry in this map associates a token identifier (String) with its
16/// corresponding `TokenEntityConfig` settings. These settings define the
17/// entity type and claim validation rules
18#[derive(Debug, Deserialize, PartialEq, Clone, Serialize)]
19pub struct TokenConfigs(HashMap<String, TokenConfig>);
20
21/// Detailed configuration for a single token-based entity.
22///
23/// This struct defines the entity type name and the validation settings for
24/// various claims associated with the token. Each claim validation setting is
25/// represented as a `FeatureToggle`.
26#[derive(Debug, Deserialize, PartialEq, Clone, Serialize)]
27pub struct TokenConfig {
28    /// The name of the Cedar entity type associated with this token.
29    entity_type_name: String,
30    #[serde(default, flatten)]
31    claims: ClaimsValidationConfig,
32}
33
34/// Claims validation config
35#[derive(Debug, Deserialize, PartialEq, Clone, Default, Serialize)]
36#[serde(default)]
37pub struct ClaimsValidationConfig {
38    /// Toggle for validating the `iss` (issuer) claim.
39    iss: FeatureToggle,
40    /// Toggle for validating the `sub` (subject) claim.
41    sub: FeatureToggle,
42    /// Toggle for validating the `aud` (audience) claim.
43    aud: FeatureToggle,
44    /// Toggle for validating the `exp` (expiration) claim.
45    exp: FeatureToggle,
46    /// Toggle for validating the `nbf` (not before) claim.
47    nbf: FeatureToggle,
48    /// Toggle for validating the `iat` (issued at) claim.
49    iat: FeatureToggle,
50    /// Toggle for validating the `jti` (JWT ID) claim.
51    jti: FeatureToggle,
52}
53
54impl TokenConfigs {
55    /// Create a config without validation
56    pub fn without_validation() -> Self {
57        Self(HashMap::from([
58            ("access_token".to_string(), TokenConfig {
59                entity_type_name: "Jans::Access_token".to_string(),
60                claims: ClaimsValidationConfig::default(),
61            }),
62            ("id_token".to_string(), TokenConfig {
63                entity_type_name: "Jans::Id_token".to_string(),
64                claims: ClaimsValidationConfig::default(),
65            }),
66            ("userinfo_token".to_string(), TokenConfig {
67                entity_type_name: "Jans::Userinfo_token".to_string(),
68                claims: ClaimsValidationConfig::default(),
69            }),
70        ]))
71    }
72}
73
74impl Default for TokenConfigs {
75    fn default() -> Self {
76        // we need define namespace for each token type, so default will be `Jans`
77        // If we avoid define namespace for each token type, the namespace would be empty string ("")
78        Self(HashMap::from([
79            ("access_token".to_string(), TokenConfig {
80                entity_type_name: "Jans::Access_token".to_string(),
81                claims: ClaimsValidationConfig {
82                    iss: FeatureToggle::Enabled,
83                    sub: FeatureToggle::Disabled,
84                    aud: FeatureToggle::Disabled,
85                    exp: FeatureToggle::Enabled,
86                    nbf: FeatureToggle::Disabled,
87                    iat: FeatureToggle::Disabled,
88                    jti: FeatureToggle::Enabled,
89                },
90            }),
91            ("id_token".to_string(), TokenConfig {
92                entity_type_name: "Jans::Id_token".to_string(),
93                claims: ClaimsValidationConfig {
94                    iss: FeatureToggle::Enabled,
95                    sub: FeatureToggle::Enabled,
96                    aud: FeatureToggle::Enabled,
97                    exp: FeatureToggle::Enabled,
98                    nbf: FeatureToggle::Disabled,
99                    iat: FeatureToggle::Disabled,
100                    jti: FeatureToggle::Disabled,
101                },
102            }),
103            ("userinfo_token".to_string(), TokenConfig {
104                entity_type_name: "Jans::Userinfo_token".to_string(),
105                claims: ClaimsValidationConfig {
106                    iss: FeatureToggle::Enabled,
107                    sub: FeatureToggle::Enabled,
108                    aud: FeatureToggle::Enabled,
109                    exp: FeatureToggle::Enabled,

111                    iat: FeatureToggle::Disabled,
112                    jti: FeatureToggle::Disabled,
113                },
114            }),
115        ]))
116    }
117}

119impl From<TokenConfig> for TokenValidationConfig {
120    fn from(setting: TokenConfig) -> Self {
121        Self {
122            iss_validation: setting.claims.iss.into(),
123            aud_validation: setting.claims.aud.into(),
124            sub_validation: setting.claims.sub.into(),
125            jti_validation: setting.claims.jti.into(),
126            iat_validation: setting.claims.iat.into(),
127            exp_validation: setting.claims.exp.into(),
128            nbf_validation: setting.claims.nbf.into(),
129        }
130    }
131}
132
133impl From<TokenConfigs> for HashMap<String, TokenConfig> {
134    fn from(value: TokenConfigs) -> Self {
135        HashMap::from_iter(value.0)
136    }
137}
138
139impl From<TokenConfigs> for TokenEntityNames {
140    fn from(value: TokenConfigs) -> Self {
141        Self(HashMap::from_iter(value.0.into_iter().map(
142            |(tkn_name, config)| (tkn_name, config.entity_type_name),
143        )))
144    }
145}
146
147impl From<TokenConfigs> for HashMap<String, TokenValidationConfig> {
148    fn from(value: TokenConfigs) -> Self {
149        HashMap::from_iter(
150            value
151                .0
152                .into_iter()
153                .map(|(tkn_name, config)| (tkn_name, config.into())),
154        )
155    }
