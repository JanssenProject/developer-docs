42/// Bootstrap configuration
43/// properties for configuration [`Cedarling`](crate::Cedarling) application.
44/// [link](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) to the documentation.
45#[derive(Debug, Clone, PartialEq)]
46pub struct BootstrapConfig {
47    /// `CEDARLING_APPLICATION_NAME` in [bootstrap properties](https://github.com/JanssenProject/jans/wiki/Cedarling-Nativity-Plan#bootstrap-properties) documentation.
48    pub application_name: String,
49    /// A set of properties used to configure logging in the `Cedarling` application.
50    pub log_config: LogConfig,
51    /// A set of properties used to load `PolicyStore` in the `Cedarling` application.
52    pub policy_store_config: PolicyStoreConfig,
53    /// A set of properties used to configure JWT in the `Cedarling` application.
54    pub jwt_config: JwtConfig,
55    /// A set of properties used to configure authorization workflow in the `Cedarling` application.
56    pub authorization_config: AuthorizationConfig,
57    /// A set of properties used to configure the JWTs to Cedar Entity mappings
58    pub entity_builder_config: EntityBuilderConfig,
59    /// Lock service configuration.
60    /// If `None` then lock service is disabled.
61    pub lock_config: Option<LockServiceConfig>,
62    /// Maximum number of default entities allowed in a policy store.
63    /// This prevents DoS attacks by limiting the number of entities that can be loaded.
64    pub max_default_entities: Option<usize>,
65    /// Maximum size of base64-encoded default entity strings in bytes.
66    /// This prevents memory exhaustion attacks from extremely large base64 strings.
67    pub max_base64_size: Option<usize>,
68}

70impl BootstrapConfig {
71    /// Loads a `BootstrapConfig` from a file.
72    ///
73    /// The file format is determined based on its extension:
74    /// - `.json`: Parses the file as JSON.
75    /// - `.yaml` or `.yml`: Parses the file as YAML.
76    /// - `.toml`: Parses the file as TOML.
77    ///
78    /// # Example
79    ///
80    /// ```rust
81    /// use cedarling::BootstrapConfig;
82    ///
83    /// let config = BootstrapConfig::load_from_file("../test_files/bootstrap_props.json").unwrap();
84    /// ```
85    #[cfg(not(target_arch = "wasm32"))]
86    pub fn load_from_file<P: AsRef<Path>>(path: P) -> Result<Self, BootstrapConfigLoadingError> {
87        let config = Config::builder()
88            .add_source(File::from(path.as_ref()))
89            .build()
90            .map_err(|e| BootstrapConfigLoadingError::DecodingJSON(e.to_string()))?;
91
92        let raw: BootstrapConfigRaw = config
93            .try_deserialize()
94            .map_err(|e| BootstrapConfigLoadingError::DecodingJSON(e.to_string()))?;
95        raw.try_into()
96    }
97
98    /// Loads a `BootstrapConfig` from a JSON string
99    pub fn load_from_json(json: &str) -> Result<Self, BootstrapConfigLoadingError> {
100        let raw: BootstrapConfigRaw = serde_json::from_str(json)
101            .map_err(|e| BootstrapConfigLoadingError::DecodingJSON(e.to_string()))?;
102        raw.try_into()
103    }
104
105    /// Load config from environment variables.
106    /// If you need with fallback to applied config use [`Self::from_raw_config_and_env`].
107    #[cfg(not(target_arch = "wasm32"))]
108    pub fn from_env() -> Result<Self, BootstrapConfigLoadingError> {
109        let config = Config::builder()
110            .add_source(config::Environment::with_prefix("CEDARLING"))
111            .build()
112            .map_err(|e| BootstrapConfigLoadingError::DecodingJSON(e.to_string()))?;
113
114        let raw: BootstrapConfigRaw = config
115            .try_deserialize()
116            .map_err(|e| BootstrapConfigLoadingError::DecodingJSON(e.to_string()))?;
117
118        raw.try_into()
119    }
120
121    /// Loads the default configuration bundled with the library.
122    /// This configuration provides sensible defaults for all components.
123    ///
124    /// # Example
125    ///
126    /// ```rust
127    /// use cedarling::BootstrapConfig;
128    ///
129    /// let config = BootstrapConfig::load_default().unwrap();
130    /// ```
131    pub fn load_default() -> Result<Self, BootstrapConfigLoadingError> {
132        const DEFAULT_CONFIG: &str = include_str!("../../config/default_config.yaml");
133
134        let config = Config::builder()
135            .add_source(File::from_str(DEFAULT_CONFIG, config::FileFormat::Yaml))
136            .build()
137            .map_err(|e| BootstrapConfigLoadingError::DecodingYAML(e.to_string()))?;
138
139        let raw: BootstrapConfigRaw = config
140            .try_deserialize()
141            .map_err(|e| BootstrapConfigLoadingError::DecodingYAML(e.to_string()))?;
142
143        raw.try_into()
144    }
145}

147impl TryFrom<BootstrapConfigRaw> for BootstrapConfig {
148    type Error = BootstrapConfigLoadingError;
149
150    fn try_from(raw: BootstrapConfigRaw) -> Result<Self, Self::Error> {
151        Self::from_raw_config(&raw)
152    }
153}

155/// Represents errors that may occur while loading a `BootstrapConfig` from a file.
156#[derive(Debug, thiserror::Error)]
157pub enum BootstrapConfigLoadingError {
158    /// Error returned when the file format is unsupported.
159    ///
160    /// Supported formats include:
161    /// - `.json`
162    /// - `.yaml` or `.yml`
163    /// - `.toml`
164    #[cfg(not(target_arch = "wasm32"))]
165    #[error(
166        "Unsupported bootstrap config file format for: {0}. Supported formats include: JSON, YAML, TOML"
167    )]
168    InvalidFileFormat(String),
169
170    /// Error returned when the file cannot be read.
171    #[cfg(not(target_arch = "wasm32"))]
172    #[error("Failed to read {0}: {1}")]
173    ReadFile(String, io::Error),
174
175    /// Error returned when parsing the file as JSON fails.
176    #[error("Failed to decode JSON string into BootstrapConfig: {0}")]
177    DecodingJSON(String),
178
179    /// Error returned when parsing the file as YAML fails.
180    #[error("Failed to decode YAML string into BootstrapConfig: {0}")]
181    DecodingYAML(String),
182
183    /// Error returned when parsing the file as TOML fails.
184    #[error("Failed to decode TOML string into BootstrapConfig: {0}")]
185    DecodingTOML(String),
186
187    /// Error returned when the boostrap property `CEDARLING_LOG_TTL` is missing.
188    #[error(
189        "Missing bootstrap property: `CEDARLING_LOG_TTL`. This property is required if \
190         `CEDARLING_LOG_TYPE` is set to Memory."
191    )]
192    MissingLogTTL,
193
194    /// Error returned when multiple policy store sources were provided.
195    #[error(
196        "Multiple store options were provided. Make sure you only one of these properties is set: \
197         `CEDARLING_POLICY_STORE_URI` or `CEDARLING_POLICY_STORE_LOCAL`"
198    )]
199    ConflictingPolicyStores,
200
201    /// Error returned when no policy store source was provided.
202    #[error("No Policy store was provided.")]
203    MissingPolicyStore,
204
205    /// Error returned when the policy store file is in an unsupported format.
206    #[error("Unsupported policy store file format for: {0}. Supported formats include: JSON, YAML")]
207    UnsupportedPolicyStoreFileFormat(String),
208
209    /// Error returned when failing to load a local JWKS
210    #[error("Failed to load local JWKS from {0}: {1}")]
211    LoadLocalJwks(String, String),
212
213    /// Error returned when both `CEDARLING_USER_AUTHZ` and `CEDARLING_WORKLOAD_AUTHZ` are disabled.
214    /// These two authentication configurations cannot be disabled at the same time.
215    #[error(
216        "Both `CEDARLING_USER_AUTHZ` and `CEDARLING_WORKLOAD_AUTHZ` cannot be disabled \
217         simultaneously."
218    )]
219    BothPrincipalsDisabled,
220
221    /// Error returned when `CEDARLING_LOCK` is set to `enabled` but `CEDARLING_LOCK_SERVER_CONFIGURATION_URI` is not set.
222    #[error(
223        "the `CEDARLING_LOCK` is set to `enabled` but `CEDARLING_LOCK_SERVER_CONFIGURATION_URI` is not set."
224    )]
225    MissingLockServerConfigUri,
226
227    /// Error returned when the lock server configuration URI is invalid.
228    #[error("Invalid lock server configuration URI: {0}")]
229    InvalidLockServerConfigUri(url::ParseError),
230
231    /// Error returned when cjar_url is missing or empty.
232    #[error(
233        "cjar_url is missing or empty. A valid URL is required for CjarUrl policy store source."
234    )]
235    MissingCjarUrl,
236}

238impl From<url::ParseError> for BootstrapConfigLoadingError {
239    fn from(err: url::ParseError) -> Self {
240        BootstrapConfigLoadingError::InvalidLockServerConfigUri(err)
241    }
242}
243
244impl From<serde_json::Error> for BootstrapConfigLoadingError {
245    fn from(err: serde_json::Error) -> Self {
246        Self::DecodingJSON(err.to_string())
247    }
248}

252        unreachable!("Infallible cannot be instantiated")
253    }

258    use jsonwebtoken::Algorithm;

264    #[test]
265    fn test_load_default_config() {
266        let config = BootstrapConfig::load_default().unwrap();
267
268        // Verify basic configuration
269        assert_eq!(config.application_name, "My App");
270
271        // Verify log configuration
272        assert!(matches!(
273            config.log_config.log_type,
274            LogTypeConfig::Memory(_)
275        ));
276        assert_eq!(config.log_config.log_level, LogLevel::DEBUG);
277
278        // Verify policy store configuration
279        assert!(matches!(
280            config.policy_store_config.source,
281            PolicyStoreSource::FileJson(_)
282        ));
283
284        // Verify JWT configuration
285        assert!(config.jwt_config.jwt_sig_validation);
286        assert!(!config.jwt_config.jwt_status_validation);
287        assert!(
288            config
289                .jwt_config
290                .signature_algorithms_supported
291                .contains(&Algorithm::HS256)
292        );
293        assert!(
294            config
295                .jwt_config
296                .signature_algorithms_supported
297                .contains(&Algorithm::RS256)
298        );
299
300        // Verify authorization configuration
301        assert!(config.authorization_config.use_user_principal);
302        assert!(config.authorization_config.use_workload_principal);
303        assert_eq!(
304            config.authorization_config.decision_log_default_jwt_id,
305            "jti"
306        );
307
308        // Verify entity builder configuration
309        assert!(config.entity_builder_config.build_user);
310        assert!(config.entity_builder_config.build_workload);
311    }
