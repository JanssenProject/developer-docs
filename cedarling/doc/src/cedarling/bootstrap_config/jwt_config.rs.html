7use jsonwebtoken::Algorithm;
8use std::collections::HashSet;
9use serde::{Deserialize, Serialize};
10
11/// The set of Bootstrap properties related to JWT validation.
12#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
13pub struct JwtConfig {
14    /// A Json Web Key Store (JWKS) with public keys.
15    ///
16    /// If this is used, Cedarling will no longer try to fetch JWK Stores from
17    /// a trustede identity provider and stick to using the local JWKS.
18    pub jwks: Option<String>,
19    /// Check the signature for all the Json Web Tokens.
20    ///
21    /// This Requires the `iss` claim to be present in all the tokens and
22    /// and the scheme must be `https`.
23    ///
24    /// This setting overrides the `iss` validation settings in the following:
25    ///
26    /// - `access_token_config`
27    /// - `id_token_config`
28    /// - `userinfo_token_config`
29    pub jwt_sig_validation: bool,
30    /// Whether to check the status of the JWT.
31    ///
32    /// On startup, the Cedarling will fetch and retreive the latest Status List
33    /// JWT from the `.well-known/openid-configuration` via the `status_list_endpoint`
34    /// claim and cache it.
35    ///
36    /// See the [`IETF Draft`] for more info.
37    ///
38    /// [`IETF Draft`]: https://datatracker.ietf.org/doc/draft-ietf-oauth-status-list/
39    pub jwt_status_validation: bool,
40    /// Only tokens signed with algorithms in this list can be valid.
41    pub signature_algorithms_supported: HashSet<Algorithm>,
42}
43
44/// Validation options related to JSON Web Tokens (JWT).
45///
46/// This struct provides the configuration for validating common JWT claims (`iss`,
47/// `aud`, `sub`, `jti`, `exp`, `nbf`) across different types of JWTs.
48///
49/// The default configuration for Access Tokens, ID Tokens, and Userinfo Tokens
50/// can be easily instantiated via the provided methods.
51#[derive(Debug, Default, PartialEq)]
52pub struct TokenValidationConfig {
53    /// Requires the `iss` claim to be present in the JWT and the scheme
54    /// must be `https`.
55    pub iss_validation: bool,
56    /// Requires the `aud` claim to be present in the JWT.
57    pub aud_validation: bool,
58    /// Requires the `sub` claim to be present in the JWT.
59    pub sub_validation: bool,
60    /// Requires the `jti` claim to be present in the JWT.
61    pub jti_validation: bool,
62    /// Requires the `iat` claim to be present in the JWT.
63    pub iat_validation: bool,
64    /// Requires the `exp` claim to be present in the JWT and the current
65    /// timestamp isn't past the specified timestamp in the token.
66    pub exp_validation: bool,
67    /// Requires the `nbf` claim to be present in the JWT.
68    pub nbf_validation: bool,
69}
70
71impl TokenValidationConfig {
72    /// Collects all the required claims into a HashSet.
73    pub fn required_claims(&self) -> HashSet<Box<str>> {
74        let mut req_claims = HashSet::new();
75        if self.iss_validation {
76            req_claims.insert("iss".into());
77        }
78        if self.aud_validation {
79            req_claims.insert("aud".into());
80        }
81        if self.sub_validation {
82            req_claims.insert("sub".into());
83        }
84        if self.jti_validation {
85            req_claims.insert("jti".into());
86        }
87        if self.iat_validation {
88            req_claims.insert("iat".into());
89        }
90        if self.exp_validation {
91            req_claims.insert("exp".into());
92        }
93        if self.nbf_validation {
94            req_claims.insert("nbf".into());
95        }
96        req_claims
97    }
98
99    /// Returns a default configuration for validating Access Tokens.
100    ///
101    /// This configuration requires the following:
102    /// - `iss` (Issuer)
103    /// - `jti` (JWT ID)
104    /// - `exp` (Expiration)
105    pub fn access_token() -> Self {
106        Self {
107            iss_validation: true,
108            jti_validation: true,
109            exp_validation: true,
110            nbf_validation: false,
111            aud_validation: false,
112            sub_validation: false,
113            iat_validation: false,
114        }
115    }
116
117    /// Returns a default configuration for validating ID Tokens.
118    ///
119    /// This configuration requires the following:
120    /// - `iss` (Issuer)
121    /// - `aud` (Audience)
122    /// - `sub` (Subject)
123    /// - `exp` (Expiration)
124    pub fn id_token() -> Self {
125        Self {
126            iss_validation: true,
127            aud_validation: true,
128            sub_validation: true,
129            exp_validation: true,
130            iat_validation: false,
131            jti_validation: false,
132            nbf_validation: false,
133        }
134    }
135
136    /// Returns a default configuration for validating Userinfo Tokens.
137    ///
138    /// This configuration requires the following:
139    /// - `iss` (issuer)
140    /// - `aud` (audience)
141    /// - `sub` (subject)
142    /// - `exp` (expiration)
143    pub fn userinfo_token() -> Self {
144        Self {
145            iss_validation: true,
146            aud_validation: true,
147            sub_validation: true,
148            exp_validation: true,
149            jti_validation: false,
150            iat_validation: false,
151            nbf_validation: false,
152        }
153    }
154}

156impl Default for JwtConfig {
157    /// Cedarling will use the strictest validation options by default.
158    fn default() -> Self {
159        Self {
160            jwks: None,
161            jwt_sig_validation: true,
162            jwt_status_validation: true,
163            signature_algorithms_supported: HashSet::new(),
164        }
165    }
166}
167
168impl JwtConfig {
169    /// Creates a new `JwtConfig` instance with validation turned off for all tokens.
170    pub fn new_without_validation() -> Self {
171        Self {
172            jwks: None,
173            jwt_sig_validation: false,
174            jwt_status_validation: false,
175            signature_algorithms_supported: HashSet::new(),
176        }
177        .allow_all_algorithms()
178    }
179
180    /// Adds all supported algorithms to to `signature_algorithms_supported`.
181    pub fn allow_all_algorithms(mut self) -> Self {
182        self.signature_algorithms_supported = HashSet::from_iter([
183            Algorithm::HS256,
184            Algorithm::HS384,
185            Algorithm::HS512,
186            Algorithm::ES256,
187            Algorithm::ES384,
188            Algorithm::RS256,
189            Algorithm::RS384,
190            Algorithm::RS512,
191            Algorithm::PS256,
192            Algorithm::PS384,
193            Algorithm::PS512,
194            Algorithm::EdDSA,
195        ]);
196        self
197    }
198}

200/// Raw JWT config
201pub struct JwtConfigRaw {
202    /// JWKS
203    pub jwks: Option<String>,
204    /// JWT signature validation
205    pub jwt_sig_validation: bool,
206    /// JWT status validation
207    pub jwt_status_validation: bool,
208    /// Supported signature algorithms
209    pub signature_algorithms_supported: Vec<String>,
210}
211
212impl From<JwtConfigRaw> for JwtConfig {
213    fn from(raw: JwtConfigRaw) -> Self {
214        let mut supported_algorithms = HashSet::new();
215        let mut unsupported_algorithms = Vec::new();
216
217        for alg in raw.signature_algorithms_supported {
218            let algorithm = match alg.as_str() {
219                "HS256" => Some(Algorithm::HS256),
220                "HS384" => Some(Algorithm::HS384),
221                "HS512" => Some(Algorithm::HS512),
222                "RS256" => Some(Algorithm::RS256),
223                "RS384" => Some(Algorithm::RS384),
224                "RS512" => Some(Algorithm::RS512),
225                "ES256" => Some(Algorithm::ES256),
226                "ES384" => Some(Algorithm::ES384),
227                "PS256" => Some(Algorithm::PS256),
228                "PS384" => Some(Algorithm::PS384),
229                "PS512" => Some(Algorithm::PS512),
230                "EdDSA" => Some(Algorithm::EdDSA),
231                _ => {
232                    unsupported_algorithms.push(alg);
233                    None
234                }
235            };
236            
237            if let Some(alg) = algorithm {
238                supported_algorithms.insert(alg);
239            }
240        }
241
242        // Log warnings for unsupported algorithms
243        if !unsupported_algorithms.is_empty() {
244            eprintln!("Warning: Unsupported JWT signature algorithms were ignored: {}", 
245                     unsupported_algorithms.join(", "));
246        }
247
248        Self {
249            jwks: raw.jwks,
250            jwt_sig_validation: raw.jwt_sig_validation,
251            jwt_status_validation: raw.jwt_status_validation,
252            signature_algorithms_supported: supported_algorithms,
253        }
254    }
