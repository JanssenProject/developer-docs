21const STORAGE_MUTEX_EXPECT_MESSAGE: &str = "MemoryLogger storage mutex should unlock";

23/// A logger that store logs in-memory.
24pub(crate) struct MemoryLogger {
25    storage: Mutex<SparKV<serde_json::Value>>,
26    log_level: LogLevel,
27    pdp_id: PdpID,
28    app_name: Option<ApplicationName>,
29}
30
31impl MemoryLogger {
32    pub fn new(
33        config: MemoryLogConfig,
34        log_level: LogLevel,
35        pdp_id: PdpID,
36        app_name: Option<ApplicationName>,
37    ) -> Self {
38        let default_config: ConfigSparKV = Default::default();
39
40        let sparkv_config = ConfigSparKV {
41            default_ttl: Duration::new(
42                config.log_ttl.try_into().expect("u64 that fits in a i64"),
43                0,
44            )
45            .expect("a valid duration"),
46            max_items: config.max_items.unwrap_or(default_config.max_items),
47            max_item_size: config.max_item_size.unwrap_or(default_config.max_item_size),
48            ..Default::default()
49        };
50
51        MemoryLogger {
52            storage: Mutex::new(SparKV::with_config_and_sizer(
53                sparkv_config,
54                Some(calculate_memory_usage),
55            )),
56            log_level,
57            pdp_id,
58            app_name,
59        }
60    }
61}

63/// In case of failure in MemoryLogger, log to stderr where supported.
64/// On WASM, stderr is not supported, so log to whatever the wasm logger uses.
65mod fallback {
66    use crate::LogLevel;
67    use crate::app_types::{ApplicationName, PdpID};
68    use crate::log::log_strategy::LogStrategyLogger;
69    use crate::log::stdout_logger::StdOutLogger;

71    /// conform to Loggable requirement imposed by LogStrategy
72    #[derive(serde::Serialize, Clone)]
73    struct StrWrap<'a>(&'a str);
74
75    impl crate::log::interface::Indexed for StrWrap<'_> {
76        fn get_id(&self) -> uuid7::Uuid {
77            crate::log::log_entry::gen_uuid7()
78        }
79
80        fn get_additional_ids(&self) -> Vec<uuid7::Uuid> {
81            Vec::new()
82        }
83
84        fn get_tags(&self) -> Vec<&str> {
85            Vec::new()
86        }
87    }
88
89    impl crate::log::interface::Loggable for StrWrap<'_> {
90        fn get_log_level(&self) -> Option<LogLevel> {
91            // These must always be logged.
92            Some(LogLevel::TRACE)
93        }
94    }
95
96    /// Fetch the correct logger. That takes some work, and it's done on every
97    /// call. But this is a fallback logger, so it is not intended to be used
98    /// often, and in this case correctness and non-fallibility are far more
99    /// important than performance.
100    ///
101    /// # Panics
102    ///
103    /// Panics when:
104    /// - A runtime to initialize a new LogStrategy could not be built.
105    /// - A fallback logger could not be initialized.
106    pub fn log(msg: &str, pdp_id: &PdpID, app_name: &Option<ApplicationName>) {
107        // level is so that all messages passed here are logged.
108        let logger = StdOutLogger::new(LogLevel::TRACE);
109
110        // This should always be a LogStrategy::StdOut(StdOutLogger)
111        let log_strategy = crate::log::LogStrategy::new_with_logger(
112            LogStrategyLogger::StdOut(logger),
113            *pdp_id,
114            app_name.clone(),
115            None,
116        );
117
118        use crate::log::interface::LogWriter;
119        // a string is always serializable
120        log_strategy.log_any(StrWrap(msg))
121    }
122}

124fn to_json_value<T: Loggable>(entry: &T) -> Value {
125    match serde_json::to_value(entry) {
126        Ok(json) => json,
127        Err(err) => {
128            let err_msg = format!("failed to serialize log entry to JSON: {err}");
129            serde_json::to_value(ErrorLogEntry::from_loggable(entry, err_msg.clone()))
130                .expect(&err_msg)
131        },
132    }
133}

135// Implementation of LogWriter
136impl LogWriter for MemoryLogger {
137    fn log_any<T: Loggable>(&self, entry: T) {
138        if !entry.can_log(self.log_level) {
139            // do nothing
140            return;
141        }
142
143        let entry_id = entry.get_id().to_string();
144        let index_keys = entry.get_index_keys();
145        let json = to_json_value(&entry);
146
147        let mut storage = self.storage.lock().expect(STORAGE_MUTEX_EXPECT_MESSAGE);
148
149        let set_result = storage.set(&entry_id, json, index_keys.as_slice());
150
151        let err = match set_result {
152            Ok(_) => return,
153            Err(Error::CapacityExceeded) => {
154                // remove oldest key and try again
155
156                let key_to_delete = storage
157                    .get_oldest_key_by_expiration()
158                    .map(|exp_entry| exp_entry.key.clone());
159
160                if let Some(key) = key_to_delete {
161                    storage.pop(&key);
162                }
163
164                // It should be rare case, so instead of cloning the whole entry,
165                // in success case we convert raw value to json (here).
166                // Or we should use Rc<LogEntry> and use Rc::clone() instead.
167                let json = to_json_value(&entry);
168
169                // set_again
170                if let Err(err) = storage.set(&entry_id, json, index_keys.as_slice()) {
171                    err
172                } else {
173                    return;
174                }
175            },
176            Err(err) => err,
177        };
178
179        fallback::log(
180            &format!("could not store LogEntry to memory: {err:?}"),
181            &self.pdp_id,
182            &self.app_name,
183        );
184    }
185}

187// Implementation of LogStorage
188impl LogStorage for MemoryLogger {
189    fn pop_logs(&self) -> Vec<serde_json::Value> {
190        self.storage
191            .lock()
192            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
193            .drain()
194            .map(|(_k, value)| value)
195            .collect()
196    }
197
198    fn get_log_by_id(&self, id: &str) -> Option<serde_json::Value> {
199        self.storage
200            .lock()
201            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
202            .get(id)
203            .cloned()
204    }
205
206    fn get_log_ids(&self) -> Vec<String> {
207        self.storage
208            .lock()
209            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
210            .get_keys()
211    }
212
213    fn get_logs_by_tag(&self, tag: &str) -> Vec<serde_json::Value> {
214        self.storage
215            .lock()
216            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
217            .get_by_index_key(tag)
218            .map(|v| v.to_owned())
219            .collect()
220    }
221
222    fn get_logs_by_request_id(&self, request_id: &str) -> Vec<serde_json::Value> {
223        self.storage
224            .lock()
225            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
226            .get_by_index_key(request_id)
227            .map(|v| v.to_owned())
228            .collect()
229    }
230
231    fn get_logs_by_request_id_and_tag(
232        &self,
233        request_id: &str,
234        tag: &str,
235    ) -> Vec<serde_json::Value> {
236        let key = composite_key(request_id, tag);
237
238        self.storage
239            .lock()
240            .expect(STORAGE_MUTEX_EXPECT_MESSAGE)
241            .get_by_index_key(&key)
242            .map(|v| v.to_owned())
243            .collect()
244    }
245}

265    #[test]
266    fn test_log_and_get_logs() {
267        let pdp_id = PdpID::new();
268        let app_name = None;
269        let logger = create_memory_logger(pdp_id, app_name.clone());
270
271        // create log entries
272        let entry1 = LogEntry::new_with_data(LogType::Decision, None)
273            .set_message("some message".to_string())
274            .set_auth_info(AuthorizationLogInfo {
275                action: "test_action".to_string(),
276                resource: "test_resource".to_string(),
277                context: serde_json::json!({}),
278                authorize_info: Default::default(),
279                authorized: true,
280                entities: serde_json::json!({}),
281            });
282
283        let entry2 = LogEntry::new_with_data(LogType::System, None);
284
285        assert!(
286            entry1.base.id < entry2.base.id,
287            "entry1.base.id should be lower than in entry2"
288        );
289
290        // log entries
291        logger.log_any(entry1.clone());
292        logger.log_any(entry2.clone());
293
294        let entry1_json = json!(entry1.clone());
295        let entry2_json = json!(entry2.clone());
296
297        // check that we have two entries in the log database
298        assert_eq!(logger.get_log_ids().len(), 2);
299        assert_eq!(
300            logger.get_log_by_id(&entry1.get_id().to_string()).unwrap(),
301            entry1_json,
302            "Failed to get log entry by id"
303        );
304        assert_eq!(
305            logger.get_log_by_id(&entry2.get_id().to_string()).unwrap(),
306            entry2_json,
307            "Failed to get log entry by id"
308        );
309
310        // get logs using `pop_logs`
311        let logs = logger.pop_logs();
312        assert_eq!(logs.len(), 2);
313        assert_eq!(logs[0], entry1_json, "First log entry is incorrect");
314        assert_eq!(logs[1], entry2_json, "Second log entry is incorrect");
315
316        // check that we have no entries in the log database
317        assert!(
318            logger.get_log_ids().is_empty(),
319            "Logs were not fully popped"
320        );
321    }

355        let request_id = gen_uuid7();
356
357        let logger = MemoryLogger::new(
358            MemoryLogConfig {
359                log_ttl: 10,
360                max_item_size: None,
361                max_items: None,

367
368        let entry_decision = LogEntry::new_with_data(LogType::Decision, None);
369        logger.log_any(entry_decision);
370
371        let entry_system_info =
372            LogEntry::new_with_data(LogType::System, Some(request_id)).set_level(LogLevel::INFO);
373        logger.log_any(entry_system_info);
374
375        let entry_system_debug =
376            LogEntry::new_with_data(LogType::System, Some(request_id)).set_level(LogLevel::DEBUG);
377        logger.log_any(entry_system_debug);
378
379        let entry_metric = LogEntry::new_with_data(LogType::Metric, None);
380        logger.log_any(entry_metric);
381
382        // without request id

384            LogEntry::new_with_data(LogType::System, None).set_level(LogLevel::WARN);
385        logger.log_any(entry_system_warn);
386
387        assert!(

403            "1 log entries should be present for request id: {request_id} and debug level"
