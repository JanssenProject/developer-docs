18pub(crate) struct LogStrategy {
19    logger: LogStrategyLogger,
20    pdp_id: PdpID,
21    app_name: Option<ApplicationName>,
22    lock_service: RwLock<Option<LockService>>,
23}
24
25/// LogStrategy implements strategy pattern for logging.
26/// It is used to provide a single point of access for logging and same api for different loggers.
27#[allow(clippy::large_enum_variant)]
28pub(crate) enum LogStrategyLogger {
29    Off(NopLogger),
30    MemoryLogger(MemoryLogger),
31    StdOut(StdOutLogger),
32}
33
34impl LogStrategy {
35    /// Creates a new `LogStrategy` based on the provided configuration.
36    /// Initializes the corresponding logger accordingly.
37    pub async fn new(
38        config: &LogConfig,
39        pdp_id: PdpID,
40        app_name: Option<ApplicationName>,
41    ) -> Result<Self, InitLockServiceError> {
42        let logger = match &config.log_type {
43            LogTypeConfig::Off => LogStrategyLogger::Off(NopLogger),
44            LogTypeConfig::Memory(memory_config) => {
45                LogStrategyLogger::MemoryLogger(MemoryLogger::new(
46                    memory_config.clone(),
47                    config.log_level,
48                    pdp_id,
49                    app_name.clone(),
50                ))
51            },
52            LogTypeConfig::StdOut => LogStrategyLogger::StdOut(StdOutLogger::new(config.log_level)),
53        };
54        Ok(Self {
55            logger,
56            pdp_id,
57            app_name,
58            lock_service: RwLock::new(None),
59        })
60    }
61
62    pub fn new_with_logger(
63        logger: LogStrategyLogger,
64        pdp_id: PdpID,
65        app_name: Option<ApplicationName>,
66        lock_service: Option<LockService>,
67    ) -> Self {
68        Self {
69            logger,
70            pdp_id,
71            app_name,
72            lock_service: RwLock::new(lock_service),
73        }
74    }
75
76    pub(crate) fn logger(&self) -> &LogStrategyLogger {
77        &self.logger
78    }
79
80    pub fn set_lock_service(&self, lock_service: LockService) {
81        *self
82            .lock_service
83            .write()
84            .expect("obtain lock_service write lock") = Some(lock_service);
85    }
86
87    pub async fn shut_down(&self) {
88        let lock = self
89            .lock_service
90            .write()
91            .expect("obtain lock_service write lock")
92            .take();
93
94        if let Some(mut lock_service) = lock {
95            lock_service.shut_down().await
96        }
97    }
98}

100#[derive(Serialize, PartialEq, Clone)]
101pub(crate) struct LogEntryWithClientInfo<Entry: Loggable> {
102    #[serde(flatten)]
103    entry: Entry,
104    pdp_id: PdpID,
105    #[serde(rename = "application_id", skip_serializing_if = "Option::is_none")]
106    app_name: Option<ApplicationName>,
107}
108
109impl<Entry: Loggable> Indexed for LogEntryWithClientInfo<Entry> {
110    fn get_id(&self) -> uuid7::Uuid {
111        self.entry.get_id()
112    }
113
114    fn get_additional_ids(&self) -> Vec<uuid7::Uuid> {
115        self.entry.get_additional_ids()
116    }
117
118    fn get_tags(&self) -> Vec<&str> {
119        self.entry.get_tags()
120    }
121}
122
123impl<Entry: Loggable + Indexed> Loggable for LogEntryWithClientInfo<Entry> {
124    fn get_log_level(&self) -> Option<super::LogLevel> {
125        self.entry.get_log_level()
126    }
127}

129// Implementation of LogWriter
130impl LogWriter for LogStrategy {
131    fn log_any<T: Loggable>(&self, entry: T) {
132        let entry =
133            LogEntryWithClientInfo::from_loggable(entry, self.pdp_id, self.app_name.clone());
134        if let Some(lock_service) = self
135            .lock_service
136            .read()
137            .expect("obtain lock_service read lock")
138            .as_ref()
139        {
140            lock_service.log_any(entry.clone());
141        }
142        match &self.logger {
143            LogStrategyLogger::Off(log) => log.log_any(entry),
144            LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.log_any(entry),
145            LogStrategyLogger::StdOut(std_out_logger) => std_out_logger.log_any(entry),
146        }
147    }
148}
149
150impl<E: Loggable> LogEntryWithClientInfo<E> {
151    pub fn from_loggable(entry: E, pdp_id: PdpID, app_name: Option<ApplicationName>) -> Self {
152        Self {
153            entry,
154            pdp_id,
155            app_name,
156        }
157    }
158}

160// Implementation of LogStorage
161// for cases where we not use memory logger we return default value
162impl LogStorage for LogStrategy {
163    fn pop_logs(&self) -> Vec<serde_json::Value> {
164        match &self.logger {
165            LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.pop_logs(),
166            _ => Vec::new(),
167        }
168    }
169
170    fn get_log_by_id(&self, id: &str) -> Option<serde_json::Value> {
171        match &self.logger {
172            LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.get_log_by_id(id),
173            _ => None,
174        }
175    }
176
177    fn get_log_ids(&self) -> Vec<String> {
178        match &self.logger {
179            LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.get_log_ids(),
180            _ => Vec::new(),
181        }
182    }
183
184    fn get_logs_by_tag(&self, tag: &str) -> Vec<serde_json::Value> {
185        match &self.logger {
186            LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.get_logs_by_tag(tag),
187            _ => Vec::new(),
188        }
189    }
190
191    fn get_logs_by_request_id(&self, request_id: &str) -> Vec<serde_json::Value> {
192        match &self.logger {
193            LogStrategyLogger::MemoryLogger(memory_logger) => {
194                memory_logger.get_logs_by_request_id(request_id)
195            },
196            // always empty vector for not MemoryLogger
197            _ => Vec::new(),
198        }
199    }
200
201    fn get_logs_by_request_id_and_tag(&self, id: &str, tag: &str) -> Vec<serde_json::Value> {
202        match &self.logger {
203            LogStrategyLogger::MemoryLogger(memory_logger) => {
204                memory_logger.get_logs_by_request_id_and_tag(id, tag)
205            },
206            _ => Vec::new(),
207        }
208    }
209}
