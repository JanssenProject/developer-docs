6use std::str::FromStr;
7
8use serde::{Deserialize, Serialize};
9
10/// Log levels
11/// Fatal level is the highest, trace is lowest
12#[derive(
13    Debug,
14    PartialEq,
15    PartialOrd,
16    Eq,
17    Ord,
18    Copy,
19    Clone,
20    Default,
21    derive_more::Display,
22    Deserialize,
23    Serialize,
24    strum::IntoStaticStr,
25)]
26pub enum LogLevel {
27    /// Fatal level
28    FATAL = 5,
29    /// Error level
30    ERROR = 4,
31    /// Warn level
32    #[default]
33    WARN = 3,
34    /// Info level
35    INFO = 2,
36    /// Debug level
37    DEBUG = 1,
38    /// Trace level
39    TRACE = 0,
40}
41
42impl FromStr for LogLevel {
43    type Err = String;
44
45    fn from_str(s: &str) -> Result<Self, Self::Err> {
46        match s {
47            "FATAL" => Ok(LogLevel::FATAL),
48            "ERROR" => Ok(LogLevel::ERROR),
49            "WARN" => Ok(LogLevel::WARN),
50            "INFO" => Ok(LogLevel::INFO),
51            "DEBUG" => Ok(LogLevel::DEBUG),
52            "TRACE" => Ok(LogLevel::TRACE),
53            _ => Err(format!("Invalid log level: {}", s)),
54        }
55    }
56}
