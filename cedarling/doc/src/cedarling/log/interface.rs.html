19/// Log Writer
20/// interface for logging events
21pub(crate) trait LogWriter {
22    /// log any serializable entry that not suitable for [`LogEntry`]
23    fn log_any<T: Loggable>(&self, entry: T);
24
25    fn log_fn<F, R>(&self, log_fn: LoggableFn<F>)
26    where
27        R: Loggable,
28        F: Fn(BaseLogEntry) -> R;
29}
30
31impl LogWriter for Option<Arc<LogStrategy>> {
32    fn log_any<T: Loggable>(&self, entry: T) {
33        if let Some(logger) = self.as_ref() {
34            logger.log_any(entry);
35        }
36    }
37
38    fn log_fn<F, R>(&self, log_fn: LoggableFn<F>)
39    where
40        R: Loggable,
41        F: Fn(BaseLogEntry) -> R,
42    {
43        if let Some(logger) = self.as_ref() {
44            logger.log_fn(log_fn);
45        }
46    }
47}
48
49impl LogWriter for Option<&Arc<LogStrategy>> {
50    fn log_any<T: Loggable>(&self, entry: T) {
51        if let Some(logger) = self.as_ref() {
52            logger.log_any(entry);
53        }
54    }
55
56    fn log_fn<F, R>(&self, log_fn: LoggableFn<F>)
57    where
58        R: Loggable,
59        F: Fn(BaseLogEntry) -> R,
60    {
61        if let Some(logger) = self.as_ref() {
62            logger.log_fn(log_fn);
63        }
64    }
65}
66
67impl LogWriter for Option<Weak<LogStrategy>> {
68    fn log_any<T: Loggable>(&self, entry: T) {
69        if let Some(log_strategy) = self.as_ref().and_then(std::sync::Weak::upgrade) {
70            log_strategy.as_ref().log_any(entry);
71            return;
72        }
73
74        // we log the error manually to stdout if the logger is gone
75        StdOutLogger::new(LogLevel::INFO, StdOutLoggerMode::Immediate).log_any(entry);
76    }
77
78    fn log_fn<F, R>(&self, log_fn: LoggableFn<F>)
79    where
80        R: Loggable,
81        F: Fn(BaseLogEntry) -> R,
82    {
83        if let Some(log_strategy) = self.as_ref().and_then(std::sync::Weak::upgrade) {
84            log_strategy.as_ref().log_fn(log_fn);
85            return;
86        }
87
88        let entry = log_fn.build();
89        // we log the error manually to stdout if the logger is gone
90        StdOutLogger::new(LogLevel::INFO, StdOutLoggerMode::Immediate).log_any(entry);
91    }
92}

96pub(crate) fn composite_key(id: &str, tag: &str) -> String {
97    [id, tag].join(SEPARATOR)
98}

100pub(crate) trait Indexed {
101    /// Get unique ID of entity
102    //  Is used in memory logger
103    fn get_id(&self) -> Uuid;
104
105    /// List of additional ids that entity can be related
106    //  Is used in memory logger
107    fn get_additional_ids(&self) -> Vec<Uuid>;
108
109    /// List of `tags` that entity can be related
110    //  Is used in memory logger
111    fn get_tags(&self) -> Vec<&str>;
112
113    fn get_index_keys(&self) -> Vec<String> {
114        let tags = self.get_tags();
115
116        let additional_ids = self
117            .get_additional_ids()
118            .into_iter()
119            .map(|v| v.to_string())
120            .collect::<Vec<String>>();
121
122        let additional_id_and_tag = additional_ids
123            .iter()
124            .flat_map(|id| tags.iter().map(move |tag| composite_key(id, tag)))
125            .collect::<Vec<String>>();
126
127        let tags_iter = tags
128            .into_iter()
129            .map(Into::<String>::into)
130            .collect::<Vec<String>>();
131
132        let mut result = Vec::with_capacity(
133            additional_ids.len() + additional_id_and_tag.len() + tags_iter.len(),
134        );
135
136        result.extend(additional_ids);
137        result.extend(additional_id_and_tag);
138        result.extend(tags_iter);
139
140        result
141    }
142}

144// static means that entities owns value or has reference with 'static lifetime
145pub(crate) trait Loggable:
146    serde::Serialize + Indexed + Clone + Send + Sync + Sized + 'static
147{
148    /// get log level for entity
149    /// not all log entities have log level, only when `log_kind` == `System`
150    fn get_log_level(&self) -> Option<LogLevel>;
151
152    /// check if entry can log to logger
153    // default implementation of method
154    // is used to avoid boilerplate code
155    fn can_log(&self, logger_level: LogLevel) -> bool {
156        can_log(self.get_log_level(), logger_level)
157    }
158}
159
160/// check if entry can log to logger
161// default implementation of method
162// is used to avoid boilerplate code
163pub(super) fn can_log(entity_level: Option<LogLevel>, logger_level: LogLevel) -> bool {
164    if let Some(entry_log_level) = entity_level {
165        // higher level is more important, ie closer to fatal
166        logger_level <= entry_log_level
167    } else {
168        // if `.get_log_level` return None
169        // it means that `log_kind` != `System` and we should log it
170        true
171    }
172}

174/// Log Storage
175/// interface for getting log entries from the storage
176pub trait LogStorage {
177    /// Return logs and remove them from the storage
178    fn pop_logs(&self) -> Vec<serde_json::Value>;
179
180    /// Get specific log entry
181    fn get_log_by_id(&self, id: &str) -> Option<serde_json::Value>;
182
183    /// Returns a list of all log ids
184    fn get_log_ids(&self) -> Vec<String>;
185
186    /// Get logs by tag, like `log_kind` or `log level`.
187    /// Tag can be `log_kind`, `log_level`.
188    fn get_logs_by_tag(&self, tag: &str) -> Vec<serde_json::Value>;
189
190    /// Get logs by `request_id`.
191    /// Return log entries that match the given `request_id`.
192    fn get_logs_by_request_id(&self, request_id: &str) -> Vec<serde_json::Value>;
193
194    /// Get log by `request_id` and tag, like composite key `request_id` + `log_kind`.
195    /// Tag can be `log_kind`, `log_level`.
196    /// Return log entries that match the given `request_id` and tag.
197    fn get_logs_by_request_id_and_tag(&self, request_id: &str, tag: &str)
198    -> Vec<serde_json::Value>;
199}
