15/// Log Writer
16/// interface for logging events
17pub(crate) trait LogWriter {
18    /// log any serializable entry that not suitable for [`LogEntry`]
19    fn log_any<T: Loggable>(&self, entry: T);
20}
21
22impl LogWriter for Option<Arc<LogStrategy>> {
23    fn log_any<T: Loggable>(&self, entry: T) {
24        if let Some(logger) = self.as_ref() {
25            logger.log_any(entry);
26        }
27    }
28}
29
30impl LogWriter for Option<Weak<LogStrategy>> {
31    fn log_any<T: Loggable>(&self, entry: T) {
32        if let Some(log_strategy) = self.as_ref().and_then(|l| l.upgrade()) {
33            match log_strategy.logger() {
34                LogStrategyLogger::Off(log) => log.log_any(entry),
35                LogStrategyLogger::MemoryLogger(memory_logger) => memory_logger.log_any(entry),
36                LogStrategyLogger::StdOut(std_out_logger) => std_out_logger.log_any(entry),
37            }
38            return;
39        }
40
41        // we log the error manually to stdout if the logger is gone
42        
43        let log = match serde_json::to_value(&entry) {
44            Ok(json) => json.to_string(),
45            Err(err) => {
46                let err_msg = format!("failed to serialize log entry to JSON: {err}");
47                serde_json::to_value(entry).expect(&err_msg).to_string()
48            },
49        };
50
51        println!("{log}");
52    }
53}

55const SEPARATOR: &str = "__";
56
57pub(crate) fn composite_key(id: &str, tag: &str) -> String {
58    [id, tag].join(SEPARATOR)
59}
60
61pub(crate) trait Indexed {
62    /// Get unique ID of entity
63    //  Is used in memory logger
64    fn get_id(&self) -> Uuid;
65
66    /// List of additional ids that entity can be related
67    //  Is used in memory logger
68    fn get_additional_ids(&self) -> Vec<Uuid>;
69
70    /// List of `tags` that entity can be related
71    //  Is used in memory logger
72    fn get_tags(&self) -> Vec<&str>;
73
74    fn get_index_keys(&self) -> Vec<String> {
75        let tags = self.get_tags();
76
77        let additional_ids = self
78            .get_additional_ids()
79            .into_iter()
80            .map(|v| v.to_string())
81            .collect::<Vec<String>>();
82
83        let additional_id_and_tag = additional_ids
84            .iter()
85            .flat_map(|id| tags.iter().map(move |tag| composite_key(id, tag)))
86            .collect::<Vec<String>>();
87
88        let tags_iter = tags
89            .into_iter()
90            .map(Into::<String>::into)
91            .collect::<Vec<String>>();
92
93        let mut result = Vec::with_capacity(
94            additional_ids.len() + additional_id_and_tag.len() + tags_iter.len(),
95        );
96
97        result.extend(additional_ids);
98        result.extend(additional_id_and_tag);
99        result.extend(tags_iter);
100
101        result
102    }
103}

105pub(crate) trait Loggable: serde::Serialize + Indexed + Clone {
106    /// get log level for entity
107    /// not all log entities have log level, only when `log_kind` == `System`
108    fn get_log_level(&self) -> Option<LogLevel>;
109
110    /// check if entry can log to logger
111    // default implementation of method
112    // is used to avoid boilerplate code
113    fn can_log(&self, logger_level: LogLevel) -> bool {
114        if let Some(entry_log_level) = self.get_log_level() {
115            // higher level is more important, ie closer to fatal
116            logger_level <= entry_log_level
117        } else {
118            // if `.get_log_level` return None
119            // it means that `log_kind` != `System` and we should log it
120            true
121        }
122    }
123}

125/// Log Storage
126/// interface for getting log entries from the storage
127pub trait LogStorage {
128    /// Return logs and remove them from the storage
129    fn pop_logs(&self) -> Vec<serde_json::Value>;
130
131    /// Get specific log entry
132    fn get_log_by_id(&self, id: &str) -> Option<serde_json::Value>;
133
134    /// Returns a list of all log ids
135    fn get_log_ids(&self) -> Vec<String>;
136
137    /// Get logs by tag, like `log_kind` or `log level`.
138    /// Tag can be `log_kind`, `log_level`.
139    fn get_logs_by_tag(&self, tag: &str) -> Vec<serde_json::Value>;
140
141    /// Get logs by request_id.
142    /// Return log entries that match the given request_id.
143    fn get_logs_by_request_id(&self, request_id: &str) -> Vec<serde_json::Value>;
144
145    /// Get log by request_id and tag, like composite key `request_id` + `log_kind`.
146    /// Tag can be `log_kind`, `log_level`.
147    /// Return log entries that match the given request_id and tag.
148    fn get_logs_by_request_id_and_tag(&self, request_id: &str, tag: &str)
149    -> Vec<serde_json::Value>;
150}
