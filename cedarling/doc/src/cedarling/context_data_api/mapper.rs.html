29/// Represents a detected extension type with its parsed value.
30#[derive(Debug, Clone, PartialEq)]
31pub enum ExtensionValue {
32    /// An IP address (IPv4 or IPv6) or CIDR range
33    IpAddr(String),
34    /// A fixed-precision decimal number (up to 4 decimal places)
35    Decimal(String),
36    /// An instant of time with millisecond precision (RFC 3339 / ISO 8601)
37    DateTime(String),
38    /// A duration of time with millisecond precision
39    Duration(String),
40}

42/// Mapper for bidirectional JSON ↔ Cedar value conversion.
43///
44/// Provides methods to convert between `serde_json::Value` and Cedar's
45/// `RestrictedExpression`, with support for all Cedar data types including
46/// extension types (IP addresses, decimals).
47#[derive(Debug, Clone)]
48pub struct CedarValueMapper {
49    /// Whether to auto-detect extension types from string patterns
50    auto_detect_extensions: bool,
51    /// Maximum allowed value size in bytes (0 = no limit)
52    max_value_size: usize,
53}
54
55impl Default for CedarValueMapper {
56    fn default() -> Self {
57        Self::new()
58    }
59}

89    /// Convert a JSON value to a Cedar `RestrictedExpression`.
90    ///
91    /// Supports all Cedar primitive types, collections, and extension types.
92    /// Null values are not supported and will return an error.
93    pub fn json_to_cedar(
94        &self,
95        value: &Value,
96    ) -> Result<Option<RestrictedExpression>, ValueMappingError> {
97        // Check size limit
98        if self.max_value_size > 0 {
99            let size = Self::estimate_value_size(value);
100            if size > self.max_value_size {
101                return Err(ValueMappingError::ValueTooLarge {
102                    size,
103                    limit: self.max_value_size,
104                });
105            }
106        }
107
108        self.convert_value(value)
109    }

111    /// Convert a JSON value to Cedar, returning the inferred Cedar type.
112    ///
113    /// This is useful when you need both the expression and type information.
114    pub fn json_to_cedar_with_type(
115        &self,
116        value: &Value,
117    ) -> Result<Option<(RestrictedExpression, CedarType)>, ValueMappingError> {
118        let cedar_type = CedarType::from_value(value);
119        let expr = self.json_to_cedar(value)?;
120        Ok(expr.map(|e| (e, cedar_type)))

139    /// Access a nested value using dot notation.
140    pub fn get_nested<'a>(value: &'a Value, path: &str) -> Result<&'a Value, ValueMappingError> {
141        if path.is_empty() {
142            return Ok(value);
143        }
144
145        let mut current = value;
146        for component in path.split('.') {
147            if component.is_empty() {
148                return Err(ValueMappingError::InvalidPath {
149                    path: path.to_string(),
150                });
151            }
152
153            current = match current {
154                Value::Object(obj) => {
155                    obj.get(component)
156                        .ok_or_else(|| ValueMappingError::PathNotFound {
157                            path: path.to_string(),
158                        })?
159                },
160                Value::Array(arr) => {
161                    // Support numeric indexing for arrays
162                    let index: usize =
163                        component
164                            .parse()
165                            .map_err(|_| ValueMappingError::PathNotFound {
166                                path: path.to_string(),
167                            })?;
168                    arr.get(index)
169                        .ok_or_else(|| ValueMappingError::PathNotFound {
170                            path: path.to_string(),
171                        })?
172                },
173                _ => {
174                    return Err(ValueMappingError::PathNotFound {
175                        path: path.to_string(),
176                    });
177                },
178            };
179        }
180
181        Ok(current)
182    }

184    /// Set a value at a nested path, creating intermediate objects as needed.
185    pub fn set_nested(
186        value: &mut Value,
187        path: &str,
188        new_value: Value,
189    ) -> Result<(), ValueMappingError> {
190        if path.is_empty() {
191            *value = new_value;
192            return Ok(());
193        }
194
195        let components: Vec<&str> = path.split('.').collect();
196        let mut current = value;
197
198        for (i, component) in components.iter().enumerate() {
199            if component.is_empty() {
200                return Err(ValueMappingError::InvalidPath {
201                    path: path.to_string(),
202                });
203            }
204
205            let is_last = i == components.len() - 1;
206
207            if is_last {
208                // Set the value
209                let Value::Object(obj) = current else {
210                    return Err(ValueMappingError::TypeMismatch {
211                        expected: "object".to_string(),
212                        actual: Self::value_type_name(current).to_string(),
213                    });
214                };
215                obj.insert((*component).to_string(), new_value);
216                return Ok(());
217            }
218
219            // Navigate or create intermediate objects
220            let Value::Object(obj) = current else {
221                return Err(ValueMappingError::TypeMismatch {
222                    expected: "object".to_string(),
223                    actual: Self::value_type_name(current).to_string(),
224                });
225            };
226            current = obj
227                .entry((*component).to_string())
228                .or_insert_with(|| Value::Object(Map::new()));
229        }
230
231        Ok(())
232    }

234    /// Detect if a string value represents a Cedar extension type.
235    ///
236    /// Detects the following extension types:
237    /// - `ipaddr`: IP addresses (IPv4/IPv6) and CIDR ranges (e.g., "192.168.1.1", "10.0.0.0/8")
238    /// - `decimal`: Fixed-precision decimals (e.g., "3.14", "-12.345")
239    /// - `datetime`: ISO 8601 / RFC 3339 timestamps (e.g., "2024-10-15T11:35:00Z")
240    /// - `duration`: Duration strings (e.g., "2h30m", "1d12h", "500ms")
241    ///
242    /// See: <https://docs.cedarpolicy.com/policies/syntax-datatypes.html#datatype-extension>
243    #[must_use]
244    pub fn detect_extension(value: &str) -> Option<ExtensionValue> {
245        // Check for plain IP address (IPv4 or IPv6)
246        if IpAddr::from_str(value).is_ok() {
247            return Some(ExtensionValue::IpAddr(value.to_string()));
248        }
249
250        // Check for CIDR notation (e.g., "192.168.1.0/24", "fe80::/10")
251        if let Some((ip_part, prefix_part)) = value.split_once('/')
252            && let Ok(ip) = IpAddr::from_str(ip_part)
253            && let Ok(prefix_len) = prefix_part.parse::<u8>()
254        {
255            // Validate prefix length: 0-32 for IPv4, 0-128 for IPv6
256            let max_prefix = if ip.is_ipv4() { 32 } else { 128 };
257            if prefix_len <= max_prefix {
258                return Some(ExtensionValue::IpAddr(value.to_string()));
259            }
260        }
261
262        // Check for datetime (ISO 8601 / RFC 3339 format)
263        // Examples: "2024-10-15", "2024-10-15T11:35:00Z", "2024-10-15T11:35:00.000+0100"
264        if Self::is_datetime_format(value) {
265            return Some(ExtensionValue::DateTime(value.to_string()));
266        }
267
268        // Check for duration format (e.g., "2h30m", "-1d12h", "500ms")
269        if Self::is_duration_format(value) {
270            return Some(ExtensionValue::Duration(value.to_string()));
271        }
272
273        // Check for decimal (must contain decimal point and be parseable as f64)
274        // Must have exactly one decimal point, not end with it, and reject exponent notation
275        if value.contains('.')
276            && !value.contains('e')
277            && !value.contains('E')
278            && !value.ends_with('.')
279            && value.chars().filter(|&c| c == '.').count() == 1
280        {
281            // Parse as f64 to validate numeric format, but ensure it's fixed-point
282            if value.parse::<f64>().is_ok() {
283                // Additional check: ensure there's at least one digit before and after the decimal point
284                if let Some(dot_pos) = value.find('.') {
285                    let before_dot = &value[..dot_pos];
286                    let after_dot = &value[dot_pos + 1..];
287                    // Require at least one ASCII digit in before_dot (optionally preceded by a single '+' or '-')
288                    let before_has_digit = before_dot.chars().any(|c| c.is_ascii_digit());
289                    let before_valid = if before_dot.is_empty() || before_dot == "+" || before_dot == "-" {
290                        false // Must have at least one digit, not just sign or empty
291                    } else {
292                        // Must have at least one digit, and all chars are digits or a single leading sign
293                        let has_leading_sign = before_dot.starts_with('+') || before_dot.starts_with('-');
294                        let sign_count = before_dot.chars().filter(|c| *c == '+' || *c == '-').count();
295                        before_has_digit
296                            && before_dot.chars().all(|c| c.is_ascii_digit() || c == '+' || c == '-')
297                            && (!has_leading_sign || sign_count == 1)
298                    };
299                    // Require at least one ASCII digit in after_dot, and all chars are digits
300                    let after_ok = !after_dot.is_empty()
301                        && after_dot.chars().all(|c| c.is_ascii_digit())
302                        && after_dot.chars().any(|c| c.is_ascii_digit());
303                    // Both sides must have digits
304                    if before_valid && after_ok {
305                        return Some(ExtensionValue::Decimal(value.to_string()));
306                    }
307                }
308            }
309        }
310
311        None
312    }

314    /// Check if a string looks like an ISO 8601 / RFC 3339 datetime.
315    ///
316    /// Uses real parsing to validate semantic correctness, not just byte patterns.
317    /// Supported formats:
318    /// - "2024-10-15" (date only)
319    /// - "2024-10-15T11:35:00Z" (UTC)
320    /// - "2024-10-15T11:35:00.000Z" (UTC with milliseconds)
321    /// - "2024-10-15T11:35:00+0100" (with timezone offset)
322    /// - "2024-10-15T11:35:00.000+0100" (with timezone and milliseconds)
323    fn is_datetime_format(value: &str) -> bool {
324        use chrono::{DateTime, NaiveDate};
325
326        // Try RFC 3339 parsing first (handles full datetime with timezone)
327        if DateTime::parse_from_rfc3339(value).is_ok() {
328            return true;
329        }
330
331        // Try ISO 8601 format with offset without colon (e.g., "+0100")
332        if DateTime::parse_from_str(value, "%Y-%m-%dT%H:%M:%S%z").is_ok() {
333            return true;
334        }
335
336        // Try ISO 8601 format with offset and fractional seconds
337        if DateTime::parse_from_str(value, "%Y-%m-%dT%H:%M:%S%.f%z").is_ok() {
338            return true;
339        }
340
341        // Try date-only format (YYYY-MM-DD)
342        if NaiveDate::parse_from_str(value, "%Y-%m-%d").is_ok() {
343            return true;
344        }
345
346        false
347    }

349    /// Check if a string looks like a Cedar duration format.
350    ///
351    /// Enforces unit ordering: d > h > m > s > ms (descending rank order).
352    /// Supported formats:
353    /// - "2h30m" (hours and minutes)
354    /// - "-1d12h" (negative, days and hours)
355    /// - "1h30m45s" (hours, minutes, seconds)
356    /// - "500ms" (milliseconds only)
357    /// - "1d" (days only)
358    fn is_duration_format(value: &str) -> bool {
359        use crate::context_data_api::entry::UnitRank;
360
361        if value.is_empty() {
362            return false;
363        }
364
365        let bytes = value.as_bytes();
366        let mut i = 0;
367
368        if bytes[i] == b'-' {
369            i += 1;
370            if i == bytes.len() {
371                return false;
372            }
373        }
374
375        let mut last_rank = UnitRank::Start;
376
377        while i < bytes.len() {
378            let start = i;
379            while i < bytes.len() && bytes[i].is_ascii_digit() {
380                i += 1;
381            }
382            if start == i {
383                return false;
384            }
385
386            let (current_rank, consumed) = match bytes.get(i) {
387                Some(b'd') if last_rank < UnitRank::Days => (UnitRank::Days, 1),
388                Some(b'h') if last_rank < UnitRank::Hours => (UnitRank::Hours, 1),
389                Some(b's') if last_rank < UnitRank::Seconds => (UnitRank::Seconds, 1),
390                Some(b'm') => {
391                    if i + 1 < bytes.len() && bytes[i + 1] == b's' {
392                        if last_rank < UnitRank::Millis {
393                            (UnitRank::Millis, 2)
394                        } else {
395                            return false;
396                        }
397                    } else if last_rank < UnitRank::Minutes {
398                        (UnitRank::Minutes, 1)
399                    } else {
400                        return false;
401                    }
402                },
403                _ => return false,
404            };
405
406            last_rank = current_rank;
407            i += consumed;
408        }
409
410        true
411    }

413    /// Check if a value represents a Cedar entity reference.
414    #[must_use]
415    pub fn is_entity_reference(value: &Value) -> bool {
416        if let Value::Object(obj) = value {
417            obj.len() == 2
418                && obj.get("type").is_some_and(serde_json::Value::is_string)
419                && obj.get("id").is_some_and(serde_json::Value::is_string)
420        } else {
421            false
422        }
423    }

425    /// Parse an entity reference from JSON.
426    pub(super) fn parse_entity_reference(
427        value: &Value,
428    ) -> Result<EntityReference, ValueMappingError> {
429        if let Value::Object(obj) = value {
430            let entity_type = obj.get("type").and_then(|v| v.as_str()).ok_or_else(|| {
431                ValueMappingError::InvalidEntityReference {
432                    reason: "missing or invalid 'type' field".to_string(),
433                }
434            })?;
435
436            let entity_id = obj.get("id").and_then(|v| v.as_str()).ok_or_else(|| {
437                ValueMappingError::InvalidEntityReference {
438                    reason: "missing or invalid 'id' field".to_string(),
439                }
440            })?;
441
442            Ok(EntityReference {
443                entity_type: entity_type.to_string(),
444                entity_id: entity_id.to_string(),
445            })
446        } else {
447            Err(ValueMappingError::InvalidEntityReference {
448                reason: "expected object with 'type' and 'id' fields".to_string(),
449            })
450        }
451    }

466    // Internal conversion method
467    fn convert_value(
468        &self,
469        value: &Value,
470    ) -> Result<Option<RestrictedExpression>, ValueMappingError> {
471        let expr = match value {
472            Value::Null => return Err(ValueMappingError::NullNotSupported),
473            Value::Bool(b) => RestrictedExpression::new_bool(*b),
474            Value::Number(n) => Self::convert_number(n)?,
475            Value::String(s) => self.convert_string(s),
476            Value::Array(arr) => self.convert_array(arr)?,
477            Value::Object(obj) => return self.convert_object(value, obj),
478        };
479
480        Ok(Some(expr))
481    }

483    /// Convert a JSON number to a Cedar expression.
484    fn convert_number(n: &serde_json::Number) -> Result<RestrictedExpression, ValueMappingError> {
485        if let Some(i) = n.as_i64() {
486            Ok(RestrictedExpression::new_long(i))
487        } else if let Some(f) = n.as_f64() {
488            // Convert floating point to decimal extension
489            // Format to 4 decimal places to avoid scientific notation and ensure Cedar compatibility
490            let decimal_str = format!("{f:.4}");
491            Ok(RestrictedExpression::new_decimal(decimal_str))
492        } else {
493            Err(ValueMappingError::NumberNotRepresentable {
494                value: n.to_string(),
495            })
496        }
497    }

499    /// Convert a JSON string to a Cedar expression.
500    fn convert_string(&self, s: &str) -> RestrictedExpression {
501        if self.auto_detect_extensions {
502            match Self::detect_extension(s) {
503                Some(ExtensionValue::IpAddr(ip)) => RestrictedExpression::new_ip(ip),
504                Some(ExtensionValue::Decimal(d)) => RestrictedExpression::new_decimal(d),
505                Some(ExtensionValue::DateTime(dt)) => RestrictedExpression::new_datetime(dt),
506                Some(ExtensionValue::Duration(dur)) => RestrictedExpression::new_duration(dur),
507                None => RestrictedExpression::new_string(s.to_string()),
508            }
509        } else {
510            RestrictedExpression::new_string(s.to_string())
511        }
512    }

514    /// Convert a JSON array to a Cedar set expression.
515    fn convert_array(&self, arr: &[Value]) -> Result<RestrictedExpression, ValueMappingError> {
516        let mut exprs = Vec::with_capacity(arr.len());
517
518        for item in arr {
519            match self.convert_value(item)? {
520                Some(expr) => exprs.push(expr),
521                None => {
522                    return Err(ValueMappingError::NullNotSupported);
523                },
524            }
525        }
526
527        Ok(RestrictedExpression::new_set(exprs))
528    }

530    /// Convert a JSON object to a Cedar expression (entity, extension, or record).
531    fn convert_object(
532        &self,
533        value: &Value,
534        obj: &serde_json::Map<String, Value>,
535    ) -> Result<Option<RestrictedExpression>, ValueMappingError> {
536        // Check for entity reference
537        if Self::is_entity_reference(value) {
538            return Self::convert_entity_reference(value);
539        }
540
541        // Check for extension type markers (__extn)
542        if let Some(extn) = obj.get("__extn") {
543            return Self::convert_extension_marker(extn);
544        }
545
546        // Regular record
547        let mut fields = HashMap::with_capacity(obj.len());
548
549        for (key, val) in obj {
550            let expr = self.convert_value(val)?;
551            fields.insert(
552                key.clone(),
553                expr.expect("convert_value should always return Some"),
554            );
555        }
556
557        Ok(Some(RestrictedExpression::new_record(fields)?))
558    }

560    /// Convert an entity reference to a Cedar entity UID expression.
561    fn convert_entity_reference(
562        value: &Value,
563    ) -> Result<Option<RestrictedExpression>, ValueMappingError> {
564        let entity_ref = Self::parse_entity_reference(value)?;
565
566        let entity_type = EntityTypeName::from_str(&entity_ref.entity_type).map_err(|e| {
567            ValueMappingError::InvalidEntityReference {
568                reason: format!("invalid entity type '{}': {}", entity_ref.entity_type, e),
569            }
570        })?;
571
572        let entity_id = EntityId::from_str(&entity_ref.entity_id).map_err(|e| {
573            ValueMappingError::InvalidEntityReference {
574                reason: format!("invalid entity id '{}': {}", entity_ref.entity_id, e),
575            }
576        })?;
577
578        let uid = EntityUid::from_type_name_and_id(entity_type, entity_id);
579        Ok(Some(RestrictedExpression::new_entity_uid(uid)))
580    }

582    /// Convert an extension marker (__extn) to a Cedar extension expression.
583    fn convert_extension_marker(
584        extn: &Value,
585    ) -> Result<Option<RestrictedExpression>, ValueMappingError> {
586        let extn_obj =
587            extn.as_object()
588                .ok_or_else(|| ValueMappingError::InvalidExtensionFormat {
589                    extension_type: "__extn".to_string(),
590                    value: extn.to_string(),
591                })?;
592
593        let fn_name = extn_obj.get("fn").and_then(|v| v.as_str()).ok_or_else(|| {
594            ValueMappingError::InvalidExtensionFormat {
595                extension_type: "__extn".to_string(),
596                value: format!(
597                    "missing or invalid 'fn' field in {}",
598                    serde_json::to_string(extn_obj).unwrap_or_default()
599                ),
600            }
601        })?;
602
603        let arg = extn_obj
604            .get("arg")
605            .and_then(|v| v.as_str())
606            .ok_or_else(|| ValueMappingError::InvalidExtensionFormat {
607                extension_type: fn_name.to_string(),
608                value: format!(
609                    "missing or invalid 'arg' field in {}",
610                    serde_json::to_string(extn_obj).unwrap_or_default()
611                ),
612            })?;
613
614        match fn_name {
615            "decimal" => Ok(Some(RestrictedExpression::new_decimal(arg))),
616            "ip" | "ipaddr" => Ok(Some(RestrictedExpression::new_ip(arg))),
617            "datetime" => Ok(Some(RestrictedExpression::new_datetime(arg))),
618            "duration" => Ok(Some(RestrictedExpression::new_duration(arg))),
619            _ => Err(ValueMappingError::InvalidExtensionFormat {
620                extension_type: fn_name.to_string(),
621                value: arg.to_string(),
622            }),
623        }
624    }

626    /// Estimates the JSON-serialized size of a value in bytes.
627    ///
628    /// This provides a rough estimate for size limiting without actual serialization.
629    fn estimate_value_size(value: &Value) -> usize {
630        match value {
631            // "null" = 4 chars
632            Value::Null => 4,
633            // "true" or "false" = 4-5 chars, use max
634            Value::Bool(_) => 5,
635            // Number as string representation
636            Value::Number(n) => n.to_string().len(),
637            // String content + 2 for surrounding quotes
638            Value::String(s) => s.len() + 2,
639            // 2 for brackets [] + elements with commas
640            Value::Array(arr) => {
641                2 + arr
642                    .iter()
643                    .map(|v| Self::estimate_value_size(v) + 1) // +1 for comma separator
644                    .sum::<usize>()
645            },
646            // 2 for braces {} + key-value pairs
647            Value::Object(obj) => {
648                2 + obj
649                    .iter()
650                    .map(|(k, v)| k.len() + 3 + Self::estimate_value_size(v) + 1) // +3 for quotes and colon, +1 for comma
651                    .sum::<usize>()
652            },
653        }
654    }

656    // Normalize Cedar JSON format to standard JSON
657    fn normalize_cedar_json(value: &Value) -> Result<Value, ValueMappingError> {
658        match value {
659            Value::Object(obj) => {
660                // Check for __entity marker (Cedar format for entity references)
661                if let Some(entity) = obj.get("__entity")
662                    && let Some(entity_obj) = entity.as_object()
663                {
664                    return Ok(serde_json::json!({
665                        "type": entity_obj.get("type"),
666                        "id": entity_obj.get("id")
667                    }));
668                }
669
670                // Check for __extn marker (extension types)
671                // Preserve the entire wrapper so json_to_cedar can consume it
672                if obj.contains_key("__extn") {
673                    return Ok(Value::Object(obj.clone()));
674                }
675
676                // Regular object - recursively process
677                let mut normalized = Map::new();
678                for (key, val) in obj {
679                    normalized.insert(key.clone(), Self::normalize_cedar_json(val)?);
680                }
681                Ok(Value::Object(normalized))
682            },
683            Value::Array(arr) => {
684                let normalized: Result<Vec<_>, _> =
685                    arr.iter().map(Self::normalize_cedar_json).collect();
686                Ok(Value::Array(normalized?))
687            },
688            // Primitives pass through unchanged
689            _ => Ok(value.clone()),
690        }
691    }
692}
