16/// Statistics about the `DataStore`.
17///
18/// Provides insight into the current state and usage of the data store,
19/// including memory usage metrics and capacity information.
20#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
21pub struct DataStoreStats {
22    /// Number of entries currently stored
23    pub entry_count: usize,
24    /// Maximum number of entries allowed (0 = unlimited)
25    pub max_entries: usize,
26    /// Maximum size per entry in bytes (0 = unlimited)
27    pub max_entry_size: usize,
28    /// Whether metrics tracking is enabled
29    pub metrics_enabled: bool,
30    /// Total size of all entries in bytes (approximate, based on JSON serialization)
31    pub total_size_bytes: usize,
32    /// Average size per entry in bytes (0 if no entries)
33    pub avg_entry_size_bytes: usize,
34    /// Percentage of capacity used (0.0-100.0, based on entry count)
35    pub capacity_usage_percent: f64,
36    /// Memory usage threshold percentage (from config)
37    pub memory_alert_threshold: f64,
38    /// Whether memory usage exceeds the alert threshold
39    pub memory_alert_triggered: bool,
40}

42/// Trait defining the public API for data store operations.
43///
44/// This trait provides a consistent interface for pushing, retrieving,
45/// and managing data in the store. All operations are thread-safe.
46///
47/// The [`Cedarling`](crate::Cedarling) struct implements this trait, providing
48/// access to the data store through the main application instance.
49///
50/// # Example
51///
52/// ```no_run
53/// use cedarling::{Cedarling, DataApi, DataError};
54/// use serde_json::json;
55/// use std::time::Duration;
56///
57/// fn use_data_api(cedarling: &Cedarling) -> Result<(), DataError> {
58///     // Push data with a 5-minute TTL
59///     cedarling.push_data_ctx(
60///         "user_roles",
61///         json!(["admin", "editor"]),
62///         Some(Duration::from_secs(300)),
63///     )?;
64///
65///     // Retrieve data
66///     if let Some(roles) = cedarling.get_data_ctx("user_roles")? {
67///         println!("User roles: {}", roles);
68///     }
69///
70///     // List all entries with metadata
71///     for entry in cedarling.list_data_ctx()? {
72///         println!("Key: {}, Type: {:?}", entry.key, entry.data_type);
73///     }
74///
75///     // Get store statistics
76///     let stats = cedarling.get_stats_ctx()?;
77///     println!("Entries: {}/{}", stats.entry_count, stats.max_entries);
78///
79///     // Remove data
80///     cedarling.remove_data_ctx("user_roles")?;
81///
82///     // Clear all data
83///     cedarling.clear_data_ctx()?;
84///
85///     Ok(())
86/// }
87/// ```
88pub trait DataApi {
89    /// Push a value into the store with an optional TTL.
90    ///
91    /// If the key already exists, the value will be replaced.
92    /// If TTL is not provided, the default TTL from configuration is used.
93    fn push_data_ctx(
94        &self,
95        key: &str,
96        value: Value,
97        ttl: Option<Duration>,
98    ) -> Result<(), DataError>;
99
100    /// Get a value from the store by key.
101    ///
102    /// Returns `Ok(None)` if the key doesn't exist or the entry has expired.
103    /// If metrics are enabled, increments the access count for the entry.
104    fn get_data_ctx(&self, key: &str) -> Result<Option<Value>, DataError>;
105
106    /// Get a data entry with full metadata by key.
107    ///
108    /// Returns `Ok(None)` if the key doesn't exist or the entry has expired.
109    /// Includes metadata like creation time, expiration, access count, and type.
110    fn get_data_entry_ctx(&self, key: &str) -> Result<Option<DataEntry>, DataError>;
111
112    /// Remove a value from the store by key.
113    ///
114    /// Returns `Ok(true)` if the key existed and was removed, `Ok(false)` otherwise.
115    fn remove_data_ctx(&self, key: &str) -> Result<bool, DataError>;
116
117    /// Clear all entries from the store.
118    fn clear_data_ctx(&self) -> Result<(), DataError>;
119
120    /// List all entries with their metadata.
121    ///
122    /// Returns a vector of `DataEntry` containing key, value, type, and timing metadata.
123    fn list_data_ctx(&self) -> Result<Vec<DataEntry>, DataError>;
124
125    /// Get statistics about the data store.
126    ///
127    /// Returns current entry count, capacity limits, and configuration state.
128    fn get_stats_ctx(&self) -> Result<DataStoreStats, DataError>;
129}
