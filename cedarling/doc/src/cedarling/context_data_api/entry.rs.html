12/// Helper module for serializing `DateTime`
13mod datetime {
14    use chrono::{DateTime, Utc};
15    use serde::{Deserialize, Deserializer, Serializer};
16
17    pub(super) fn serialize<S>(date: &DateTime<Utc>, serializer: S) -> Result<S::Ok, S::Error>
18    where
19        S: Serializer,
20    {
21        serializer.serialize_str(&date.to_rfc3339())
22    }
23
24    pub(super) fn deserialize<'de, D>(deserializer: D) -> Result<DateTime<Utc>, D::Error>
25    where
26        D: Deserializer<'de>,
27    {
28        let s: String = String::deserialize(deserializer)?;
29        DateTime::parse_from_rfc3339(&s)
30            .map(|dt| dt.with_timezone(&Utc))
31            .map_err(serde::de::Error::custom)
32    }
33}

35/// Helper module for serializing Optional `DateTime`
36mod datetime_option {
37    use chrono::{DateTime, Utc};
38    use serde::{Deserialize, Deserializer, Serializer};
39
40    /// Serialize an optional `DateTime` to RFC 3339 format.
41    ///
42    /// Note: This function signature uses `&Option<T>` rather than `Option<&T>`
43    /// because serde's `#[serde(serialize_with)]` attribute passes references
44    /// to field values, and the field type is `Option<DateTime<Utc>>`.
45    #[allow(clippy::ref_option)]
46    pub(super) fn serialize<S>(
47        date: &Option<DateTime<Utc>>,
48        serializer: S,
49    ) -> Result<S::Ok, S::Error>
50    where
51        S: Serializer,
52    {
53        match date {
54            Some(dt) => serializer.serialize_some(&dt.to_rfc3339()),
55            None => serializer.serialize_none(),
56        }
57    }
58
59    pub(super) fn deserialize<'de, D>(deserializer: D) -> Result<Option<DateTime<Utc>>, D::Error>
60    where
61        D: Deserializer<'de>,
62    {
63        let opt: Option<String> = Option::deserialize(deserializer)?;
64        match opt {
65            Some(s) => DateTime::parse_from_rfc3339(&s)
66                .map(|dt| Some(dt.with_timezone(&Utc)))
67                .map_err(serde::de::Error::custom),
68            None => Ok(None),
69        }
70    }
71}

73/// Represents the type of a Cedar value based on JSON structure.
74#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
75#[serde(rename_all = "lowercase")]
76pub enum CedarType {
77    /// String type
78    String,
79    /// Long (integer) type
80    Long,
81    /// Boolean type
82    Bool,
83    /// Set (array) type
84    Set,
85    /// Record (object) type
86    Record,
87    /// Entity reference type
88    Entity,
89    /// IP address extension type (ipaddr)
90    Ip,
91    /// Decimal extension type
92    Decimal,
93    /// `DateTime` extension type
94    DateTime,
95    /// Duration extension type
96    Duration,
97}
98
99// Track which units are seen to prevent repeats and enforce order
100#[derive(PartialEq, PartialOrd)]
101pub(crate) enum UnitRank {
102    Start,
103    Days,
104    Hours,
105    Minutes,
106    Seconds,
107    Millis,
108}
109
110impl CedarType {
111    /// Infer the Cedar type from a JSON value.
112    ///
113    /// This method detects extension types by recognizing:
114    /// - `__extn` markers with `fn` = "ip", "decimal", "datetime", or "duration"
115    /// - Strings that look like IP addresses, decimals, datetimes, or durations
116    ///
117    /// See: <https://docs.cedarpolicy.com/policies/syntax-datatypes.html#datatype-extension>
118    #[must_use]
119    pub fn from_value(value: &Value) -> Self {
120        match value {
121            Value::String(s) => Self::from_string_value(s),
122            Value::Number(n) => {
123                if n.is_i64() || n.is_u64() {
124                    Self::Long
125                } else {
126                    // Floating point numbers become decimals in Cedar
127                    Self::Decimal
128                }
129            },
130            Value::Bool(_) => Self::Bool,
131            Value::Array(_) => Self::Set,
132            Value::Object(obj) => {
133                // Check for extension type marker (__extn)
134                if let Some(extn) = obj.get("__extn")
135                    && let Some(extn_obj) = extn.as_object()
136                    && let Some(fn_name) = extn_obj.get("fn").and_then(|v| v.as_str())
137                {
138                    return match fn_name {
139                        "ip" | "ipaddr" => Self::Ip,
140                        "decimal" => Self::Decimal,
141                        "datetime" => Self::DateTime,
142                        "duration" => Self::Duration,
143                        _ => Self::Record,
144                    };
145                }
146
147                // Check if it's an entity reference with explicit marker
148                // Entity references must have exactly "type" and "id" fields,
149                // and the "type" value must be a string (to avoid misclassifying
150                // normal records that happen to have these fields)
151                if obj.len() == 2
152                    && obj.contains_key("type")
153                    && obj.contains_key("id")
154                    && obj.get("type").is_some_and(serde_json::Value::is_string)
155                    && obj.get("id").is_some_and(serde_json::Value::is_string)
156                {
157                    Self::Entity
158                } else {
159                    Self::Record
160                }
161            },
162            Value::Null => {
163                // Null is not a valid Cedar type, but we'll default to String
164                Self::String
165            },
166        }
167    }
168
169    /// Infer Cedar type from a string value by detecting extension patterns.
170    fn from_string_value(s: &str) -> Self {
171        use crate::context_data_api::mapper::CedarValueMapper;
172        use crate::context_data_api::mapper::ExtensionValue;
173
174        match CedarValueMapper::detect_extension(s) {
175            Some(ExtensionValue::IpAddr(_)) => Self::Ip,
176            Some(ExtensionValue::DateTime(_)) => Self::DateTime,
177            Some(ExtensionValue::Duration(_)) => Self::Duration,
178            Some(ExtensionValue::Decimal(_)) => Self::Decimal,
179            None => Self::String,
180        }
181    }
182
183    /// Checks whether the value is either:
184    /// - a valid RFC 3339 datetime, or
185    /// - a valid ISO 8601 date-only string (YYYY-MM-DD)
186    pub(crate) fn is_datetime_format(value: &str) -> bool {
187        DateTime::parse_from_rfc3339(value).is_ok()
188            || NaiveDate::parse_from_str(value, "%Y-%m-%d").is_ok()
189    }
190
191    /// Cedar-compliant duration format validator.
192    /// Enforces:
193    /// - Optional leading '-'
194    /// - Units only d > h > m > s > ms (in that order)
195    /// - No repeated units
196    /// - Each segment: digits then unit
197    pub(crate) fn is_duration_format(value: &str) -> bool {
198        if value.is_empty() {
199            return false;
200        }
201
202        let bytes = value.as_bytes();
203        let mut i = 0;
204
205        if bytes[i] == b'-' {
206            i += 1;
207            if i == bytes.len() {
208                return false;
209            }
210        }
211
212        let mut last_rank = UnitRank::Start;
213
214        while i < bytes.len() {
215            let start = i;
216            while i < bytes.len() && bytes[i].is_ascii_digit() {
217                i += 1;
218            }
219            if start == i {
220                return false;
221            }
222
223            let (current_rank, consumed) = match bytes.get(i) {
224                Some(b'd') if last_rank < UnitRank::Days => (UnitRank::Days, 1),
225                Some(b'h') if last_rank < UnitRank::Hours => (UnitRank::Hours, 1),
226                Some(b's') if last_rank < UnitRank::Seconds => (UnitRank::Seconds, 1),
227                Some(b'm') => {
228                    if i + 1 < bytes.len() && bytes[i + 1] == b's' {
229                        if last_rank < UnitRank::Millis {
230                            (UnitRank::Millis, 2)
231                        } else {
232                            return false;
233                        }
234                    } else if last_rank < UnitRank::Minutes {
235                        (UnitRank::Minutes, 1)
236                    } else {
237                        return false;
238                    }
239                },
240                _ => return false,
241            };
242
243            last_rank = current_rank;
244            i += consumed;
245        }
246
247        true
248    }
249}

251/// A data entry in the `DataStore` with value and metadata.
252///
253/// This structure wraps the actual value with metadata including creation time,
254/// expiration time, access count, and type information.
255#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
256pub struct DataEntry {
257    /// The key for this entry
258    pub key: String,
259    /// The actual value stored
260    pub value: Value,
261    /// The inferred Cedar type of the value
262    pub data_type: CedarType,
263    /// Timestamp when this entry was created
264    #[serde(with = "datetime")]
265    pub created_at: DateTime<Utc>,
266    /// Timestamp when this entry expires (if TTL is set)
267    #[serde(with = "datetime_option")]
268    pub expires_at: Option<DateTime<Utc>>,
269    /// Number of times this entry has been accessed
270    pub access_count: u64,
271}
272
273impl DataEntry {
274    /// Create a new `DataEntry` with the given key and value.
275    ///
276    /// The `created_at` timestamp is set to the current time,
277    /// and `expires_at` is calculated from the optional TTL.
278    /// Uses saturating conversion for TTL to prevent overflow.
279    #[must_use]
280    pub fn new(key: String, value: Value, ttl: Option<StdDuration>) -> Self {
281        let created_at = Utc::now();
282        let expires_at = ttl.map(|duration| {
283            // Use saturating conversion to prevent overflow
284            let chrono_duration = std_duration_to_chrono_duration(duration);
285            created_at + chrono_duration
286        });
287
288        Self {
289            key,
290            data_type: CedarType::from_value(&value),
291            value,
292            created_at,
293            expires_at,
294            access_count: 0,
295        }
296    }
297
298    /// Increment the access count for this entry.
299    pub fn increment_access(&mut self) {
300        self.access_count = self.access_count.saturating_add(1);
301    }
302
303    /// Check if this entry has expired.
304    #[must_use]
305    pub fn is_expired(&self, now: DateTime<Utc>) -> bool {
306        if let Some(expires_at) = self.expires_at {
307            now > expires_at
308        } else {
309            false
310        }
311    }
312}

319    #[test]
320    fn test_cedar_type_from_value() {
321        assert_eq!(CedarType::from_value(&json!("test")), CedarType::String);
322        assert_eq!(CedarType::from_value(&json!(42)), CedarType::Long);
323        assert_eq!(CedarType::from_value(&json!(true)), CedarType::Bool);
324        assert_eq!(CedarType::from_value(&json!([1, 2, 3])), CedarType::Set);
325        assert_eq!(CedarType::from_value(&json!({"a": 1})), CedarType::Record);
326        assert_eq!(
327            CedarType::from_value(&json!({"type": "User", "id": "123"})),
328            CedarType::Entity
329        );
330    }

334        use test_utils::assert_eq;

336        assert_eq!(entry.key, "key1");
337        assert_eq!(&entry.value, &json!("value1"));
338        assert_eq!(entry.data_type, CedarType::String);
339        assert_eq!(entry.access_count, 0);
340        assert!(entry.expires_at.is_none());
341    }

381    #[test]
382    fn test_serialization() {
383        let entry = DataEntry::new(
384            "key1".to_string(),
385            json!("value1"),
386            Some(StdDuration::from_secs(3600)),
387        );
388        let serialized = serde_json::to_string(&entry).expect("should serialize");
389        let deserialized: DataEntry =

391        assert_eq!(entry.key, deserialized.key);
392        assert_eq!(entry.value, deserialized.value);
393        assert_eq!(entry.data_type, deserialized.data_type);
394        assert_eq!(entry.access_count, deserialized.access_count);
395
396        // Verify datetime fields survive round-trip using RFC3339 representation
397        assert_eq!(
398            entry.created_at.to_rfc3339(),
399            deserialized.created_at.to_rfc3339(),
400            "created_at should survive serde_json round-trip"
401        );
402        assert_eq!(
403            entry.expires_at.map(|dt| dt.to_rfc3339()),
404            deserialized.expires_at.map(|dt| dt.to_rfc3339()),
405            "expires_at should survive serde_json round-trip"
406        );
