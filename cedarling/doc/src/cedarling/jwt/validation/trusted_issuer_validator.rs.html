6//! Trusted issuer JWT validation module.
7//!
8//! This module provides standalone functionality to validate JWT tokens against configured
9//! trusted issuers, including issuer matching, required claims validation, JWKS fetching,
10//! and signature verification.
11//!
12//! ## Features
13//!
14//! - **Issuer matching**: Validates that JWT tokens come from configured trusted issuers
15//! - **Required claims validation**: Ensures tokens contain all claims specified in issuer configuration
16//! - **JWKS management**: Fetches and caches JWKS keys from issuer's OIDC endpoint with configurable TTL
17//! - **Signature verification**: Validates JWT signatures using cached JWKS keys

36/// Errors that can occur during trusted issuer validation.
37#[derive(Debug, Error)]
38pub enum TrustedIssuerError {
39    /// Token issuer is not in the list of trusted issuers
40    #[error("Untrusted issuer: '{0}'")]
41    UntrustedIssuer(String),
42
43    /// Token is missing a required claim
44    #[error("Missing required claim: '{claim}' for token type '{token_type}'")]
45    MissingRequiredClaim {
46        /// The name of the missing claim
47        claim: String,
48        /// The type of token being validated
49        token_type: String,
50    },
51
52    /// Failed to decode JWT header
53    #[error("Failed to decode JWT header: {0}")]
54    DecodeHeader(#[from] jsonwebtoken::errors::Error),
55
56    /// Failed to fetch OpenID configuration
57    #[error("Failed to fetch OpenID configuration from '{endpoint}': {source}")]
58    OpenIdConfigFetch {
59        /// The OIDC endpoint that failed
60        endpoint: String,
61        /// The underlying error
62        #[source]
63        source: Box<dyn std::error::Error + Send + Sync>,
64    },
65
66    /// Failed to fetch or process JWKS
67    #[error("Failed to fetch JWKS: {0}")]
68    JwksFetch(#[from] KeyServiceError),
69
70    /// No matching key found in JWKS
71    #[error(
72        "No matching key found for kid: {}, algorithm: '{alg:?}'",
73        kid.as_ref().map(|s| s.as_str()).unwrap_or("none")
74    )]
75    NoMatchingKey {
76        /// The key ID from the JWT header
77        kid: Option<String>,
78        /// The algorithm from the JWT header
79        alg: Algorithm,
80    },
81
82    /// JWT signature validation failed
83    #[error("Invalid JWT signature: {0}")]
84    InvalidSignature(String),
85
86    /// Token type not configured for issuer
87    #[error("Token type '{token_type}' not configured for issuer '{issuer}'")]
88    TokenTypeNotConfigured {
89        /// The token type that wasn't configured
90        token_type: String,
91        /// The issuer missing the token type configuration
92        issuer: String,
93    },
94
95    /// Missing issuer claim in token
96    #[error("Token missing 'iss' claim")]
97    MissingIssuerClaim,
98
99    /// Token metadata has empty entity_type_name
100    #[error(
101        "Invalid token metadata configuration: entity_type_name is empty for token type '{token_type}'"
102    )]
103    EmptyEntityTypeName {
104        /// The token type with empty entity_type_name
105        token_type: String,
106    },
107}

109/// Result type for trusted issuer validation operations.
110pub type Result<T> = std::result::Result<T, TrustedIssuerError>;

112/// Default JWKS cache duration (1 hour) used when no Cache-Control header is present
113const DEFAULT_JWKS_CACHE_DURATION_SECS: u64 = 3600;
114
115/// Minimum JWKS cache duration (5 minutes) to prevent excessive requests
116const MIN_JWKS_CACHE_DURATION_SECS: u64 = 300;
117
118/// Maximum JWKS cache duration (24 hours) to ensure keys are refreshed regularly
119const MAX_JWKS_CACHE_DURATION_SECS: u64 = 86400;

121/// Validator for JWT tokens against trusted issuer configurations.
122///
123/// This validator provides the following functionality:
124/// - Issuer matching against configured trusted issuers
125/// - Required claims validation based on token metadata
126/// - JWKS fetching and caching with configurable TTL
127/// - JWT signature verification
128pub struct TrustedIssuerValidator {
129    /// Map of issuer identifiers to their configurations
130    trusted_issuers: HashMap<String, Arc<TrustedIssuer>>,
131    /// Reverse lookup map: OIDC base URL -> issuer
132    /// This optimizes issuer lookup when dealing with hundreds of trusted issuers
133    url_to_issuer: HashMap<String, Arc<TrustedIssuer>>,
134    /// Key service for managing JWKS keys
135    key_service: KeyService,
136    /// Cache of fetched OpenID configurations (issuer URL -> config)
137    oidc_configs: HashMap<String, Arc<OpenIdConfig>>,
138    /// Timestamp of last JWKS fetch for expiration tracking
139    /// Maps issuer OIDC endpoint to (fetch_time, cache_duration)
140    keys_fetch_time: HashMap<String, (DateTime<Utc>, Duration)>,
141    /// Optional logger for diagnostic messages
142    logger: Option<Logger>,
143}
144
145impl TrustedIssuerValidator {
146    /// Creates a new trusted issuer validator with the given trusted issuers.
147    ///
148    /// This is a convenience constructor equivalent to `with_logger(trusted_issuers, None)`.
149    ///
150    /// # Arguments
151    ///
152    /// * `trusted_issuers` - Map of issuer IDs to their configurations
153    pub fn new(trusted_issuers: HashMap<String, TrustedIssuer>) -> Self {
154        Self::with_logger(trusted_issuers, None)
155    }
156
157    /// Creates a new trusted issuer validator with a logger.
158    pub fn with_logger(
159        trusted_issuers: HashMap<String, TrustedIssuer>,
160        logger: Option<Logger>,
161    ) -> Self {
162        let trusted_issuers: HashMap<String, Arc<TrustedIssuer>> = trusted_issuers
163            .into_iter()
164            .map(|(k, v)| (k, Arc::new(v)))
165            .collect();
166
167        // Build reverse lookup map: OIDC base URL -> issuer
168        let mut url_to_issuer = HashMap::with_capacity(trusted_issuers.len());
169        for (id, issuer) in &trusted_issuers {
170            // Extract base URL from OIDC endpoint
171            if let Some(base_url) = issuer
172                .oidc_endpoint
173                .as_str()
174                .strip_suffix("/.well-known/openid-configuration")
175            {
176                let normalized_url = base_url.trim_end_matches('/');
177                url_to_issuer.insert(normalized_url.to_string(), issuer.clone());
178            }
179
180            // Also add the issuer ID if it's a URL format
181            if id.starts_with("http://") || id.starts_with("https://") {
182                let normalized_id = id.trim_end_matches('/');
183                url_to_issuer.insert(normalized_id.to_string(), issuer.clone());
184            }
185        }
186
187        Self {
188            trusted_issuers,
189            url_to_issuer,
190            key_service: KeyService::new(),
191            oidc_configs: HashMap::new(),
192            keys_fetch_time: HashMap::new(),
193            logger,
194        }
195    }
196
197    /// Finds a trusted issuer by the issuer claim value.
198    ///
199    /// This method matches the token's `iss` claim against the configured trusted issuers.
200    /// The matching is done by comparing the issuer URL or issuer ID.
201    pub fn find_trusted_issuer(&self, issuer_claim: &str) -> Result<Arc<TrustedIssuer>> {
202        // Try exact match first by issuer ID
203        if let Some(issuer) = self.trusted_issuers.get(issuer_claim) {
204            return Ok(issuer.clone());
205        }
206
207        // Try matching by URL using reverse lookup map (O(1) instead of O(n))
208        // Parse the issuer claim as a URL and normalize it for lookup
209        if let Ok(iss_url) = Url::parse(issuer_claim) {
210            let normalized_url = iss_url.as_str().trim_end_matches('/');
211
212            if let Some(issuer) = self.url_to_issuer.get(normalized_url) {
213                return Ok(issuer.clone());
214            }
215        }
216
217        Err(TrustedIssuerError::UntrustedIssuer(
218            issuer_claim.to_string(),
219        ))
220    }
221
222    /// Fetches and caches the OpenID configuration for a trusted issuer.
223    ///
224    /// If the configuration has already been fetched, returns the cached version.
225    async fn get_or_fetch_oidc_config(
226        &mut self,
227        trusted_issuer: &TrustedIssuer,
228    ) -> Result<Arc<OpenIdConfig>> {
229        let endpoint_str = trusted_issuer.oidc_endpoint.as_str();
230
231        // Check cache first
232        if let Some(config) = self.oidc_configs.get(endpoint_str) {
233            return Ok(config.clone());
234        }
235
236        // Fetch from endpoint
237        let config = OpenIdConfig::get_from_url(&trusted_issuer.oidc_endpoint)
238            .await
239            .map_err(|e| TrustedIssuerError::OpenIdConfigFetch {
240                endpoint: endpoint_str.to_string(),
241                source: Box::new(e),
242            })?;
243
244        let config_arc = Arc::new(config);
245        self.oidc_configs
246            .insert(endpoint_str.to_string(), Arc::clone(&config_arc));
247
248        Ok(config_arc)
249    }
250
251    /// Ensures JWKS keys are loaded for the given issuer.
252    ///
253    /// Fetches the OpenID configuration and loads keys from the JWKS endpoint.
254    /// Implements automatic key refresh based on cache duration.
255    async fn ensure_keys_loaded(&mut self, trusted_issuer: &TrustedIssuer) -> Result<()> {
256        let oidc_config = self.get_or_fetch_oidc_config(trusted_issuer).await?;
257        let endpoint_str = trusted_issuer.oidc_endpoint.as_str();
258
259        // Check if keys for this endpoint have been loaded and if they've expired
260        // Use endpoint-specific check instead of global has_keys() to avoid skipping issuers
261        let should_refresh = if let Some((fetch_time, cache_duration)) =
262            self.keys_fetch_time.get(endpoint_str)
263        {
264            // Keys have been loaded for this endpoint - check if they've expired
265            let elapsed = Utc::now().signed_duration_since(*fetch_time);
266            // Refresh if elapsed time exceeds cache duration
267            // Note: chrono::Duration can represent negative values if time went backwards
268            elapsed
269                >= chrono::Duration::from_std(*cache_duration).unwrap_or(chrono::Duration::zero())
270        } else {
271            // No timestamp recorded for this endpoint - keys haven't been loaded yet
272            true
273        };
274
275        if !should_refresh {
276            return Ok(());
277        }
278
279        // Fetch keys using the key service
280        self.key_service
281            .get_keys_using_oidc(&oidc_config, &self.logger)
282            .await?;
283
284        // Determine cache duration
285        let cache_duration = self.determine_cache_duration(trusted_issuer);
286
287        // Record fetch time for expiration tracking
288        self.keys_fetch_time
289            .insert(endpoint_str.to_string(), (Utc::now(), cache_duration));
290
291        // Log key refresh for monitoring
292        self.logger.log_any(JwtLogEntry::new(
293            format!(
294                "JWKS keys loaded for issuer '{}', cache duration: {}s",
295                endpoint_str,
296                cache_duration.as_secs()
297            ),
298            Some(crate::LogLevel::INFO),
299        ));
300
301        Ok(())
302    }
303
304    /// Determines the appropriate cache duration for JWKS keys.
305    fn determine_cache_duration(&self, _trusted_issuer: &TrustedIssuer) -> Duration {
306        let cache_secs = DEFAULT_JWKS_CACHE_DURATION_SECS;
307
308        let bounded_secs =
309            cache_secs.clamp(MIN_JWKS_CACHE_DURATION_SECS, MAX_JWKS_CACHE_DURATION_SECS);
310
311        Duration::from_secs(bounded_secs)
312    }

314    /// Validates that a token contains all required claims based on token metadata.
315    ///
316    /// This is a convenience method that delegates to the standalone `validate_required_claims` function.
317    /// It validates claims explicitly specified in `required_claims` set from the token metadata.
318    pub fn validate_required_claims(
319        &self,
320        claims: &JsonValue,
321        token_type: &str,
322        token_metadata: &TokenEntityMetadata,
323    ) -> Result<()> {
324        validate_required_claims(claims, token_type, token_metadata)
325    }

328/// Validates that a token contains all required claims based on token metadata.
329///
330/// This is a standalone function that can be used independently of `TrustedIssuerValidator`.
331/// It validates:
332/// - `entity_type_name` is not empty (configuration validation)
333/// - All claims in `required_claims` set exist
334///
335/// Note: Mapping fields like `user_id`, `role_mapping`, `workload_id`, and `token_id`
336/// are configuration hints for claim extraction, not strictly required claims.
337/// They are validated only if explicitly included in `required_claims`.
338///
339/// # Arguments
340///
341/// * `claims` - The JWT claims as a JSON value
342/// * `token_type` - The type of token (e.g., "access_token", "id_token")
343/// * `token_metadata` - The token metadata configuration from the trusted issuer
344///
345pub fn validate_required_claims(
346    claims: &JsonValue,
347    token_type: &str,
348    token_metadata: &TokenEntityMetadata,
349) -> Result<()> {
350    // Check for entity_type_name (configuration validation, always required)
351    if token_metadata.entity_type_name.is_empty() {
352        return Err(TrustedIssuerError::EmptyEntityTypeName {
353            token_type: token_type.to_string(),
354        });
355    }
356
357    // Validate all claims explicitly specified in required_claims set
358    // This is the authoritative list of required claims from the issuer configuration
359    for claim in &token_metadata.required_claims {
360        if claims.get(claim).is_none() {
361            return Err(TrustedIssuerError::MissingRequiredClaim {
362                claim: claim.clone(),
363                token_type: token_type.to_string(),
364            });
365        }
366    }
367
368    Ok(())
369}

371impl TrustedIssuerValidator {
372    /// Validates a JWT token against a trusted issuer with JWKS preloading.
373    ///
374    /// This performs comprehensive validation including:
375    /// 1. Extracts the issuer claim from the token
376    /// 2. Matches the issuer against configured trusted issuers
377    /// 3. Preloads JWKS if not already cached
378    /// 4. Validates the JWT signature using JWKS
379    /// 5. Validates required claims based on token metadata
380    /// 6. Validates exp/nbf claims if present
381    ///
382    /// Returns the validated claims and the matched trusted issuer.
383    pub async fn preload_and_validate_token(
384        &mut self,
385        token: &str,
386        token_type: &str,
387    ) -> Result<(JsonValue, Arc<TrustedIssuer>)> {
388        // Decode the JWT header to get the key ID and algorithm
389        let header = decode_header(token)?;
390
391        // First, we need to decode without verification to get the issuer claim
392        // and check for exp/nbf to configure validation later
393        let mut validation = Validation::new(header.alg);
394        validation.insecure_disable_signature_validation();
395        validation.validate_exp = false;
396        validation.validate_nbf = false;
397        validation.required_spec_claims.clear();
398
399        let unverified_token = decode::<JsonValue>(
400            token,
401            &DecodingKey::from_secret(&[]), // Dummy key since we disabled validation
402            &validation,
403        )?;
404
405        let has_exp = unverified_token.claims.get("exp").is_some();
406        let has_nbf = unverified_token.claims.get("nbf").is_some();
407
408        // Extract issuer claim
409        let issuer_claim = unverified_token
410            .claims
411            .get("iss")
412            .and_then(|v| v.as_str())
413            .ok_or(TrustedIssuerError::MissingIssuerClaim)?;
414
415        // Find the trusted issuer
416        let trusted_issuer = self.find_trusted_issuer(issuer_claim)?;
417
418        // Get token metadata for this token type
419        let token_metadata = trusted_issuer
420            .token_metadata
421            .get(token_type)
422            .ok_or_else(|| TrustedIssuerError::TokenTypeNotConfigured {
423                token_type: token_type.to_string(),
424                issuer: issuer_claim.to_string(),
425            })?;
426
427        // Check if token is trusted
428        if !token_metadata.trusted {
429            return Err(TrustedIssuerError::UntrustedIssuer(
430                issuer_claim.to_string(),
431            ));
432        }
433
434        // Ensure JWKS keys are loaded for this issuer
435        self.ensure_keys_loaded(&trusted_issuer).await?;
436
437        // Now validate the signature
438        let key_info = DecodingKeyInfo {
439            issuer: Some(issuer_claim.to_string()),
440            kid: header.kid.clone(),
441            algorithm: header.alg,
442        };
443
444        let decoding_key = self.key_service.get_key(&key_info).ok_or_else(|| {
445            TrustedIssuerError::NoMatchingKey {
446                kid: header.kid,
447                alg: header.alg,
448            }
449        })?;
450
451        // Create validation with signature checking enabled
452        let mut validation = Validation::new(header.alg);
453        validation.set_issuer(&[issuer_claim]);
454
455        validation.validate_exp = has_exp;
456        validation.validate_nbf = has_nbf;
457
458        validation.required_spec_claims.clear();
459        validation.validate_aud = false;
460
461        // Decode and validate signature
462        let verified_token = decode::<JsonValue>(token, decoding_key, &validation)
463            .map_err(|e| TrustedIssuerError::InvalidSignature(e.to_string()))?;
464
465        // Validate required claims (after signature verification)
466        self.validate_required_claims(&verified_token.claims, token_type, token_metadata)?;
467
468        Ok((verified_token.claims, trusted_issuer))
469    }
