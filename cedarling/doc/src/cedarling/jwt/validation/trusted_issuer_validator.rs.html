6//! Trusted issuer JWT validation module.
7//!
8//! This module provides standalone functionality to validate JWT tokens against configured
9//! trusted issuers, including issuer matching, required claims validation, JWKS fetching,
10//! and signature verification.
11//!
12//! ## Features
13//!
14//! - **Issuer matching**: Validates that JWT tokens come from configured trusted issuers
15//! - **Required claims validation**: Ensures tokens contain all claims specified in issuer configuration
16//! - **Signature verification**: Validates JWT signatures using cached JWKS keys
17

29/// Errors that can occur during trusted issuer validation.
30#[derive(Debug, Error)]
31pub enum TrustedIssuerError {
32    /// Token issuer is not in the list of trusted issuers
33    #[error("Untrusted issuer: '{0}'")]
34    UntrustedIssuer(String),
35
36    /// Token is missing a required claim
37    #[error("Missing required claim: '{claim}' for token type '{token_type}'")]
38    MissingRequiredClaim {
39        /// The name of the missing claim
40        claim: String,
41        /// The type of token being validated
42        token_type: String,
43    },
44
45    /// Token metadata has empty `entity_type_name`
46    #[error(
47        "Invalid token metadata configuration: entity_type_name is empty for token type '{token_type}'"
48    )]
49    EmptyEntityTypeName {
50        /// The token type with empty `entity_type_name`
51        token_type: String,
52    },
53}
54
55/// Result type for trusted issuer validation operations.
56type Result<T> = std::result::Result<T, TrustedIssuerError>;
57
58/// Validator for JWT tokens against trusted issuer configurations.
59///
60/// This validator provides the following functionality:
61/// - Issuer matching against configured trusted issuers
62/// - Required claims validation based on token metadata
63/// - JWKS fetching and caching with configurable TTL
64/// - JWT signature verification
65pub(crate) struct TrustedIssuerValidator {
66    /// Reverse lookup map: OIDC base URL -> issuer
67    /// This optimizes issuer lookup when dealing with hundreds of trusted issuers
68    url_to_issuer: HashMap<IssClaim, Arc<TrustedIssuer>>,
69}
70
71impl TrustedIssuerValidator {
72    /// Creates a new trusted issuer validator with a logger.
73    pub(crate) fn new(trusted_issuers: HashMap<String, TrustedIssuer>) -> Self {
74        // Build reverse lookup map: OIDC base URL -> issuer
75        let mut url_to_issuer = HashMap::with_capacity(trusted_issuers.len());
76        for issuer in trusted_issuers.into_values().map(Arc::new) {
77            url_to_issuer.insert(issuer.iss_claim(), issuer);
78        }
79
80        Self { url_to_issuer }
81    }
82
83    /// Finds a trusted issuer by the issuer claim value.
84    ///
85    /// This method matches the token's `iss` claim against the configured trusted issuers.
86    /// The matching is done by comparing the issuer URL or issuer ID.
87    pub(crate) fn find_trusted_issuer(
88        &self,
89        issuer_claim: &IssClaim,
90    ) -> Result<Arc<TrustedIssuer>> {
91        if let Some(issuer) = self.url_to_issuer.get(issuer_claim) {
92            return Ok(issuer.clone());
93        }
94
95        Err(TrustedIssuerError::UntrustedIssuer(
96            issuer_claim.to_string(),
97        ))
98    }
99}

101/// Validates that a token contains all required claims based on token metadata.
102///
103/// This is a standalone function that can be used independently of `TrustedIssuerValidator`.
104/// It validates:
105/// - `entity_type_name` is not empty (configuration validation)
106/// - All claims in `required_claims` set exist
107///
108/// Note: Mapping fields like `user_id`, `role_mapping`, `workload_id`, and `token_id`
109/// are configuration hints for claim extraction, not strictly required claims.
110/// They are validated only if explicitly included in `required_claims`.
111///
112/// # Arguments
113///
114/// * `claims` - The JWT claims as a JSON value
115/// * `token_type` - The type of token (e.g., "`access_token`", "`id_token`")
116/// * `token_metadata` - The token metadata configuration from the trusted issuer
117///
118pub(crate) fn validate_required_claims(
119    claims: &JsonValue,
120    token_type: &str,
121    token_metadata: &TokenEntityMetadata,
122) -> Result<()> {
123    // Check for entity_type_name (configuration validation, always required)
124    if token_metadata.entity_type_name.is_empty() {
125        return Err(TrustedIssuerError::EmptyEntityTypeName {
126            token_type: token_type.to_string(),
127        });
128    }
129
130    // Validate all claims explicitly specified in required_claims set
131    // This is the authoritative list of required claims from the issuer configuration
132    for claim in &token_metadata.required_claims {
133        if claims.get(claim).is_none() {
134            return Err(TrustedIssuerError::MissingRequiredClaim {
135                claim: claim.clone(),
136                token_type: token_type.to_string(),
137            });
138        }
139    }
140
141    Ok(())
142}

152    fn create_test_issuer(id: &str, endpoint: &str) -> TrustedIssuer {
153        let mut token_metadata = HashMap::new();
154        token_metadata.insert(
155            "access_token".to_string(),
156            TokenEntityMetadata::access_token(),
157        );
158        token_metadata.insert("id_token".to_string(), TokenEntityMetadata::id_token());
159
160        TrustedIssuer::new(
161            format!("Test Issuer {id}"),
162            "Test issuer for validation".to_string(),
163            Url::parse(endpoint).unwrap(),
164            token_metadata,
165        )
166    }

168    #[test]
169    fn test_find_trusted_issuer_by_url() {
170        let issuers = HashMap::from([(
171            "issuer1".to_string(),
172            create_test_issuer("1", "https://issuer1.com/.well-known/openid-configuration"),
173        )]);
174
175        let validator = TrustedIssuerValidator::new(issuers);
176
177        let result = validator.find_trusted_issuer(&IssClaim::new("https://issuer1.com"));
178        assert!(result.is_ok());
179        assert_eq!(result.unwrap().name, "Test Issuer 1");
180    }
