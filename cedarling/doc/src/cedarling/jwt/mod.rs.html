6//! # `JwtEngine`
7//!
8//! The `JwtEngine` is responsible for handling JSON Web Tokens (JWTs). It provides
9//! robust functionality to support authentication and authorization flows, including:
10//!
11//! - Fetching and storing decoding keys from a JSON Web Key Set (JWKS) provided by
12//!   Identity Providers (IDPs).
13//! - Extracting and processing claims from JWTs.
14//! - Validating JWT signatures to ensure token integrity and authenticity.
15//! - Verifying token validity based on standard claims such as expiration (`exp`) and
16//!   audience (`aud`).
17//!
18//! ## Initialization
19//!
20//! The behavior of the `JwtEngine` is determined by parameters passed to [`JwtService::new`].
21//! These parameters are primarily configured via the [`jwt_config`] argument:
22//!
23//! - **JWKS (Optional)**: A JWKS string can be provided through the
24//!   `CEDARLING_LOCAL_JWKS` bootstrap property.
25//! - **Signature Validation**: JWT signature verification is supported using a wrapper
26//!   around the [`jsonwebtoken`] crate.
27//! - **Status Validation (WIP)**: Support for token status validation is being
28//!   developed in accordance with the [`IETF draft spec`].
29//! - **Algorithm Restrictions**: Only tokens signed using supported algorithms will
30//!   be validated. Tokens with unsupported algorithms will trigger a warning.
31//!
32//! Additionally, you can provide a list of **trusted issuers** during initialization.
33//! Only tokens issued by these trusted issuers, as defined in the [`policy store`],
34//! should be validated.
35//!
36//! [`jwt_config`]: JwtConfig
37//! [`policy store`]: crate::common::policy_store::PolicyStore
38//! [`IETF draft spec`]: https://www.ietf.org/archive/id/draft-ietf-oauth-status-list-10.html
39//!
40//! ## Usage
41//!
42//! The primary interface for token validation is [`JwtService::validate_tokens`].
43//!
44//! This method accepts a [`HashMap<String, String>`] representing the tokens, typically
45//! passed in through [`Cedarling::authorize`]. Each entry in the map is a
46//! `(token_name, jwt_string)` pair.
47//!
48//! Only tokens that:
49//! - match a trusted issuer defined in the [`policy store`], and
50//! - have a matching token name,
51//!
52//! will be processed. Untrusted tokens are ignored with a warning.
53//!
54//! If any token is invalid (e.g., malformed, expired, or fails validation), the
55//! method returns an error. Successfully validated tokens are returned in a `HashMap`
56//! keyed by token name.
57//!
58//! [`Cedarling::authorize`]: crate::Cedarling::authorize
59//!
60//! ## Security Features
61//!
62//! - [x] Only Accept tokens defined from the policy store
63//!   ones expire.
64//! - [x] Statuslist Check: The `status` claim of a JWT should be validated if present.
65//!   This is done through the [`status_list`] crate for the implementation.
66//! - [ ] JWK rotation (WIP): The service should automatically fetch new keys if the old

123    /// The [`TrustedIssuer`] config loaded from the policy store

129impl JwtService {
130    pub async fn new(
131        jwt_config: &JwtConfig,
132        trusted_issuers: Option<HashMap<String, TrustedIssuer>>,
133        logger: Option<Logger>,
134        token_cache_max_ttl_sec: usize,
135    ) -> Result<Self, JwtServiceInitError> {
136        let mut status_lists = StatusListCache::default();
137        let mut issuer_configs = HashMap::default();
138        let mut validators = JwtValidatorCache::default();
139        let mut key_service = KeyService::new();
140
141        for (issuer_id, iss) in trusted_issuers.unwrap_or_default().into_iter() {
142            // this is what we expect to find in the JWT `iss` claim
143            let mut iss_claim = iss.oidc_endpoint.origin().ascii_serialization();
144
145            let mut iss_config = IssuerConfig {
146                issuer_id,
147                policy: Arc::new(iss),
148                openid_config: None,
149            };
150
151            if jwt_config.jwt_sig_validation || jwt_config.jwt_status_validation {
152                iss_claim = update_openid_config(&mut iss_config, &logger).await?;
153            }
154
155            insert_keys(&mut key_service, jwt_config, &iss_config, &logger).await?;
156
157            validators.init_for_iss(&iss_config, jwt_config, &status_lists, logger.clone());
158
159            if jwt_config.jwt_status_validation {
160                status_lists
161                    .init_for_iss(&iss_config, &validators, &key_service, logger.clone())
162                    .await?;
163            }
164
165            issuer_configs.insert(normalize_issuer(&iss_claim), iss_config);
166        }
167
168        // quick check so we don't get surprised if the program runs but can't validate
169        // anything
170        if !key_service.has_keys() && jwt_config.jwt_sig_validation {
171            return Err(JwtServiceInitError::KeyServiceMissingKeys);
172        }
173        let key_service = Arc::new(key_service);
174
175        Ok(Self {
176            validators,
177            key_service,
178            issuer_configs,
179            logger,
180            token_cache: Arc::new(RwLock::new(SparKV::new())),
181            token_cache_max_ttl: token_cache_max_ttl_sec,
182        })
183    }

185    pub async fn validate_tokens<'a>(
186        &'a self,
187        tokens: &'a HashMap<String, String>,
188    ) -> Result<HashMap<String, Arc<Token>>, JwtProcessingError> {
189        let mut validated_tokens = HashMap::new();
190        const ID_TOKEN_NAME: &str = "id_token";
191
192        let now = Utc::now();
193
194        // clear expired tokens from cache
195        self.token_cache
196            .write()
197            .expect("validated_jwt_cache mutex shouldn't be poisoned")
198            .clear_expired();
199
200        for (token_name, jwt) in tokens.iter() {
201            let token_kind = TokenKind::AuthzRequestInput(token_name);
202            let token = if let Some(validated_token) = self.find_token_in_cache(&token_kind, jwt) {
203                validated_token
204            } else {
205                // validate token and save to cache
206                let validated_jwt = match self.validate_single_token(token_kind, jwt) {
207                    Ok(jwt) => jwt,
208                    Err(err) => {
209                        if matches!(err, ValidateJwtError::MissingValidator(_)) {
210                            self.logger
211                                .log_any(JwtLogEntry::new(err.to_string(), Some(LogLevel::WARN)));
212                            continue;
213                        } else {
214                            return Err(JwtProcessingError::ValidateJwt(token_name.clone(), err));
215                        }
216                    },
217                };
218
219                let mut claims = serde_json::from_value::<TokenClaims>(validated_jwt.claims)
220                    .map_err(|err| {
221                        self.logger.log_any(JwtLogEntry::new(
222                            format!("failed to deserialize token claims: {err}"),
223                            Some(LogLevel::ERROR),
224                        ));
225                        err
226                    })
227                    .map_err(JwtProcessingError::StringDeserialization)?;
228
229                if token_name == ID_TOKEN_NAME {
230                    claims = fix_aud_claim_value_to_array(claims);
231                };
232
233                let token = Arc::new(Token::new(token_name, claims, validated_jwt.trusted_iss));
234                self.save_token_in_cache(&token_kind, jwt, token.clone(), now);
235                token
236            };
237
238            validated_tokens.insert(token_name.to_string(), token);
239        }
240
241        Ok(validated_tokens)
242    }

244    fn find_token_in_cache(&self, kind: &TokenKind, jwt: &str) -> Option<Arc<Token>> {
245        let key = hash_jwt_token(kind, jwt);
246        self.token_cache
247            .read()
248            .expect("validated_jwt_cache mutex shouldn't be poisoned")
249            .get(&key)
250            .map(|v| v.to_owned())
251    }
252
253    fn save_token_in_cache(
254        &self,
255        kind: &TokenKind,
256        jwt: &str,
257        token: Arc<Token>,
258        now: DateTime<Utc>,
259    ) {
260        let key = hash_jwt_token(kind, jwt);
261
262        let cache_duration_opt = token
263            .claims
264            .get_claim("exp")
265            .and_then(|exp| exp.value().as_i64())
266            .and_then(|exp| {
267                // calculate duration until token expiration
268                let duration = exp - now.timestamp();
269                if duration > 0 {
270                    // if duration bigger than configured max ttl, use the max ttl
271                    Some(
272                        if self.token_cache_max_ttl > 0
273                            && duration > self.token_cache_max_ttl as i64
274                        {
275                            self.token_cache_max_ttl as i64
276                        } else {
277                            duration
278                        },
279                    )
280                } else {
281                    None
282                }
283            })
284            .or({
285                // if no exp claim, use the configured max ttl if set
286                if self.token_cache_max_ttl > 0 {
287                    Some(self.token_cache_max_ttl as i64)
288                } else {
289                    None
290                }
291            });
292
293        if let Some(duration) = cache_duration_opt {
294            let _ = self
295                .token_cache
296                .write()
297                .expect("validated_jwt_cache mutex shouldn't be poisoned")
298                .set_with_ttl(&key, token, Duration::seconds(duration), &[]);
299        } else {
300            // set with SparkKV default TTL (5 minutes)
301            let _ = self
302                .token_cache
303                .write()
304                .expect("validated_jwt_cache mutex shouldn't be poisoned")
305                .set(&key, token, &[]);
306        }
307    }

309    fn validate_single_token(
310        &self,
311        token_kind: TokenKind,
312        jwt: &str,
313    ) -> Result<ValidatedJwt, ValidateJwtError> {
314        let decoded_jwt = decode_jwt(jwt)?;
315
316        // Get decoding key
317        let decoding_key_info = decoded_jwt.decoding_key_info();
318        let decoding_key = self.key_service.get_key(&decoding_key_info);
319
320        // get validator
321        let normalized_iss = decoded_jwt.iss().map(normalize_issuer);
322        let validator_key = ValidatorInfo {
323            iss: normalized_iss.as_deref(),
324            token_kind,
325            algorithm: decoded_jwt.header.alg,
326        };
327        let validator: Arc<RwLock<JwtValidator>> = self
328            .validators
329            .get(&validator_key)
330            .ok_or(ValidateJwtError::MissingValidator(validator_key.owned()))?;
331
332        // validate JWT
333        // NOTE: the JWT will be validated depending on the validator's settings that
334        // was set on initialization
335        let mut validated_jwt = {
336            validator
337                .read()
338                .expect("acquire JwtValidator read lock")
339                .validate_jwt(jwt, decoding_key)?
340        };
341
342        // The users of the validated JWT will need a reference to the TrustedIssuer
343        // to do some processing so we include it here for convenience
344        validated_jwt.trusted_iss = decoded_jwt.iss().and_then(|iss| self.get_issuer_ref(iss));
345
346        Ok(validated_jwt)
347    }

349    /// Validate multiple tokens from different issuers
350    ///
351    /// This method validates JWT tokens from multiple issuers, checking for:
352    /// - JWT signature validation
353    /// - Token expiration and other standard claims
354    /// - Non-deterministic token detection (duplicate issuer+type combinations)
355    ///
356    /// Returns a result containing validated tokens or detailed error information.
357    pub fn validate_multi_issuer_tokens(
358        &self,
359        tokens: &[TokenInput],
360    ) -> Result<HashMap<String, Arc<Token>>, MultiIssuerValidationError> {
361        if tokens.is_empty() {
362            return Err(MultiIssuerValidationError::EmptyTokenArray);
363        }
364
365        let mut validated_tokens = HashMap::new();
366        let mut seen_combinations = HashSet::new();
367
368        // clear expired tokens from cache
369        self.token_cache
370            .write()
371            .expect("validated_jwt_cache mutex shouldn't be poisoned")
372            .clear_expired();
373
374        let now = Utc::now();
375
376        for (index, token) in tokens.iter().enumerate() {
377            // Basic validation first
378            if let Err(err) = token.validate() {
379                if let Some(logger) = &self.logger {
380                    logger.log_any(JwtLogEntry::new(
381                        format!("Token validation failed at index {}: {}", index, err),
382                        Some(LogLevel::WARN),
383                    ));
384                }
385                continue;
386            }
387
388            // Find the corresponding token metadata key for the entity type name
389            let token_type = self.find_token_metadata_key(&token.mapping);
390
391            let token_kind = TokenKind::AuthorizeMultiIssuer(token_type);
392            // Create Token with the mapping as the name
393            let token_name = token.mapping.clone();
394
395            if let Some(cedar_token) = self.find_token_in_cache(&token_kind, &token.payload) {
396                validated_tokens.insert(token_name, cedar_token);
397            } else {
398                // Validate JWT using existing single token validation
399                match self.validate_single_token(token_kind, &token.payload) {
400                    Ok(validated_jwt) => {
401                        // Extract issuer for non-deterministic check
402                        let issuer = validated_jwt
403                            .claims
404                            .get("iss")
405                            .and_then(|iss| iss.as_str())
406                            .ok_or(MultiIssuerValidationError::MissingIssuer)?;
407
408                        // Check for non-deterministic tokens (graceful validation)
409                        let combination = format!("{}:{}", issuer, token.mapping);
410                        if !seen_combinations.insert(combination.clone()) {
411                            // Log warning but continue processing
412                            if let Some(logger) = &self.logger {
413                                logger.log_any(JwtLogEntry::new(
414                                format!(
415                                    "Non-deterministic token detected: type '{}' from issuer '{}' (duplicate found, skipping)",
416                                    token.mapping, issuer
417                                ),
418                                Some(LogLevel::WARN),
419                            ));
420                            }
421                        } else {
422                            // Convert ValidatedJwt to Token
423                            let claims =
424                                serde_json::from_value::<TokenClaims>(validated_jwt.claims)
425                                    .map_err(|err| {
426                                        if let Some(logger) = &self.logger {
427                                            logger.log_any(JwtLogEntry::new(
428                                                format!(
429                                                    "failed to deserialize token claims: {err}"
430                                                ),
431                                                Some(LogLevel::ERROR),
432                                            ));
433                                        }
434                                        MultiIssuerValidationError::TokenValidationFailed
435                                    })?;
436
437                            let cedar_token = Arc::new(Token::new(
438                                &token_name,
439                                claims,
440                                validated_jwt.trusted_iss,
441                            ));
442
443                            self.save_token_in_cache(
444                                &token_kind,
445                                &token.payload,
446                                cedar_token.clone(),
447                                now,
448                            );
449
450                            validated_tokens.insert(token_name, cedar_token);
451                        }
452                    },
453                    Err(err) => {
454                        if let Some(logger) = &self.logger {
455                            logger.log_any(JwtLogEntry::new(
456                                format!("Token validation failed at index {}: {}", index, err),
457                                Some(LogLevel::WARN),
458                            ));
459                        }
460                    },
461                }
462            }
463        }
464
465        // If no tokens were successfully validated, return a detailed error
466        if validated_tokens.is_empty() {
467            if let Some(logger) = &self.logger {
468                logger.log_any(JwtLogEntry::new(
469                    "No valid tokens found in multi-issuer request".to_string(),
470                    Some(LogLevel::ERROR),
471                ));
472            }
473
474            return Err(MultiIssuerValidationError::TokenValidationFailed);
475        }
476
477        Ok(validated_tokens)
478    }

480    /// Use the `iss` claim of a token to retrieve a reference to a [`TrustedIssuer`]
481    #[inline]
482    fn get_issuer_ref(&self, iss_claim: &str) -> Option<Arc<TrustedIssuer>> {
483        self.issuer_configs
484            .get(&normalize_issuer(iss_claim))
485            .map(|config| config.policy.clone())
486    }
487
488    /// Find the token metadata key for a given entity type name
489    /// e.g., "Dolphin::Access_Token" -> "access_token"
490    fn find_token_metadata_key<'a>(&'a self, entity_type_name: &'a str) -> &'a str {
491        // Look through all trusted issuers to find the matching entity type name
492        for issuer_config in self.issuer_configs.values() {
493            for (token_key, token_metadata) in &issuer_config.policy.token_metadata {
494                if token_metadata.entity_type_name == entity_type_name {
495                    return token_key;
496                }
497            }
498        }
499
500        // If not found, return the original mapping (fallback)
501        entity_type_name
502    }

505async fn update_openid_config(
506    iss_config: &mut IssuerConfig,
507    logger: &Option<Logger>,
508) -> Result<String, JwtServiceInitError> {
509    let openid_config = OpenIdConfig::get_from_url(&iss_config.policy.oidc_endpoint)
510        .await
511        .inspect_err(|e| {
512            logger.log_any(JwtLogEntry::new(
513                format!(
514                    "failed to get openid configuration for trusted issuer: '{}': {}",
515                    iss_config.issuer_id, e
516                ),
517                Some(LogLevel::ERROR),
518            ));
519        })?;
520
521    let iss_claim = openid_config.issuer.clone();
522    iss_config.openid_config = Some(openid_config);
523
524    Ok(iss_claim)
525}

527async fn insert_keys(
528    key_service: &mut KeyService,
529    jwt_config: &JwtConfig,
530    iss_config: &IssuerConfig,
531    logger: &Option<Logger>,
532) -> Result<(), KeyServiceError> {
533    if !jwt_config.jwt_sig_validation {
534        return Ok(());
535    }
536
537    if let Some(jwks) = jwt_config.jwks.as_ref() {
538        key_service.insert_keys_from_str(jwks)?;
539    }
540
541    if let Some(openid_config) = iss_config.openid_config.as_ref() {
542        key_service
543            .get_keys_using_oidc(openid_config, logger)
544            .await?;
545    }
546
547    Ok(())
548}

550// Fix String `aud` claim value to array
551fn fix_aud_claim_value_to_array(claims: TokenClaims) -> TokenClaims {
552    // make owned value mutable
553    let mut claims = claims;
554
555    const AUD_KEY: &str = "aud";
556    let mut aud_value = serde_json::Value::Null;
557
558    if let Some(claim) = claims.get_claim(AUD_KEY) {
559        if let Some(claim_str_value) = claim.value().as_str() {
560            // convert String to Array for backward compatibility
561            aud_value = json!([claim_str_value]);
562        } else {
563            aud_value = claim.value().clone()
564        }
565    }
566
567    if aud_value != serde_json::Value::Null {
568        claims = claims.with_claim(AUD_KEY.to_string(), aud_value)
569    }
570
571    claims
572}

574/// Hash a string using `ahash` and return the hash value as a string
575/// This is used to create a key for caching tokens
576/// The hash value is used instead of the original string to have shorter keys for SparKV which utilizes BTree.
577fn hash_jwt_token(kind: &TokenKind, jwt: &str) -> String {
578    use core::hash::BuildHasher;
579    use std::hash::Hasher;
580    use std::sync::LazyLock;
581    static HASHER_KEYS: LazyLock<(u64, u64, u64, u64)> = LazyLock::new(|| {
582        (
583            rand::random(),
584            rand::random(),
585            rand::random(),
586            rand::random(),
587        )
588    });
589
590    let mut hasher =
591        ahash::RandomState::with_seeds(HASHER_KEYS.0, HASHER_KEYS.1, HASHER_KEYS.2, HASHER_KEYS.3)
592            .build_hasher();
593
594    kind.hash(&mut hasher);
595    jwt.hash(&mut hasher);
596
597    hasher.finish().to_string()
598}
