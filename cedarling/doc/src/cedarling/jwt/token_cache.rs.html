16/// A dedicated cache for storing validated JWT tokens.
17///
18/// The cache uses a thread-safe key-value store with automatic expiration
19/// based on token expiration claims and a configurable maximum TTL.
20#[derive(Clone)]
21pub struct TokenCache {
22    cache: Arc<RwLock<SparKV<Arc<Token>>>>,
23    max_ttl: usize,
24    logger: Option<Logger>,
25}
26
27#[cfg(test)]
28impl Default for TokenCache {
29    fn default() -> Self {
30        // default parameters, is used only for testing
31
32        use crate::log::TEST_LOGGER;
33        Self::new(60 * 5, 100, true, Some(TEST_LOGGER.clone()))
34    }
35}
36
37impl TokenCache {
38    /// Creates a new `TokenCache` with the specified maximum TTL in seconds.
39    ///
40    /// If `max_ttl` is set to 0, tokens will use their natural expiration or default TTL.
41    pub fn new(
42        max_ttl: usize,
43        capacity: usize,
44        earliest_expiration_eviction: bool,
45        logger: Option<Logger>,
46    ) -> Self {
47        Self {
48            cache: Arc::new(RwLock::new(SparKV::with_config(Config {
49                max_ttl: Duration::seconds(max_ttl as i64),
50                max_items: capacity,
51                earliest_expiration_eviction,
52                ..Default::default()
53            }))),
54            max_ttl,
55            logger,
56        }
57    }
58
59    fn log_warn(&self, msg: String) {
60        if let Some(logger) = &self.logger {
61            logger.log_any(
62                LogEntry::new(BaseLogEntry::new_system_opt_request_id(LogLevel::WARN, None))
63                    .set_message(msg),
64            );
65        }
66    }
67
68    /// Attempts to find a token in the cache.
69    ///
70    /// Returns `Some(Arc<Token>)` if the token is found in cache and not expired,
71    /// otherwise returns `None`.
72    pub fn find(&self, kind: &TokenKind, jwt: &str) -> Option<Arc<Token>> {
73        let key = hash_jwt_token(kind, jwt);
74        self.cache
75            .read()
76            .expect("token cache mutex shouldn't be poisoned")
77            .get(&key)
78            .map(|v| v.to_owned())
79    }
80
81    /// Saves a token to the cache with appropriate TTL.
82    ///
83    /// The TTL is calculated based on:
84    /// 1. The token's expiration claim (if present and valid)
85    /// 2. The configured maximum TTL (if set)
86    /// 3. Default TTL (5 minutes) if neither applies
87    pub fn save(&self, kind: &TokenKind, jwt: &str, token: Arc<Token>, now: DateTime<Utc>) {
88        if self.check_token_expired(&token, now) {
89            // token is expired, no need to save it
90            return;
91        }
92
93        let key = hash_jwt_token(kind, jwt);
94
95        // Extract issuer for indexing
96        let index_keys = token
97            .extract_normalized_issuer()
98            .map(|iss| vec![IndexKey::Iss(iss).index_value()])
99            .unwrap_or_default();
100
101        let result = if let Some(duration) = self.cache_duration(&token, now) {
102            self.cache
103                .write()
104                .expect("token cache mutex shouldn't be poisoned")
105                .set_with_ttl(&key, token, Duration::seconds(duration), &index_keys)
106        } else {
107            // set with SparkKV default TTL (5 minutes)
108            self.cache
109                .write()
110                .expect("token cache mutex shouldn't be poisoned")
111                .set(&key, token, &index_keys)
112        };
113        if let Err(err) = result {
114            self.log_warn(format!("could not set token to token cache: {}", err));
115        }
116    }
117
118    /// Check if token is expired
119    /// true - means is expired
120    fn check_token_expired(&self, token: &Arc<Token>, now: DateTime<Utc>) -> bool {
121        token
122            .claims
123            .get_claim("exp")
124            .and_then(|exp| exp.value().as_i64())
125            .map(|exp| exp <= now.timestamp())
126            .unwrap_or(false)
127    }
128
129    /// Extract cache duration, result is optional
130    fn cache_duration(&self, token: &Arc<Token>, now: DateTime<Utc>) -> Option<i64> {
131        token
132            .claims
133            .get_claim("exp")
134            .and_then(|exp| exp.value().as_i64())
135            .and_then(|exp| {
136                // calculate duration until token expiration
137                let duration = exp - now.timestamp();
138                if duration > 0 {
139                    // if duration bigger than configured max ttl, use the max ttl
140                    Some(if self.max_ttl > 0 && duration > self.max_ttl as i64 {
141                        self.max_ttl as i64
142                    } else {
143                        duration
144                    })
145                } else {
146                    None
147                }
148            })
149            .or({
150                // if no exp claim, use the configured max ttl if set
151                if self.max_ttl > 0 {
152                    Some(self.max_ttl as i64)
153                } else {
154                    None
155                }
156            })
157    }
158
159    /// Clears expired tokens from the cache.
160    ///
161    /// This should be called periodically to prevent memory leaks from
162    /// accumulated expired tokens.
163    pub fn clear_expired(&self) {
164        self.cache
165            .write()
166            .expect("token cache mutex shouldn't be poisoned")
167            .clear_expired();
168    }
169
170    /// Remove tokens from cache by index key
171    pub fn invalidate_by_index(&self, index_key: IndexKey) {
172        self.cache
173            .write()
174            .expect("token cache mutex shouldn't be poisoned")
175            .remove_by_index(&index_key.index_value());
176    }
177}

179/// Hash a string using `ahash` and return the hash value as a string
180/// This is used to create a key for caching tokens
181/// The hash value is used instead of the original string to have shorter keys for SparKV which utilizes BTree.
182fn hash_jwt_token(kind: &TokenKind, jwt: &str) -> String {
183    use core::hash::BuildHasher;
184    use std::hash::Hasher;
185    use std::sync::LazyLock;
186    static HASHER_KEYS: LazyLock<(u64, u64, u64, u64)> = LazyLock::new(|| {
187        (
188            rand::random(),
189            rand::random(),
190            rand::random(),
191            rand::random(),
192        )
193    });
194
195    let mut hasher =
196        ahash::RandomState::with_seeds(HASHER_KEYS.0, HASHER_KEYS.1, HASHER_KEYS.2, HASHER_KEYS.3)
197            .build_hasher();
198
199    kind.hash(&mut hasher);
200    jwt.hash(&mut hasher);
201
202    hasher.finish().to_string()
203}

205/// IndexKey is structure that is used for indexing [TokenCache]
206//
207// thiserror is used for usefull string conversation
208#[derive(Debug, derive_more::Display)]
209pub enum IndexKey {
210    #[display("iss:{_0}")]
211    Iss(String),
212}
213
214impl IndexKey {
215    // We use special method to have string representation of index value
216    // because in future it may change, for example applying hash function
217    fn index_value(&self) -> String {
218        self.to_string()
219    }
