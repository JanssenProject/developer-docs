6use crate::common::policy_store::{ClaimMappings, TokenEntityMetadata, TrustedIssuer};
7use serde::Deserialize;
8use serde_json::Value;
9use std::{collections::HashMap, sync::Arc};
10
11/// Structure representing a validated JWT token, used to derive a Cedar token entity.
12/// Make sure to provide a `TrustedIssuer`; otherwise, the `iss` field may not be constructed.
13#[derive(Debug, PartialEq, Clone)]
14pub struct Token {
15    pub name: String,
16    pub iss: Option<Arc<TrustedIssuer>>,
17    pub(crate) claims: TokenClaims,
18}
19
20impl Token {
21    pub fn new(name: &str, claims: TokenClaims, iss: Option<Arc<TrustedIssuer>>) -> Token {
22        Self {
23            name: name.to_string(),
24            iss,
25            claims,
26        }
27    }
28
29    pub fn get_metadata(&self) -> Option<&TokenEntityMetadata> {
30        self.iss.as_ref()?.get_token_metadata(&self.name)
31    }
32
33    pub fn claim_mappings(&self) -> Option<&ClaimMappings> {
34        self.iss.as_ref()?.get_claim_mapping(&self.name)
35    }
36
37    pub fn get_claim(&self, name: &str) -> Option<TokenClaim<'_>> {
38        self.claims.get_claim(name)
39    }
40
41    pub fn get_claim_val(&self, name: &str) -> Option<&Value> {
42        self.claims.claims.get(name)
43    }
44
45    pub fn logging_info(&self, claim: &str) -> HashMap<String, serde_json::Value> {
46        self.claims.logging_info(claim)
47    }
48
49    pub fn claims_value(&self) -> &HashMap<String, Value> {
50        &self.claims.claims
51    }
52}
53
54#[derive(Debug, PartialEq, Default, Deserialize, Clone)]
55pub struct TokenClaims {
56    #[serde(flatten)]
57    claims: HashMap<String, Value>,
58}
59
60impl From<HashMap<String, Value>> for TokenClaims {
61    fn from(claims: HashMap<String, Value>) -> Self {
62        Self { claims }
63    }
64}
65
66impl From<Value> for TokenClaims {
67    fn from(claims: Value) -> Self {
68        let claims = serde_json::from_value(claims).expect("should deserialize claims to hashmap");
69        Self { claims }
70    }
71}
72
73impl TokenClaims {
74    pub fn get_claim(&self, name: &str) -> Option<TokenClaim<'_>> {
75        self.claims.get(name).map(|value| TokenClaim {
76            key: name.to_string(),
77            value,
78        })
79    }
80
81    pub fn logging_info(&self, claim: &str) -> HashMap<String, serde_json::Value> {
82        let claim = if !claim.is_empty() { claim } else { "jti" };
83
84        self.claims
85            .get(claim)
86            .map(|value| HashMap::from([(claim.to_string(), value.clone())]))
87            .unwrap_or_default()
88    }
89
90    // Update TokenClaims claim value by consuming itself.
91    // Consuming allows to be sure that only one instance exist.
92    pub fn with_claim(self, k: String, v: Value) -> Self {
93        let mut claims = self;
94        claims.claims.insert(k, v);
95
96        claims
97    }
98}
99
100pub struct TokenClaim<'a> {
101    key: String,
102    value: &'a serde_json::Value,
103}
104
105impl TokenClaim<'_> {
106    pub fn as_str(&self) -> Result<&str, TokenClaimTypeError> {
107        self.value
108            .as_str()
109            .ok_or(TokenClaimTypeError::type_mismatch(
110                &self.key, "String", self.value,
111            ))
112    }
113
114    pub fn value(&self) -> &Value {
115        self.value
116    }
117}
118
119#[derive(Debug, thiserror::Error, PartialEq)]
120#[error(
121    "type mismatch for token claim '{key}'. expected: '{expected_type}', but found: '{actual_type}'"
122)]
123pub struct TokenClaimTypeError {
124    pub key: String,
125    pub expected_type: String,
126    pub actual_type: String,
127}
128
129impl TokenClaimTypeError {
130    /// Returns the JSON type name of the given value.
131    pub fn json_value_type_name(value: &Value) -> String {
132        match value {
133            Value::Null => "null".to_string(),
134            Value::Bool(_) => "bool".to_string(),
135            Value::Number(_) => "number".to_string(),
136            Value::String(_) => "string".to_string(),
137            Value::Array(_) => "array".to_string(),
138            Value::Object(_) => "object".to_string(),
139        }
140    }
141
142    /// Constructs a `TypeMismatch` error with detailed information about the expected and actual types.
143    pub fn type_mismatch(key: &str, expected_type_name: &str, got_value: &Value) -> Self {
144        let got_value_type_name = Self::json_value_type_name(got_value);
145
146        Self {
147            key: key.to_string(),
148            expected_type: expected_type_name.to_string(),
149            actual_type: got_value_type_name,
150        }
151    }
152}
