15static HTTP_CLIENT: LazyLock<Client> = LazyLock::new(Client::new);
16
17// async_traits are Send by default but wasm-bindgen doesn't support those
18// so we opt out of it for the wasm bindings to compile.
19//
20// see this relevant discussion: https://github.com/rustwasm/wasm-bindgen/issues/2409
21#[async_trait(?Send)]
22pub trait GetFromUrl<T> {
23    /// Send a get request to receive the resource from a URL
24    async fn get_from_url(url: &Url) -> Result<T, HttpError>;
25}
26
27#[derive(Deserialize)]
28pub struct OpenIdConfig {
29    pub issuer: String,
30    #[serde(deserialize_with = "deserialize_url")]
31    pub jwks_uri: Url,
32    #[serde(deserialize_with = "deserialize_opt_url", default)]
33    pub status_list_endpoint: Option<Url>,
34}
35
36pub fn deserialize_url<'de, D>(deserializer: D) -> Result<Url, D::Error>
37where
38    D: Deserializer<'de>,
39{
40    let url_str = String::deserialize(deserializer)?;
41    let url = Url::parse(&url_str)
42        .map_err(|e| de::Error::custom(format!("invalid url '{url_str}': {e}")))?;
43    Ok(url)
44}
45
46pub fn deserialize_opt_url<'de, D>(deserializer: D) -> Result<Option<Url>, D::Error>
47where
48    D: Deserializer<'de>,
49{
50    let url_str = Option::<String>::deserialize(deserializer)?;
51    let url = url_str
52        .as_ref()
53        .map(|url_str| {
54            Url::parse(url_str)
55                .map_err(|e| de::Error::custom(format!("invalid url '{url_str}': {e}")))
56        })
57        .transpose()?;
58    Ok(url)
59}

61#[async_trait(?Send)]
62impl GetFromUrl<OpenIdConfig> for OpenIdConfig {
63    async fn get_from_url(url: &Url) -> Result<Self, HttpError> {
64        let openid_config = HTTP_CLIENT
65            .get(url.as_str())
66            .send()
67            .await
68            .map_err(HttpError::GetRequest)?
69            .error_for_status()
70            .map_err(HttpError::ErrorCode)?
71            .json::<OpenIdConfig>()
72            .await
73            .map_err(HttpError::JsonDeserializeResponse)?;
74
75        Ok(openid_config)
76    }
77}

97// NOTE: we cant use the async_trait here since this is called from another aysnc 
98// function which requires this to be Send.
99impl StatusListJwtStr {
100    pub async fn get_from_url(url: &Url) -> Result<Self, HttpError> {
101        let response = HTTP_CLIENT
102            .get(url.as_str())
103            .header("Content-Type", "application/statuslist+jwt")
104            .send()
105            .await
106            .map_err(HttpError::GetRequest)?
107            .error_for_status()
108            .map_err(HttpError::ErrorCode)?;
109
110        if let Some(content_type) = response.headers().get("Content-Type") {
111            let content_type = content_type
112                .to_str()
113                .map_err(|e| HttpError::InvalidHeader("Content-Type".to_string(), e))?;
114            if content_type != "application/statuslist+jwt" {
115                return Err(HttpError::Unsupported(format!(
116                    "got unsupported status list type, '{0}', from '{1}'. Cedarling currently only supports 'application/statuslist+jwt'",
117                    content_type,
118                    url.as_str(),
119                )));
120            }
121        }
122
123        let status_list_jwt = response.text().await.map_err(HttpError::ReadTextResponse)?;
124
125        Ok(StatusListJwtStr::new(status_list_jwt))
126    }
127}

129#[derive(Debug, thiserror::Error)]
130pub enum HttpError {
131    #[error("failed to complete GET request: {0}")]
132    GetRequest(#[source] reqwest::Error),
133    #[error("received an error response: {0}")]
134    ErrorCode(#[source] reqwest::Error),
135    #[error("failed to deserialize respose from JSON: {0}")]
136    JsonDeserializeResponse(#[source] reqwest::Error),
137    #[error("failed to read the respose text: {0}")]
138    ReadTextResponse(#[source] reqwest::Error),
139    #[error("the value of the '{0}' header is invalid: {1}")]
140    InvalidHeader(String, ToStrError),
141    #[error("{0}")]
142    Unsupported(String),
143}
