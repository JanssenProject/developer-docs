6use crate::common::{issuer_utils::IssClaim, policy_store::TrustedIssuer};
7
8use super::IssuerConfig;
9use ahash::{HashMap, HashMapExt};
10use std::{
11    collections::HashSet,
12    sync::{Arc, RwLock},
13};
14
15const MUTEX_POISONED_ERR: &str =
16    "IssuerIndex RwLock poisoned due to another thread panicking while holding the lock";
17
18/// The value of the `iss` claim from a JWT
19///
20/// An index mapping `iss` claims to their corresponding `IssuerConfig`s
21/// This structure allows efficient lookup of issuer configurations
22///
23/// This structure is thread-safe for concurrent reads and writes
24pub(super) struct IssuerIndex {
25    issuer_idx: RwLock<HashMap<String, Arc<IssuerConfig>>>,
26    iss_index: RwLock<HashMap<IssClaim, Arc<IssuerConfig>>>,
27}
28
29impl IssuerIndex {
30    /// Create a new, empty `IssuerIndex`
31    pub(super) fn new() -> Self {
32        Self {
33            issuer_idx: RwLock::new(HashMap::new()),
34            iss_index: RwLock::new(HashMap::new()),
35        }
36    }
37
38    /// Insert or update an `IssuerConfig` for a given iss claim
39    pub(super) fn insert(&self, iss: IssClaim, config: IssuerConfig) {
40        let issuer_id = config.issuer_id.clone();
41        let rc_config = Arc::new(config);
42
43        self.issuer_idx
44            .write()
45            .expect(MUTEX_POISONED_ERR)
46            .insert(issuer_id, rc_config.clone());
47
48        self.iss_index
49            .write()
50            .expect(MUTEX_POISONED_ERR)
51            .insert(iss, rc_config);
52    }
53
54    /// Get the `TrustedIssuer` for a given iss claim, if it exists
55    pub(super) fn get_trusted_issuer(&self, iss: &IssClaim) -> Option<Arc<TrustedIssuer>> {
56        let index = self.iss_index.read().expect(MUTEX_POISONED_ERR);
57        index.get(iss).map(|config| config.policy.clone())
58    }
59
60    /// Find the token metadata key for a given entity type name
61    /// e.g., "`Dolphin::Access_Token`" -> "`access_token`"
62    pub(super) fn find_token_metadata_key(&self, entity_type_name: &str) -> Option<String> {
63        // Look through all trusted issuers to find the matching entity type name
64
65        // TODO: Optimize this lookup to have O(1) complexity, to have index structure
66        let index = self.iss_index.read().expect(MUTEX_POISONED_ERR);
67        for issuer_config in index.values() {
68            for (token_key, token_metadata) in &issuer_config.policy.token_metadata {
69                if token_metadata.entity_type_name == entity_type_name {
70                    return Some(token_key.clone());
71                }
72            }
73        }
74        None
75    }
76
77    /// Get the number of issuer configurations in the index
78    pub(super) fn len(&self) -> usize {
79        let index = self.iss_index.read().expect(MUTEX_POISONED_ERR);
80        index.len()
81    }
82
83    /// Check if the index contains a specific iss claim
84    pub(super) fn contains_iss(&self, iss: &IssClaim) -> bool {
85        let index = self.iss_index.read().expect(MUTEX_POISONED_ERR);
86        index.contains_key(iss)
87    }
88
89    /// Find the iss claim for a given issuer ID (policy store key)
90    /// Returns `true` if issuer with the given ID is found
91    pub(super) fn is_issuer_id_present(&self, issuer_id: &str) -> bool {
92        self.issuer_idx
93            .read()
94            .expect(MUTEX_POISONED_ERR)
95            .get(issuer_id)
96            .is_some()
97    }
98
99    /// Get all issuer IDs in the index of loaded trusted issuers
100    pub(super) fn loaded_issuer_ids(&self) -> HashSet<String> {
101        let index = self.iss_index.read().expect(MUTEX_POISONED_ERR);
102        index
103            .values()
104            .map(|config| config.issuer_id.clone())
105            .collect()
106    }
107}
