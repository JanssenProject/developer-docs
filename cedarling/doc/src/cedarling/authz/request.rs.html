11/// Box to store authorization data
12#[derive(Debug, Clone, Serialize, Deserialize)]
13pub struct Request {
14    /// Contains the JWTs that will be used for the AuthZ request
15    #[serde(default, deserialize_with = "deserialize_tokens")]
16    pub tokens: HashMap<String, String>,
17    /// cedar_policy action
18    pub action: String,
19    /// cedar_policy resource data
20    pub resource: EntityData,
21    /// context to be used in cedar_policy
22    pub context: Value,
23}

25/// Custom parser for an Option<String> which returns `None` if the string is empty.
26fn deserialize_tokens<'de, D>(deserializer: D) -> Result<HashMap<String, String>, D::Error>
27where
28    D: Deserializer<'de>,
29{
30    let tokens = HashMap::<String, Value>::deserialize(deserializer)?;
31    let (tokens, errs): (Vec<_>, Vec<_>) = tokens
32        .into_iter()
33        .filter_map(|(tkn_name, val)| match val {
34            Value::Null => None,
35            Value::String(token) => Some(Ok((tkn_name, token))),
36            val => Some(Err((tkn_name, value_to_str(&val)))),
37        })
38        .partition(Result::is_ok);
39
40    let tokens: HashMap<String, String> = if errs.is_empty() {
41        tokens.into_iter().flatten().collect()
42    } else {
43        let err_msgs = errs
44            .into_iter()
45            .map(|e| e.unwrap_err())
46            .map(|(tkn_name, got_type)| {
47                format!(
48                    "expected `{}` to be 'string' or 'null' but got '{}'",
49                    tkn_name, got_type
50                )
51            })
52            .collect::<Vec<_>>();
53        return Err(de::Error::custom(format!(
54            "failed to deserialize input tokens: {:?}",
55            err_msgs
56        )));
57    };
58
59    Ok(tokens)
60}
61
62fn value_to_str(value: &Value) -> &'static str {
63    match value {
64        Value::Null => "null",
65        Value::Bool(_) => "bool",
66        Value::Number(_) => "number",
67        Value::String(_) => "string",
68        Value::Array(_) => "array",
69        Value::Object(_) => "object",
70    }
71}

73/// Box to store authorization data, with any additional principals
74#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
75pub struct RequestUnsigned {
76    /// Contains the JWTs that will be used for the AuthZ request
77    pub principals: Vec<EntityData>,
78    /// cedar_policy action
79    pub action: String,
80    /// cedar_policy resource data
81    pub resource: EntityData,
82    /// context to be used in cedar_policy
83    pub context: Value,
84}
85
86/// Cedar policy entity data
87/// fields represent EntityUid
88#[derive(Serialize, Deserialize, Debug, Clone)]
89pub struct EntityData {
90    /// Cedar entity mapping info
91    #[serde(rename = "cedar_entity_mapping")]
92    pub cedar_mapping: CedarEntityMapping,
93    /// entity attributes
94    #[serde(flatten)]
95    pub attributes: HashMap<String, Value>,
96}
97
98/// Cedar entity mapping information
99#[derive(Serialize, Deserialize, Debug, Clone)]
100pub struct CedarEntityMapping {
101    /// entity type name
102    #[serde(rename = "entity_type")]
103    pub entity_type: String,
104    /// entity id
105    pub id: String,
106}
107
108impl EntityData {
109    /// Deserializes a JSON string into [`EntityData`]
110    pub fn from_json(entity_data: &str) -> Result<Self, serde_json::Error> {
111        serde_json::from_str::<Self>(entity_data)
112    }
113}
