14/// Constructs the authorization context by adding the built entities from the tokens
15pub fn build_context(
16    config: &AuthzConfig,
17    request_context: Value,
18    build_entities: &BuiltEntities,
19    schema: &cedar_policy::Schema,
20    action: &cedar_policy::EntityUid,
21) -> Result<cedar_policy::Context, BuildContextError> {
22    let namespace = action.type_name().namespace();
23    let action_name = &action.id().escaped();
24    let json_schema = &config.policy_store.schema.json;
25
26    // TODO: we would to implement a way for the user to decide which entities
27    // should be added to the context that doesn't use the Cedar schema.
28    let action_schema = json_schema
29        .get_action(&namespace, action_name)
30        .ok_or(BuildContextError::UnknownAction(action_name.to_string()))?;
31
32    // Get the entities required for the context
33    let mut ctx_entity_refs = json!({});
34
35    if let Some(ctx) = action_schema.applies_to.context.as_ref() {
36        match ctx {
37            Attribute::Record { attrs, .. } => {
38                for (key, attr) in attrs.iter() {
39                    if let Some(entity_ref) =
40                        build_entity_refs_from_attr(&namespace, attr, build_entities, json_schema)?
41                    {
42                        ctx_entity_refs[key] = entity_ref;
43                    }
44                }
45            },
46            Attribute::EntityOrCommon { name, .. } => {
47                if let Some((_entity_type_name, attr)) = json_schema
48                    .get_common_type(name, Some(&namespace))
49                    .map_err(|err| BuildContextError::ParseEntityName(name.clone(), err))?
50                {
51                    match attr {
52                        Attribute::Record { attrs, .. } => {
53                            for (key, attr) in attrs.iter() {
54                                if let Some(entity_ref) = build_entity_refs_from_attr(
55                                    &namespace,
56                                    attr,
57                                    build_entities,
58                                    json_schema,
59                                )? {
60                                    ctx_entity_refs[key] = entity_ref;
61                                }
62                            }
63                        },
64                        attr => {
65                            return Err(BuildContextError::InvalidKind(
66                                attr.kind_str().to_string(),
67                                "record".to_string(),
68                            ));
69                        },
70                    }
71                }
72            },
73            attr => {
74                return Err(BuildContextError::InvalidKind(
75                    attr.kind_str().to_string(),
76                    "record or common".to_string(),
77                ));
78            },
79        }
80    }
81
82    let context = merge_json_values(request_context, ctx_entity_refs)?;
83    let context: cedar_policy::Context =
84        cedar_policy::Context::from_json_value(context, Some((schema, action)))?;
85
86    Ok(context)
87}

89/// Builds the JSON entity references from a given attribute.
90///
91/// Returns `Ok(None)` if the attr is not an Entity Reference
92fn build_entity_refs_from_attr(
93    namespace: &str,
94    attr: &Attribute,
95    built_entities: &BuiltEntities,
96    schema: &CedarSchemaJson,
97) -> Result<Option<Value>, BuildContextError> {
98    match attr {
99        Attribute::Entity { name, required } => {
100            let Some((type_name, _type_schema)) =
101                schema
102                    .get_entity_schema(name, Some(namespace))
103                    .map_err(|e| BuildContextError::ParseEntityName(name.to_string(), e))?
104            else {
105                return Ok(None);
106            };
107
108            match map_entity_id(&type_name, built_entities) {
109                Ok(res) => Ok(res),
110                Err(err) => {
111                    if *required {
112                        Err(err)
113                    } else {
114                        Ok(None)
115                    }
116                },
117            }
118        },
119        Attribute::EntityOrCommon { name, required } => {
120            let Some((type_name, _type_schema)) =
121                schema
122                    .get_entity_schema(name, Some(namespace))
123                    .map_err(|e| BuildContextError::ParseEntityName(name.to_string(), e))?
124            else {
125                return Ok(None);
126            };
127
128            match map_entity_id(&type_name, built_entities) {
129                Ok(res) => Ok(res),
130                Err(err) => {
131                    if *required {
132                        Err(err)
133                    } else {
134                        Ok(None)
135                    }
136                },
137            }
138        },
139        _ => Ok(None),
140    }
141}

143/// Maps a known entity ID to the entity reference
144fn map_entity_id(
145    name: &EntityTypeName,
146    built_entities: &BuiltEntities,
147) -> Result<Option<Value>, BuildContextError> {
148    if let Some(type_id) = built_entities.get_single(&name.to_smolstr()) {
149        Ok(Some(json!({"type": name.to_string(), "id": type_id})))
150    } else {
151        Err(BuildContextError::MissingEntityId(name.to_string()))
152    }
153}

155#[derive(Debug, thiserror::Error)]
156pub enum BuildContextError {
157    /// Error encountered while validating context according to the schema
158    #[error("failed to merge JSON objects due to conflicting keys: {0}")]
159    KeyConflict(String),
160    /// Error encountered while deserializing the Context from JSON
161    #[error(transparent)]
162    DeserializeFromJson(Box<ContextJsonError>),
163    /// Error encountered if the action being used as the reference to build the Context
164    /// is not in the schema
165    #[error("failed to find the action `{0}` in the schema")]
166    UnknownAction(String),
167    /// Error encountered while building entity references in the Context
168    #[error("failed to build entity reference for `{0}` since an entity id was not provided")]
169    MissingEntityId(String),
170    #[error("invalid action context type: {0}. expected: {1}")]
171    InvalidKind(String, String),
172    #[error("failed to parse the entity name `{0}`: {1}")]
173    ParseEntityName(String, Box<ParseErrors>),
174}
175
176impl From<ContextJsonError> for BuildContextError {
177    fn from(err: ContextJsonError) -> Self {
178        BuildContextError::DeserializeFromJson(Box::new(err))
179    }
180}

182pub fn merge_json_values(mut base: Value, other: Value) -> Result<Value, BuildContextError> {
183    if let (Some(base_map), Some(additional_map)) = (base.as_object_mut(), other.as_object()) {
184        for (key, value) in additional_map {
185            if let Entry::Vacant(entry) = base_map.entry(key) {
186                entry.insert(value.clone());
187            } else {
188                return Err(BuildContextError::KeyConflict(key.clone()));
189            }
190        }
191    }
192    Ok(base)
193}

195#[cfg(test)]
196mod test {
197    use super::*;
198    use serde_json::json;
199
200    #[test]
201    fn can_merge_json_objects() {
202        let obj1 = json!({ "a": 1, "b": 2 });
203        let obj2 = json!({ "c": 3, "d": 4 });
204        let expected = json!({"a": 1, "b": 2, "c": 3, "d": 4});
205
206        let result = merge_json_values(obj1, obj2).expect("Should merge JSON objects");
207
208        assert_eq!(result, expected);
209    }
210
211    #[test]
212    fn errors_on_same_keys() {
213        // Test for only two objects
214        let obj1 = json!({ "a": 1, "b": 2 });
215        let obj2 = json!({ "b": 3, "c": 4 });
216        let result = merge_json_values(obj1, obj2);
217
218        assert!(
219            matches!(result, Err(BuildContextError::KeyConflict(key)) if key.as_str() == "b"),
220            "Expected an error due to conflicting keys"
221        );
222    }
