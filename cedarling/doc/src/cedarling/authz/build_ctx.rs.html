18/// Constructs the authorization context by adding the built entities from the tokens and pushed data.
19///
20/// ## Context Merging
21///
22/// Values are merged in the following order:
23/// 1. **Request Context** - Values passed directly in the authorization request
24/// 2. **Entity References** - Built entity references from tokens
25/// 3. **Pushed Data** - Data from the `DataStore` (injected under `context.data`)
26///
27/// **Important:** Merges will error on key conflicts via `merge_json_values`, which returns
28/// `BuildContextError::KeyConflict` when any key collision occurs. Callers must resolve
29/// conflicts before invoking this function. Pushed data is always placed under the `data`
30/// namespace to avoid conflicts.
31pub(super) fn build_context(
32    config: &AuthzConfig,
33    request_context: Value,
34    build_entities: &BuiltEntities,
35    schema: &cedar_policy::Schema,
36    action: &cedar_policy::EntityUid,
37    pushed_data: &HashMap<String, Value>,
38) -> Result<cedar_policy::Context, BuildContextError> {
39    let namespace = action.type_name().namespace();
40    let action_name = &action.id().escaped();
41    let json_schema = &config.policy_store.schema.json;
42
43    // TODO: we would to implement a way for the user to decide which entities
44    // should be added to the context that doesn't use the Cedar schema.
45    let action_schema = json_schema
46        .get_action(&namespace, action_name)
47        .ok_or(BuildContextError::UnknownAction(action_name.to_string()))?;
48
49    // Get the entities required for the context
50    let mut ctx_entity_refs = json!({});
51
52    if let Some(ctx) = action_schema.applies_to.context.as_ref() {
53        match ctx {
54            Attribute::Record { attrs, .. } => {
55                for (key, attr) in attrs {
56                    if let Some(entity_ref) =
57                        build_entity_refs_from_attr(&namespace, attr, build_entities, json_schema)?
58                    {
59                        ctx_entity_refs[key] = entity_ref;
60                    }
61                }
62            },
63            Attribute::EntityOrCommon { name, .. } => {
64                if let Some((_entity_type_name, attr)) = json_schema
65                    .get_common_type(name, Some(&namespace))
66                    .map_err(|err| BuildContextError::ParseEntityName(name.clone(), err))?
67                {
68                    match attr {
69                        Attribute::Record { attrs, .. } => {
70                            for (key, attr) in attrs {
71                                if let Some(entity_ref) = build_entity_refs_from_attr(
72                                    &namespace,
73                                    attr,
74                                    build_entities,
75                                    json_schema,
76                                )? {
77                                    ctx_entity_refs[key] = entity_ref;
78                                }
79                            }
80                        },
81                        attr => {
82                            return Err(BuildContextError::InvalidKind(
83                                attr.kind_str().to_string(),
84                                "record".to_string(),
85                            ));
86                        },
87                    }
88                }
89            },
90            attr => {
91                return Err(BuildContextError::InvalidKind(
92                    attr.kind_str().to_string(),
93                    "record or common".to_string(),
94                ));
95            },
96        }
97    }
98
99    // Inject pushed data under context.data namespace, merging so key conflicts are surfaced.
100    if !pushed_data.is_empty() {
101        let data_value = Value::Object(pushed_data.iter().map(|(k, v)| (k.clone(), v.clone())).collect());
102        let pushed_wrapper = json!({ "data": data_value });
103        ctx_entity_refs = merge_json_values(ctx_entity_refs, &pushed_wrapper)?;
104    }
105
106    let context = merge_json_values(request_context, &ctx_entity_refs)?;
107    let context: cedar_policy::Context =
108        cedar_policy::Context::from_json_value(context, Some((schema, action)))?;
109
110    Ok(context)
111}

113/// Constructs the authorization context for multi-issuer requests with token collection
114///
115/// This function implements the design document's token collection context structure:
116/// - Individual token entities are accessible via `context.tokens.{issuer}_{token_type}`
117/// - All JWT claims are stored as entity tags (Set of String by default)
118/// - Provides ergonomic policy syntax for cross-token validation
119///
120/// ## Context Merging
121///
122/// Values are merged in the following order:
123/// 1. **Request Context** - Values passed directly in the authorization request
124/// 2. **Token References** - Built token entity references
125/// 3. **Pushed Data** - Data from the `DataStore` (injected under `context.data`)
126///
127/// **Important:** Merges will error on key conflicts via `merge_json_values`, which returns
128/// `BuildContextError::KeyConflict` when any key collision occurs. Callers must resolve
129/// conflicts before invoking this function.
130pub(super) fn build_multi_issuer_context(
131    request_context: Value,
132    token_entities: &HashMap<String, Entity>,
133    schema: &cedar_policy::Schema,
134    action: &cedar_policy::EntityUid,
135    pushed_data: &HashMap<String, Value>,
136) -> Result<cedar_policy::Context, BuildContextError> {
137    // Start with the request context
138    let mut context_json = request_context;
139
140    // Create the tokens context as specified in the design document
141    let mut tokens_context = serde_json::Map::new();
142
143    // Add individual token entity references to context
144    // Format: {issuer}_{token_type} -> {"type": "EntityType", "id": "entity_id"}
145    for (field_name, entity) in token_entities {
146        // Create entity reference like the existing build_context function
147        let entity_ref = json!({
148            "type": entity.uid().type_name().to_string(),
149            "id": entity.uid().id().as_ref() as &str
150        });
151        tokens_context.insert(field_name.clone(), entity_ref);
152    }
153
154    // Add total token count as specified in design
155    tokens_context.insert("total_token_count".to_string(), json!(token_entities.len()));
156
157    // Merge tokens into request context, checking for conflicts
158    let tokens_value = Value::Object(tokens_context);
159    let tokens_wrapper = json!({ "tokens": tokens_value });
160    context_json = merge_json_values(context_json, &tokens_wrapper)?;
161
162    // Inject pushed data under context.data namespace, merging so key conflicts are surfaced.
163    if !pushed_data.is_empty() {
164        let data_value = Value::Object(pushed_data.iter().map(|(k, v)| (k.clone(), v.clone())).collect());
165        let pushed_wrapper = json!({ "data": data_value });
166        context_json = merge_json_values(context_json, &pushed_wrapper)?;
167    }
168
169    // Create Cedar context
170    let context = cedar_policy::Context::from_json_value(context_json, Some((schema, action)))
171        .map_err(|e| BuildContextError::ContextCreation(e.to_string()))?;
172
173    Ok(context)
174}

176/// Builds the JSON entity references from a given attribute.
177///
178/// Returns `Ok(None)` if the attr is not an Entity Reference
179fn build_entity_refs_from_attr(
180    namespace: &str,
181    attr: &Attribute,
182    built_entities: &BuiltEntities,
183    schema: &CedarSchemaJson,
184) -> Result<Option<Value>, BuildContextError> {
185    match attr {
186        Attribute::Entity { name, required } => {
187            let Some((type_name, _type_schema)) =
188                schema
189                    .get_entity_schema(name, Some(namespace))
190                    .map_err(|e| BuildContextError::ParseEntityName(name.clone(), e))?
191            else {
192                return Ok(None);
193            };
194
195            match map_entity_id(&type_name, built_entities) {
196                Ok(res) => Ok(res),
197                Err(err) => {
198                    if *required {
199                        Err(err)
200                    } else {
201                        Ok(None)
202                    }
203                },
204            }
205        },
206        Attribute::EntityOrCommon { name, required } => {
207            let Some((type_name, _type_schema)) =
208                schema
209                    .get_entity_schema(name, Some(namespace))
210                    .map_err(|e| BuildContextError::ParseEntityName(name.clone(), e))?
211            else {
212                return Ok(None);
213            };
214
215            match map_entity_id(&type_name, built_entities) {
216                Ok(res) => Ok(res),
217                Err(err) => {
218                    if *required {
219                        Err(err)
220                    } else {
221                        Ok(None)
222                    }
223                },
224            }
225        },
226        _ => Ok(None),
227    }
228}

230/// Maps a known entity ID to the entity reference
231fn map_entity_id(
232    name: &EntityTypeName,
233    built_entities: &BuiltEntities,
234) -> Result<Option<Value>, BuildContextError> {
235    if let Some(type_id) = built_entities.get_single(&name.to_smolstr()) {
236        Ok(Some(json!({"type": name.to_string(), "id": type_id})))
237    } else {
238        Err(BuildContextError::MissingEntityId(name.to_string()))
239    }
240}

242fn merge_json_values(mut base: Value, other: &Value) -> Result<Value, BuildContextError> {
243    if let (Some(base_map), Some(additional_map)) = (base.as_object_mut(), other.as_object()) {
244        for (key, value) in additional_map {
245            if let Entry::Vacant(entry) = base_map.entry(key) {
246                entry.insert(value.clone());
247            } else {
248                return Err(BuildContextError::KeyConflict(key.clone()));
249            }
250        }
251    }
252    Ok(base)
253}

256mod test {

260    #[test]
261    fn can_merge_json_objects() {
262        let obj1 = json!({ "a": 1, "b": 2 });
263        let obj2 = json!({ "c": 3, "d": 4 });
264        let expected = json!({"a": 1, "b": 2, "c": 3, "d": 4});
265
266        let result = merge_json_values(obj1, &obj2).expect("Should merge JSON objects");
267
268        assert_eq!(result, expected);
269    }
270
271    #[test]
272    fn errors_on_same_keys() {
273        // Test for only two objects
274        let obj1 = json!({ "a": 1, "b": 2 });
275        let obj2 = json!({ "b": 3, "c": 4 });
276        let result = merge_json_values(obj1, &obj2);
277
278        assert!(
279            matches!(result, Err(BuildContextError::KeyConflict(key)) if key.as_str() == "b"),
280            "Expected an error due to conflicting keys"
281        );
282    }
