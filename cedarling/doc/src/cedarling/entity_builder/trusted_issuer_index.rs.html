14/// Fast lookup index for trusted issuers by URL or origin.
15///
16/// This structure maintains indexes for O(1) lookups of trusted issuers by either:
17/// - Full OIDC endpoint URL
18/// - Origin only
19///
20/// # Origin Collision Handling
21///
22/// When multiple issuers share the same origin, the last issuer encountered during construction
23/// will be used for origin-based lookups. A warning is logged when this occurs. Full URL lookups
24/// remain unaffected and will correctly resolve to each specific issuer.
25#[derive(Debug, Clone)]
26pub(crate) struct TrustedIssuerIndex {
27    /// Index mapping full URL to trusted issuer
28    url_index: HashMap<String, Arc<TrustedIssuer>>,
29    /// Index mapping origin to trusted issuer
30    origin_index: HashMap<String, Arc<TrustedIssuer>>,
31}
32
33impl TrustedIssuerIndex {
34    pub(crate) fn new(issuers: &HashMap<String, TrustedIssuer>, logger: Option<&Logger>) -> Self {
35        let mut origin_index: HashMap<String, Arc<TrustedIssuer>> = HashMap::new();
36        let mut url_index = HashMap::new();
37
38        for iss in issuers.values() {
39            let origin = iss.oidc_endpoint.origin().ascii_serialization();
40            let full_url = iss.oidc_endpoint.as_str();
41            let issuer = Arc::new(iss.clone());
42
43            if let Some(existing) = origin_index.get(&origin) {
44                logger.log_any(
45                    LogEntry::new(BaseLogEntry::new_system_opt_request_id(LogLevel::WARN, None))
46                        .set_message(format!(
47                            "Duplicate origin '{}': issuer '{}' will override existing issuer '{}' for origin-based lookups",
48                            origin,
49                            iss.name,
50                            existing.name,
51                        )),
52                );
53            }
54            origin_index.insert(origin, issuer.clone());
55            url_index.insert(full_url.to_string(), issuer);
56        }
57
58        Self {
59            url_index,
60            origin_index,
61        }
62    }
63
64    /// Finds a trusted issuer by URL, checking full URL first, then origin.
65    pub(super) fn find(&self, url: &str) -> Option<&Arc<TrustedIssuer>> {
66        self.url_index
67            .get(url)
68            .or_else(|| self.origin_index.get(url))
69    }
70
71    /// Returns an iterator over references to all trusted issuers.
72    pub(super) fn values(&self) -> impl Iterator<Item = &TrustedIssuer> + '_ {
73        self.url_index.values().map(AsRef::as_ref)
74    }
75}

77#[cfg(test)]
78mod tests {
79    use super::*;
80    use crate::common::policy_store::TrustedIssuer;
81    use std::collections::HashMap;
82    use url::Url;
83
84    fn create_issuer_index() -> TrustedIssuerIndex {
85        let mut trusted_issuers = HashMap::new();
86
87        let issuer_one = TrustedIssuer {
88            name: "Issuer One".to_string(),
89            description: "Issuer One".to_string(),
90            oidc_endpoint: Url::parse("https://issuer1.example.com/auth").unwrap(),
91            token_metadata: HashMap::new(),
92        };
93        trusted_issuers.insert("issuer_one".to_string(), issuer_one);
94
95        let jans = TrustedIssuer {
96            name: "Jans".to_string(),
97            description: "Janssen".to_string(),
98            oidc_endpoint: Url::parse("https://account.gluu.org/.well-known/openid-configuration")
99                .unwrap(),
100            token_metadata: HashMap::new(),
101        };
102        trusted_issuers.insert("jans".to_string(), jans);
103
104        let microsoft_issuer = TrustedIssuer {
105            name: "Microsoft".to_string(),
106            description: "Microsoft Azure AD".to_string(),
107            oidc_endpoint: Url::parse("https://login.microsoftonline.com/tenant").unwrap(),
108            token_metadata: HashMap::new(),
109        };
110        trusted_issuers.insert("microsoft".to_string(), microsoft_issuer);
111
112        let company_issuer = TrustedIssuer {
113            name: "Company".to_string(),
114            description: "Company Internal Auth".to_string(),
115            oidc_endpoint: Url::parse("https://auth.company.internal:8443/oauth").unwrap(),
116            token_metadata: HashMap::new(),
117        };
118        trusted_issuers.insert("company".to_string(), company_issuer);
119
120        TrustedIssuerIndex::new(&trusted_issuers, None)
121    }
122
123    #[test]
124    fn test_empty_issuer_index() {
125        let issuer_index = TrustedIssuerIndex::new(&HashMap::new(), None);
126        assert!(
127            issuer_index
128                .find("https://login.microsoftonline.com/tenant")
129                .is_none()
130        );
131    }
132
133    #[test]
134    fn test_find_non_existing_issuer() {
135        let issuer_index = create_issuer_index();
136        assert!(issuer_index.find("https://account.google.com").is_none());
137    }
138
139    #[test]
140    fn test_find_by_full_url() {
141        let issuer_index = create_issuer_index();
142        let issuer = issuer_index
143            .find("https://login.microsoftonline.com/tenant")
144            .expect("Should find issuer by full URL");
145
146        assert_eq!(issuer.name, "Microsoft");
147        assert_eq!(
148            issuer.oidc_endpoint.as_str(),
149            "https://login.microsoftonline.com/tenant"
150        );
151    }
152
153    #[test]
154    fn test_find_by_origin() {
155        let issuer_index = create_issuer_index();
156        let issuer = issuer_index
157            .find("https://account.gluu.org")
158            .expect("Should find issuer by origin");
159
160        assert_eq!(issuer.name, "Jans");
161        assert_eq!(
162            issuer.oidc_endpoint.as_str(),
163            "https://account.gluu.org/.well-known/openid-configuration"
164        );
165    }
166
167    #[test]
168    fn test_find_by_origin_with_port() {
169        let issuer_index = create_issuer_index();
170        let issuer = issuer_index

179    }
